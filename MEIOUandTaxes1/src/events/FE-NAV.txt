	id = hee_NAV.1
	title = hee_NAV.1.t
	desc = hee_NAV.1.d
	#TODO: this should be checking balance of power between ARA and NAV, not assuming NAV is weak
	picture = GOOD_WITH_MONARCH_eventPicture
	fire_only_once = yes
	major = yes
	trigger = {
		NOT = { dip_tech = 31 }
		tag = NAV

		is_lesser_in_union = no
		OR = {
			is_subject = no
			is_subject_of_type = tributary_state
			is_subject_of_type = close_tributary_state
		}

		ARA = {
			is_lesser_in_union = no
			OR = {
				is_subject = no
				is_subject_of_type = tributary_state
				is_subject_of_type = close_tributary_state
			}
		}

		has_opinion = {
			who = ARA
			value = -10
		}

		OR = {
			AND = {
				is_female = yes
				ARA = {	is_female = no }
			}
			AND = {
				OR = {
					has_female_heir = yes
					has_heir = no
				}
				ARA = {
					has_heir = yes
					has_female_heir = no
				}
			}
			AND = {
				is_female = yes
				ARA = {
					has_female_heir = no
					heir_age = 15
				}
			}
		}
	}
	mean_time_to_happen = {
		years = 7
	}
	option = {
		name = hee_NAV.1.a
		log = "FE-NAV:[GetYear]:[Root.GetName]:offered to become junior partner of [ARA.GetName]:hee_NAV.1"
		ai_chance = {
			factor = 90
		}
		ARA = {
			country_event = {
				id = hee_NAV.2
			}
		}
	}
	option = {
		name = hee_NAV.1.b
		log = "FE-NAV:[GetYear]:[Root.GetName]:have not offered to become junior partner of [ARA.GetName]:hee_NAV.1"
		ai_chance = {
			factor = 10
		}
		add_legitimacy = -5
		add_prestige = 5
	}
}

### THE CIUDADELA DE PAMPLONA ###

country_event = {
	id = hee_NAV.11 #The Ciudadela de Pamplona
	title = hee_NAV.11.t
	desc = hee_NAV.11.d
	picture = FORT_eventPicture
	fire_only_once = yes

	trigger = {
		owns = 210 #Navarre
		mil_tech = 37
		NOT = { is_year = 1630 }
		OR = {
			tag = NAV
			AND = {
				tag = SPA
				NOT = {	exists = NAV }
			}
			AND = {
				tag = CAS
				NOT = {	exists = NAV }
			}
			AND = {
				tag = ARA
				NOT = {	exists = NAV }
			}
		}
		exists = FRA
		NOT = {
			army_strength = {
				who = FRA
				value = 0.66
			}
		}
		NOT = {	alliance_with = FRA }
		NOT = {
			has_opinion = {
				who = FRA
				value = 50
			}
		}
	}
	mean_time_to_happen = {
		years = 25
	}
	option = {
		name = hee_NAV.11.a
		ai_chance = {
			factor = 90
			modifier = {
				factor = 0
				OR = {
					num_of_loans = 2
					NOT = { treasury = 125 }
				}
			}
		}
		add_treasury = -100
		210 = {
			add_province_modifier = {
				name = hee_ciudadela_pamplona
				duration = -1
			}
		}
		if = {
			limit = { NOT = { 210 = { fort_level = 2 } } }
			210 = {
				add_building = fort_15th
			}
		}
	}
	option = {
		name = hee_NAV.11.b
		ai_chance = {
			factor = 10
		}
		add_mil_power = 20
	}
}
