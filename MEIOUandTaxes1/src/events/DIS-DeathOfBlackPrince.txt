#Alice Perrers
country_event = {
	id = death_of_black_prince.10
	title = "death_of_black_prince.10.t"
	desc = "death_of_black_prince.10.d"
	picture = DIPLOMACY_eventPicture
	
	is_triggered_only = yes
	trigger = {
		tag = ENG
		has_ruler = "Edward III"
		OR = {
			had_country_flag = {
				flag = alice_perrers
				days = 180
			}
			NOT = { has_country_flag = alice_perrers }
		}
	}

	option = {
		name = "death_of_black_prince.10.a"
		if = {
			limit = {
				NOT = { has_country_flag = alice_perrers } 
			}
			set_country_flag = alice_perrers
		}
		
		add_legitimacy = -10
		Public_ChangePowerbrokerLoyaltyTooltipProv = {
			Powerbroker = BU
			Amount = -5
		}
		Public_ChangePowerbrokerLoyaltyTooltipProv = {
			Powerbroker = NO
			Amount = -5
		}
		add_corruption = 5
		add_treasury = -100
		random_owned_province = {
			Public_ChangeClassWealthVal = { class= NO type = value inp= 100 }
		}
	}
}

		name = "death_of_black_prince.11.c"

#Richard II
country_event = {
	id = death_of_black_prince.100
	title = "death_of_black_prince.100.t"
	desc = "death_of_black_prince.100.d"
	picture = DIPLOMACY_eventPicture
	
	is_triggered_only = yes

	option = {
		name = "death_of_black_prince.100.a"
		if = {
			limit = {
				has_ruler = "Edward IV 'the Black Prince'"
			}

			kill_heir = {
				allow_new_heir = no
			}
			
			define_heir = {
				male = yes
				dynasty = ROOT
				name = "Richard"
				age = 9
				claim = 100
				block_disinherit = yes
			}

			set_ruler_flag = richard_ii
		}
	}
}
