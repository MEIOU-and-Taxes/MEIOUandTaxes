			}

				who = PREV

			}

		name = "flavor_rus.EVTOPTB1106"
		log = "FE-TimeOfTroubles:[GetYear]:[Root.GetName]:First Dimitri was expelled:time_of_troubles.6"

			factor = 50
		}

		if = {
			limit = {
				NOT = { has_country_flag = dimitri_supporter_found }
			}
			random_enemy_country = {
				limit = {
				    is_neighbor_of = ROOT
					OR = {
						is_rival = ROOT
						NOT = {
							has_opinion = {

								value = 0
