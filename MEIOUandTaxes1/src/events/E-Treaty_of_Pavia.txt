	title = "treaty_pavia.1000.t"

	is_triggered_only = yes
	hidden = yes
	trigger = {
		NOT = {	has_country_flag = treaty_of_pavia_ignored }

			tag = BAV
			tag = BAW
			tag = BAX
			tag = BAY
			tag = BAZ
			tag = PAL
		}
		NOT = { is_at_war = yes }
		OR = {
			AND = {
				is_female = yes
				has_consort_regency = no
			}
			AND = {
				NOT = {	dynasty = "von Wittelsbach"	}
				has_new_dynasty = yes
			}
			has_country_flag = bav_no_heir
			AND = {
				has_regency = yes
				OR = {
					has_heir = no
					has_foreign_heir = yes
				}
			}
		}
		any_country = {
			exists = yes
			government = monarchy
			NOT = {	tag = ROOT }
			dynasty = "von Wittelsbach"
		}
	}
	immediate = {
		log = "FE-Treaty_of_Pavia:[GetYear]:[Root.GetName]:The treaty of Pavia - Root event 0 happened:treaty_pavia.1000"
		hidden_effect = {
			if = {
				limit = {
					is_subject_other_than_tributary_trigger = yes
					any_country = {
						exists = yes
						dynasty = "von Wittelsbach"
						overlord_of = ROOT
					}
				}
				every_core_province = {
					random_country = {
						limit = {
							dynasty = "von Wittelsbach"
							overlord_of = ROOT
						}
						PREV = {
							add_core = PREV
						}
					}
				}
				if = {
					limit = {
						ai = yes
					}
					random_country = {
						limit = {
							dynasty = "von Wittelsbach"
							overlord_of = ROOT
						}
						inherit = ROOT
					}
				}
			}
			if = {
				limit = {
					NOT = { dynasty = "von Wittelsbach" }
					NOT = {
						is_subject_other_than_tributary_trigger = yes
						any_country = {
							exists = yes
							dynasty = "von Wittelsbach"
							overlord_of = ROOT
						}
					}

						any_country = {
