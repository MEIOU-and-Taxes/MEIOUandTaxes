# THE DECENTRALIZING EFFECT OF THE PROVINCIAL SYSTEM #

province_event = {
	id = flavor_tur.5
	title = "flavor_tur.EVTNAME5"
	desc = "flavor_tur.EVTDESC5"
	picture = muslim_minority_advisor
	trigger = {
		owner = {
			turkish_empire_trigger = yes
			NOT = { has_reform = Pol_Structure5 } #do not have Unitary govt reform
			NOT = { has_country_flag = restricting_the_beys }
			OR = { #event can't happen more often than once per 5 years
				NOT = { has_country_flag = bey_event_timer }
				had_country_flag = {
					flag = bey_event_timer
					days = 1825
				}
			}
		}
		NOT = { region = capital }
		NOT = {	has_province_modifier = autonomous_bey }
		NOT = {	has_province_modifier = semi_autonomous_bey }
	}
	mean_time_to_happen = {
		years = 166
		modifier = {
			factor = 0.5
			NOT = {	continent = owner }
		}
		modifier = {
			factor = 0.75
			any_neighbor_province = {
				NOT = {	owned_by = ROOT	}
			}
		}
		modifier = {
			factor = 0.5
			NOT = {	culture = owner	}
		}
		modifier = {
			factor = 0.5
			NOT = {	culture_group = owner }
		}
	}
	immediate = {
		hidden_effect = {
			owner = {
				clr_country_flag = bey_event_timer
				set_country_flag = bey_event_timer
			}
		}
	}
	option = {
		name = "flavor_tur.EVTOPTA5"
		add_province_modifier = {
			name = autonomous_bey
			duration = 3650
		}
		custom_tooltip = autonomous_bey_tt
	}
	option = {
		name = "flavor_tur.EVTOPTB5"
		add_province_modifier = {
			name = semi_autonomous_bey
			duration = 3650
		}
		custom_tooltip = semi_autonomous_bey_tt
	}
	option = {
		name = "flavor_tur.EVTOPTC5"
		set_province_flag = supressed_bey
		if = {
			limit = {
				likely_rebels = nationalist_rebels
				NOT = {	has_province_modifier = nationalists_organizing	}
			}
			add_province_modifier = {
				name = "nationalists_organizing"
				duration = 3650
			}
			custom_tooltip = nationalists_organizing_tt
			add_unrest = 10
		}
		else = {
			add_unrest = 25 #as nationalists_organizing gives +15
		}
	}
}

			}

				NOT = { is_core = owner }
				NOT = { continent = owner }
				any_neighbor_province = {
					NOT = { owned_by = ROOT }
				}
				NOT = { culture = owner }
				NOT = { culture_group = owner }
			}
		}
	}
	mean_time_to_happen = {
		years = 20
		modifier = {
			factor = 0.5
			num_of_owned_provinces_with = {
				value = 5
				has_province_flag = supressed_bey
			}

		modifier = {
			factor = 0.5
			num_of_owned_provinces_with = {
				value = 10
				has_province_flag = supressed_bey
			}
		}

			factor = 0.5
			num_of_owned_provinces_with = {
				value = 3
				has_province_modifier = upset_bey
			}
		}
		modifier = {
			factor = 0.5
			num_of_owned_provinces_with = {
				value = 5
				has_province_modifier = upset_bey
			}
		}
	}
	immediate = {
		hidden_effect = {
			random_owned_province = {
				limit = {
					has_province_flag = supressed_bey
					likely_rebels = nationalist_rebels
					OR = {
						NOT = {	is_core = owner }
						NOT = { continent = owner }
						any_neighbor_province = {
							NOT = {	owned_by = ROOT }
						}
						NOT = { culture = owner }
						NOT = { culture_group = owner }
					}
				}
				set_province_flag = minor_uprising
			}
			random_owned_province = {
				limit = {
					has_province_flag = supressed_bey
					likely_rebels = nationalist_rebels
					OR = {
						NOT = {	is_core = owner }
						NOT = { continent = owner }
						any_neighbor_province = {
							NOT = {	owned_by = ROOT }
						}
						NOT = { culture = owner }
						NOT = { culture_group = owner }
					}
				}
				set_province_flag = minor_uprising
			}
			random_owned_province = {
				limit = {
					has_province_flag = supressed_bey
					likely_rebels = nationalist_rebels
					OR = {
						NOT = {	is_core = owner }
						NOT = { continent = owner }
						any_neighbor_province = {
							NOT = {	owned_by = ROOT }
						}
						NOT = { culture = owner }
						NOT = { culture_group = owner }
					}
				}
				set_province_flag = minor_uprising
			}
		}
		hidden_effect = {
			clr_country_flag = bey_event_timer
			set_country_flag = bey_event_timer
		}
	}

		name = "flavor_tur.EVTOPTA9" #Grant them more autonomy

			limit = {
				has_province_flag = minor_uprising
			}

					has_province_modifier = semi_autonomous_bey
				}

				custom_tooltip = removed_semi_autonomous_bey_tt
			}
			add_province_modifier = {
				name = autonomous_bey
				duration = 3650
			}
			custom_tooltip = autonomous_bey_tt
			clr_province_flag = supressed_bey
			clr_province_flag = minor_uprising
		}
	}

		name = "flavor_tur.EVTOPTB9" #Squash them!
		every_owned_province = {
			limit = {
				has_province_flag = minor_uprising
			}
			if = {
				limit = {
					likely_rebels = nationalist_rebels
				}
				spawn_rebels = {
						type = nationalist_rebels
						size = 2
				}
			}
			else = {
				spawn_rebels = {
						type = warlord_rebels
						size = 2
				}
			}
			clr_province_flag = minor_uprising
		}
