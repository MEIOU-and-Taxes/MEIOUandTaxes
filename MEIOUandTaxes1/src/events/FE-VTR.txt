country_event = {
	id = flavor_vtr.1 # Signore of Volterra considers selling to Pisa
	title = "flavor_vtr.1.ttl"
	desc = "flavor_vtr.1.dsc"
	picture = MERCHANTS_TALKING_eventPicture
	fire_only_once = yes

	trigger = {
		tag = VTR
		owns = 3697
		AND = {
			num_of_cities = 1
			NOT = { num_of_cities = 2 }
		}
		is_subject = no
		has_regency = no
		exists = PIS
	}

	mean_time_to_happen = {
		years = 5
	}

	option = { # offer to sell to Pisa
		name = "flavor_vtr.1.a"
		PIS = {
			country_event = {
				id = flavor_vtr.2
				days = 15
			}
		}
	}
	option = { # do not sell to Pisa
		name = "flavor_vtr.1.b"
		add_adm_power = -50
		add_prestige = 10
	}
}
