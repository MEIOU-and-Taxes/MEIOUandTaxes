namespace = SYS_Colony

province_event = {
    id = SYS_Colony.001
    title = no_localisation
    desc = no_localisation
	picture = MEIOU_AND_TAXES_eventPicture

    hidden = yes

    is_triggered_only = yes
	
	trigger = {
		# These triggers do not work on a freshly colonized province YET, when triggered by: on_province_owner_change or on_colony_established
		#OR = {
		#	is_city = yes
		#	is_colony = yes
		#}
		# Owner check does work though.
		owner = {
			exists = yes
		}
		NOT = { has_province_flag = Colony_Init }
		NOT = { has_global_flag = POP_Sim }
	}

    immediate = {
		owner = {
			set_country_flag = tf_recalculate_communication_efficiency
		}
        # log = "Colony event fired [This.GetName]"
        set_province_flag = Colony_Init
		set_province_flag = Siberia_Init
		
		add_building = pelpel
		add_building = Placeholder1
		add_building = Placeholder2
		add_building = Placeholder3
		add_building = Placeholder4
		add_building = Placeholder5
		add_building = Placeholder6
		add_building = Placeholder7
		
		Infra_InitPrp = { Building=Agriculture SF=70 NM=10 RE=0 NO=5 CL=10 BG=0 BU=5 }
		Infra_InitPrp = { Building=Extraction SF=80 NM=0 RE=0 NO=5 CL=10 BG=0 BU=5 }
		Infra_InitPrp = { Building=Fishery SF=80 NM=0 RE=0 NO=5 CL=10 BG=0 BU=5 }
		Infra_InitPrp = { Building=Forestry SF=80 NM=0 RE=0 NO=5 CL=10 BG=0 BU=5 }
		Infra_InitPrp = { Building=Industrial SF=20 NM=0 RE=55 NO=5 CL=5 BG=10 BU=5 }
		Infra_InitPrp = { Building=Academic SF=0 NM=0 RE=5 NO=5 CL=80 BG=5 BU=5 }
		Infra_InitPrp = { Building=Commercial SF=0 NM=0 RE=5 NO=0 CL=0 BG=90 BU=5 }
		
		if = {
			limit = {
				NOT = { check_key = { lhs = SF_Total value = 10.0 } }
			}
			Public_ChangeClassVal = { class=SF type=value inp=7.5 }
		}
		change_key = { lhs = SF_Wealth value = 100 }
		
		if = {
			limit = {
				check_key = { lhs = Prov_TRPow value = 75 }
			}
			set_key = { lhs = Prov_TRPow value = 75 }
		}

		# We multiply the rural food industry that gets created by the size of the class (peasants or nomads)
		# Peasant size scales industry
		if = {
			limit = {
				check_key = { lhs = SF_Total value = 20.0 }
			}
			set_key = { lhs = Tmp_1 value = 2.0 }
		} else_if = {
			limit = {
				NOT = { check_key = { lhs = SF_Total value = 5.0 } }
			}
			set_key = { lhs = Tmp_1 value = 0.5 }
		} else = {
			set_key = { lhs = Tmp_1 which = SF_Total }
			multiply_key = { lhs = Tmp_1 value = 0.1 } #10k is base
		}
		# Nomad size scales industry
		if = {
			limit = {
				check_key = { lhs = NM_Total value = 30.0 }
			}
			set_key = { lhs = Tmp_2 value = 3.0 }
		} else_if = {
			limit = {
				NOT = { check_key = { lhs = NM_Total value = 5.0 } }
			}
			set_key = { lhs = Tmp_2 value = 0.5 }
		} else = {
			set_key = { lhs = Tmp_2 which = NM_Total }
			multiply_key = { lhs = Tmp_2 value = 0.1 } #10k is base
		}

		#COL_CreationLog = { info=Pre_InitIndustry logCrops = yes }
		COL_InitAgriculture = { RE_MultiplierVar=Tmp_1 NM_MultiplierVar=Tmp_2}
		COL_InitForestry = { RE_MultiplierVar=Tmp_1 }
		COL_InitExtraction = { RE_MultiplierVar=Tmp_1 }
		COL_InitFishery = { RE_MultiplierVar=Tmp_1 }
		COL_InitMiscRuralIndustry = { RE_MultiplierVar=Tmp_1 }
		COL_InitUrbanIndustry = yes
		#COL_CreationLog = { info=Post_InitIndustry logIndustry = yes }
		
		#Remove temp keys, used for industry size initialization
		set_key = { lhs = Tmp_1 value = 0 }
		set_key = { lhs = Tmp_2 value = 0 }
		set_key = { lhs = Tmp_9 value = 0 }

		set_key = { lhs = Mil_StateSpendMulti value = 1 } # remove after early feb
		Rights_SyncFromOwner = yes
		#add_permanent_province_modifier = { name = Tax_Mod duration = -1 }
		Stat_Set = yes
		Prov_InitAutonomy = { multiplierValue = 0.666 }
		
		owner = {
			capital_scope = {
				ROOT = {
                    set_key = { lhs = TN_ProvPrc1 which = PREV }
                    set_key = { lhs = TN_ProvPrc2 which = PREV }
                    set_key = { lhs = TN_ProvPrc4 which = PREV }
                    set_key = { lhs = TN_ProvPrc5 which = PREV }
                    set_key = { lhs = TN_ProvPrc6 which = PREV }
                    set_key = { lhs = TN_ProvPrc9 which = PREV }
                    set_key = { lhs = TN_ProvPrc10 which = PREV }
                    set_key = { lhs = TN_ProvPrc21 which = PREV }
                    set_key = { lhs = TN_ProvPrc22 which = PREV }
                    set_key = { lhs = TN_ProvPrc23 which = PREV }
                    set_key = { lhs = TN_ProvPrc24 which = PREV }
                    set_key = { lhs = TN_ProvPrc25 which = PREV }
                    set_key = { lhs = TN_ProvPrc41 which = PREV }
                    
                    set_key = { lhs = Tmp_9 value = 0 }
                    
                        #add immunity based on capital's resistance
                    if = {
                        limit = {
                            OR = {
                                continent = south_america
                                continent = north_america
                            }
                        }
                        set_key = { lhs = Plague_Resistance1 which = PREV }
                        set_key = { lhs = Plague_Resistance2 which = PREV }
                        set_key = { lhs = Plague_Resistance4 which = PREV }
                    }
                }
			}
		}

		#COL_CreationLog = { info=Pre_Modi_ProdProvince logIndustry = yes}

		Modi_ProdProvince = yes
		Modi_BuildingProvince = yes
		Infra_SetBuildingMax = yes
		Prod_DoMain_Prov = yes
		Prod_DoInfraAndProperty_Prov = yes
		Prod_DoMain_Prov = yes
		Prod_DoInfraAndProperty_Prov = yes
		Prod_DoMain_Prov = yes
		Prod_DoInfraAndProperty_Prov = yes

		COL_CreationLog = { info=ColonyCreatedEnd logIndustry = yes}
	}

    option = {
		name = no_localisation
    }
	
	# System for spawning smallpox
    after = {
		random_list = {
			1 = {
				if = {
					limit = {
						OR = {
						   continent = south_america
						   continent = north_america
						}
					}
					if = {
						limit = {
							owner = {
								any_owned_province = {
									check_key = { lhs = Plague_SpawnChance4 value = 0.1 }
								}
							}
						}
						if = {
							limit = {
								has_province_flag = possible_malaria
							}
							America_Malaria_Calculator = yes
						}
						every_neighbor_province = {
							limit = {
								has_province_flag = possible_malaria
							}
							America_Malaria_Calculator = yes
						}                                
					}
					if = {
						limit = {
							continent = south_america
							has_global_flag = not_samerica_smallpox
						}
						every_neighbor_province = {
							limit = {
								NOT = {
									check_key = { lhs = Plague_SpawnChance2 value = 0.1 }
									check_key = { lhs = Plague_Resistance2 value = 0.1 }
								}
								owner = {
									OR = {
										technology_group = andean
										technology_group = mesoamerican
										technology_group = south_american
										technology_group = andean
										technology_group = high_american
						   
									}
								}
							}
							province_event = { id = Plague_Spawner.222 days = 2555 random = 365 }
						}
					}
					else_if = {
						limit = {
							superregion = central_america_superregion
							has_global_flag = not_camerica_smallpox
						}
						every_neighbor_province = {
							limit = {
								NOT = {
									check_key = { lhs = Plague_SpawnChance2 value = 0.1 }
									check_key = { lhs = Plague_Resistance2 value = 0.1 }
								}
								owner = {
									OR = {
										technology_group = andean
										technology_group = mesoamerican
										technology_group = south_american
										technology_group = andean
										technology_group = high_american
						   
									}
								}
							}
							province_event = { id = Plague_Spawner.221 days = 2555 random = 365 }
						}
					}
					else_if = {
						limit = {
							superregion = east_america_superregion
							has_global_flag = not_namerica_smallpox
						}
						every_neighbor_province = {
							limit = {
								NOT = {
									check_key = { lhs = Plague_SpawnChance2 value = 0 }
									check_key = { lhs = Plague_Resistance2 value = 0 }
								}
								owner = {
									OR = {
										technology_group = andean
										technology_group = mesoamerican
										technology_group = south_american
										technology_group = andean
										technology_group = high_american
						   
									}
								}
							}
							province_event = { id = Plague_Spawner.221 days = 2555 random = 365 }
						}
					}
				}
			}
			2 = {
			}
		}
    }
}

country_event = { # Apply Exploitation to Provinces
	id = SYS_Colony.005
	title = no_localisation
	desc = no_localisation
	picture = MEIOU_AND_TAXES_eventPicture
	is_triggered_only = yes
	hidden = yes
	trigger = {
		isValidCountry = yes
		NOT = { has_institution = Academia }
		any_owned_province = {
			OR = {
				region = brazil_region
				region = colombia_region
				region = guyana_region
				region = carribeans_region
				region = texas_region
				region = southeast_america_region
				region = mississippi_region
				region = central_america_region
			}
		}
		OR = {
			is_colonial_nation = yes
			is_former_colonial_nation = yes
			has_country_flag = changed_from_colonial_nation
			capital_scope = {
				NOT = {
					continent = south_america
					continent = north_america
				}
			}
		}
	}
	immediate = {
		every_owned_province = {
			limit = {
				NOT = { has_province_modifier = exploitation_economy_province }
				OR = {
					region = brazil_region
					region = colombia_region
					region = guyana_region
					region = carribeans_region
					region = texas_region
					region = southeast_america_region
					region = mississippi_region
					region = central_america_region
				}
			}
			add_permanent_province_modifier = {
				name = exploitation_economy_province
				duration = -1
			}
		}
	}
	option = {
		name = no_localisation
    }
}

		has_institution = Academia
		any_owned_province = {
			has_province_modifier = exploitation_economy_province
		}
	}
	immediate = {
		every_owned_province = {
			limit = {
				has_province_modifier = exploitation_economy_province
			}
			remove_province_modifier = exploitation_economy_province
		}
	}
	option = {
		name = no_localisation
    }
