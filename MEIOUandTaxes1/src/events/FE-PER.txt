country_event = {
	id = flavor_per.3
	title = "flavor_per.EVTNAME3"
	desc = "flavor_per.EVTDESC3"
	picture = ADVISOR_eventPicture
	fire_only_once = yes
	trigger = {
		religion = shiite
		OR = {
			culture_group = persian_group
			culture_group = mashreqi
		}
		num_of_cities = 15
		OR = { #there is a Christian European power with more advanced BU Service reform level (or same but ahead of tech)
			AND = {
				NOT = { Rights_BUSer3HasGreater = yes }
				any_known_country = {
					religion_group = christian
					capital_scope = {
						continent = europe
					}
					num_of_cities = 15
					Rights_BUSer3HasGreater = yes
				}
			}
			AND = {
				NOT = { Rights_BUSer4HasGreater = yes }
				any_known_country = {
					religion_group = christian
					capital_scope = {
						continent = europe
					}
					num_of_cities = 15
					OR = {
						Rights_BUSer4HasGreater = yes
						AND = {
							Rights_BUSer3HasGreater = yes
							tech_difference = 3
						}
					}
				}
			}
			AND = {
				Rights_BUSer4HasGreater = yes
				any_known_country = {
					religion_group = christian
					capital_scope = {
						continent = europe
					}
					num_of_cities = 15
					Rights_BUSer4HasGreater = yes
					tech_difference = 3
				}
			}
		}
	}
	mean_time_to_happen = {
		years = 25
	}
	immediate = {
		if = {
			limit = {
				NOT = { Rights_BUSer3HasGreater = yes }
			}
			random_country = {
				limit = {
					religion_group = christian
					capital_scope = {
						continent = europe
					}
					num_of_cities = 15
					Rights_BUSer3HasGreater = yes
				}
				capital_scope = {
					save_event_target_as = origin_of_advisor
				}
			}
		}
		else_if = {
			limit = {
				NOT = { Rights_BUSer4HasGreater = yes }
			}
			random_country = {
				limit = {
					religion_group = christian
					capital_scope = {
						continent = europe
					}
					num_of_cities = 15
					OR = {
						Rights_BUSer4HasGreater = yes
						AND = {
							tech_difference = 3
							Rights_BUSer3HasGreater = yes
						}
					}
				}
				capital_scope = {
					save_event_target_as = origin_of_advisor
				}
			}
		}
		else = {
			random_country = {
				limit = {
					religion_group = christian
					capital_scope = {
						continent = europe
					}
					num_of_cities = 15
					Rights_BUSer4HasGreater = yes
					tech_difference = 3
				}
				capital_scope = {
					save_event_target_as = origin_of_advisor
				}
			}
		}
	}
	option = {
		name = "flavor_per.EVTOPTA3"
		add_ruler_modifier = {
			name = foreign_advisors
		}
		define_advisor = {
			type = army_organiser_BU
			skill = 2
			culture = event_target:origin_of_advisor
			religion = event_target:origin_of_advisor
		}
		define_advisor = {
			type = army_reformer_BU
			skill = 2
			culture = event_target:origin_of_advisor
			religion = event_target:origin_of_advisor
		}
		Pol_ChangeRelationsStateTooltip = { CL=yes type=value val=-5 }
		Pol_ChangeRelationsStateTooltip = { NO=yes type=value val=-5 }
		Pol_ChangeRelationsStateTooltip = { TR=yes type=value val=-5 }
	}
	option = {
		name = "flavor_per.EVTOPTB3"
	}
}
