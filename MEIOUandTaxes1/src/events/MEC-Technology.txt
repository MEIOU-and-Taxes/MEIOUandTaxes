namespace = MEC_Technology

country_event = { ### Forced tech upgrades
	id = MEC_Technology.1
	title = "no_localisation"
	desc = "no_localisation"
	picture = CITY_DEVELOPMENT_AU_eventPicture
	hidden = yes
	is_triggered_only = yes
	trigger = {
		ai = yes
		OR = {
			adm_power = 600
			dip_power = 600
			mil_power = 600
		}
	}
	immediate = {
		set_variable = { which = Tmp_0 value = 0 }
		set_variable = { which = Tmp_1 value = 0 }
		set_variable = { which = Tmp_2 value = 0 }
		adm_available = { output=Tmp_0 }
		multiply_variable = { which = Tmp_0 value = 0.75 }
		dip_available = { output=Tmp_1 }
		multiply_variable = { which = Tmp_1 value = 0.75 }
		mil_available = { output=Tmp_2 }
		multiply_variable = { which = Tmp_2 value = 0.75 }
		export_tech_cost = yes
		
		if = {
			limit = {
				check_variable = { which = Tmp_0 which = Tech_ADMCost }
			}
			set_variable = { which = Tmp_0 which = Tech_ADMCost }
			#log = "TECH_FIX added administrative tech to [This.GetName]"
			scaled_milit_cost_country = { input = Tmp_0 }
			add_adm_tech = 1
		}
		if = {
			limit = {
				check_variable = { which = Tmp_1 which = Tech_DIPCost }
			}
			set_variable = { which = Tmp_1 which = Tech_DIPCost }
			#log = "TECH_FIX added diplomatic tech to [This.GetName]"
			scaled_milit_cost_country = { input = Tmp_1 }
			add_dip_tech = 1
		}
		if = {
			limit = {
				check_variable = { which = Tmp_2 which = Tech_MILCost }
			}
			set_variable = { which = Tmp_2 which = Tech_MILCost }
			#log = "TECH_FIX added military tech to [This.GetName]"
			scaled_milit_cost_country = { input = Tmp_2 }
			add_mil_tech = 1
		}
		
		set_variable = { which = Tech_ADMCost value = 0 }
		set_variable = { which = Tech_DIPCost value = 0 }
		set_variable = { which = Tech_MILCost value = 0 }
	}
	option = {
		name = "no_localisation"
	}
}
