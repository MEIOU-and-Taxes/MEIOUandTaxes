### THE GREAT NORTHERN CONFLICT ###

country_event = {
	id = flavor_dan.3300
	title = "flavor_dan.EVTNAME3300"
	desc = "flavor_dan.EVTDESC3300"
	picture = MILITARY_CAMP_eventPicture
	fire_only_once = yes
	trigger = {
		OR = {
			tag = DEN
			tag = DAN
		}
		dip_tech = 50
		owns = 12
		NOT = {
			has_opinion = {
				who = SWE
				value = 100
			}
		}
		NOT = {
			SWE = {
				has_opinion = {
					who = ROOT
					value = 100
				}
			}
		}
		NOT = {
			alliance_with = SWE
		}
		skaneland_area = {
			owned_by = SWE
		}
		SWE = {
		    is_subject = no
		}
	}
	mean_time_to_happen = {
		years = 16
	}
	option = {
		name = "flavor_dan.EVTOPTA3300"
		every_province = {
		    limit = {
			    area = skaneland_area
				NOT = { is_core = ROOT }
				NOT = { owned_by = ROOT }
			}
			add_core = ROOT
		}
		if = {
			limit = {
				NOT = {
					exists = DAN
				}
			}
			SWE = {
				add_opinion = {
					who = DEN
					modifier = have_claim
				}
			}
			SWE = {
				every_rival_country = {
					add_opinion = {
						who = DEN
						modifier = opinion_common_enemy
						years = 30
					}
				}
			}
		}
		if = {
			limit = {
				NOT = {
					exists = DEN
				}
			}
			SWE = {
				add_opinion = {
					who = DAN
					modifier = have_claim
				}
			}
			SWE = {
				every_rival_country = {
					add_opinion = {
						who = DAN
						modifier = opinion_common_enemy
						years = 30
					}
				}
			}
		}
	}
	option = {
		name = "flavor_dan.EVTOPTB3300"
		every_province = {
		    limit = {
			    area = skaneland_area
				is_core = ROOT
				NOT = { owned_by = ROOT }
			}
			remove_core = ROOT
		}
		add_prestige = -10
		if = {
			limit = {
				NOT = {
					exists = DAN
				}
			}
			SWE = {
				add_opinion = {
					who = DEN
					modifier = opinion_ignored_claim
				}
			}
		}
		if = {
			limit = {
				NOT = {
					exists = DEN
				}
			}
			SWE = {
				add_opinion = {
					who = DAN
					modifier = opinion_ignored_claim
				}
			}
		}
	}
}
