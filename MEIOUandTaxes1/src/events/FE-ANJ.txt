	title = "flavor_anj.1.name"
	desc = "flavor_anj.1.desc"
	picture = COURT_eventPicture
	fire_only_once = yes
	is_triggered_only = yes #on_monarch_death
	trigger = {

				}
			}

	}

			factor = 100
		}
		every_province = {

					region = naples_region
					area = provence_area

			remove_claim = ROOT
		}
	}

		name = "flavor_anj.1.optb"

			factor = 0

	}
