    hidden = yes

    is_triggered_only = yes

    immediate = {
		hidden_effect = {
			every_country = {
				limit = {
					isValidCountry = yes
				}
				Merc_Update = yes

				## AI section to disband mercs after being at peace
				if = {
					limit = {
						is_month = 11
						ai = yes
						is_at_war = no
						had_recent_war = 3
					}
					while = {
						limit = {
							num_of_mercenaries = 1
						}
						random_hired_mercenary_company = {
							disband_mercenary_company = THIS
						}
					}
				}
			}
		}
    }

    option = {
		name = no_localisation
    }
}
