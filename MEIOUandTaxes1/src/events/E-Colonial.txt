	id = colonial.17
	title = "EVTNAME6401"
	desc = "EVTDESC6401"
	picture = COLONIZATION_eventPicture
	is_triggered_only = yes
	trigger = {
		is_colony = yes
		colonysize = 100
		NOT = { colonysize = 500 }
		owner = {
			any_owned_province = {
				is_colony = yes
				colonysize = 500
				NOT = { province_id = ROOT }
			}
		}
	}

	immediate = {
		hidden_effect = {
			owner = {
				random_owned_province = {
					limit = {
						is_colony = yes
						colonysize = 500
						NOT = { province_id = ROOT }
					}
					save_event_target_as = migration_province_target
				}
			}
		}
	}

	option = {
		name = "EVTOPTA6401"
		goto = migration_province_target
		add_colonysize = -200
		event_target:migration_province_target = {
			add_colonysize = 200
		}
	}
}
