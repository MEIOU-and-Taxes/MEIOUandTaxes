				flag = claimed_naples

		}
	}

		set_country_flag = FRA_had_event_3108
	}
	option = { #No one seems to acknowledge our claims
		name = "flavor_fra.EVTOPTA3108"
		Stab_Add1 = yes

			limit = {
				italy_superregion = {
					owned_by = PREV
					is_claim = ROOT
				}
			}

				who = ROOT

			}
		}

			limit = {
				is_claim = ROOT
			}

}
