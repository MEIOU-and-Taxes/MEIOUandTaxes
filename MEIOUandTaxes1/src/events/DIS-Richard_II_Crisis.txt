#Alice Perrers
country_event = {
	id = richardii_crisis.10
	title = "richardii_crisis.10.t"
	desc = "richardii_crisis.10.d"
	picture = DIPLOMACY_eventPicture
	
	is_triggered_only = yes
	trigger = {
		tag = ENG
		has_ruler = "Edward III"
		OR = {
			had_country_flag = {
				flag = alice_perrers
				days = 180
			}
			NOT = { has_country_flag = alice_perrers }
		}
	}

	option = {
		name = "richardii_crisis.10.a"
		if = {
			limit = {
				NOT = { has_country_flag = alice_perrers } 
			}
			set_country_flag = alice_perrers
		}
		
		add_legitimacy = -10
		Public_ChangePowerbrokerLoyaltyTooltipProv = {
			Powerbroker = BU
			Amount = -5
		}
		Public_ChangePowerbrokerLoyaltyTooltipProv = {
			Powerbroker = NO
			Amount = -5
		}
		add_corruption = 5
		add_treasury = -100
		random_owned_province = {
			Public_ChangeClassWealthVal = { class= NO type = value inp= 100 }
		}
	}
}

	}
	option = {
		name = "richardii_crisis.11.c"

			who = LNC
			modifier = suspicion_of_lancaster
		}
		set_country_flag = lancaster_suspecion
	}
}
