# flavor_hyw.11 - The Plantagenets are replaced on the throne of England by another dynasty
country_event = {
	id = flavor_hyw.11
	title = "flavor_hyw.11.t"
	desc = "flavor_hyw.11.d"
	picture = DIPLOMACY_eventPicture
	major = yes
	is_triggered_only = yes #on_yearly_pulse
	trigger = {
		has_global_flag = hundred_years_war
		tag = ENG
		NOT = {
			dynasty = "Plantagenet"
			dynasty = "Lancaster"
			dynasty = "York"
		}
		OR = {
			has_regency = no
			has_consort_regency = yes #TODO: with consort regency conditions can be met, isn't it a mistake?
		}
	}
	immediate = {
		clr_global_flag = hundred_years_war
		log = "FE-TempHYW:[GetYear]::flag hundred_years_war was cleared:flavor_hyw.11"
		log = "FE-TempHYW:[GetYear]:[Root.GetName]:was ruled by [Root.Monarch.GetTitle] [Root.Monarch.GetName]:flavor_hyw.12"
		set_global_flag = hundred_years_war_FRA_win
		FRA = {
			country_event = {
				id = flavor_hyw.01
				days = 1
			}
		}
	}
	option = {
		name = "OPT.FIGHT"
		random_list = {
			30 = {
			}
			40 = {
				Stab_Subtract50Percent = yes
			}
			30 = {
				Stab_Subtract1 = yes
			}
		}
		capital_scope = {
			pretender_rebels = 2
		}
		custom_tooltip = hyw_ending_explanation
		log = "FE-TempHYW:[GetYear]:[Root.GetName]:Plantagenet dynasty died out, but HYW continued:flavor_hyw.11"
	}
	option = {
		name = "OPT.YIELD"
		eng_loses_hyw_effect = yes
		custom_tooltip = hyw_ending_explanation
		log = "FE-TempHYW:[GetYear]:[Root.GetName]:Plantagenet dynasty died out, HYW ended:flavor_hyw.11"
	}
}

# flavor_hyw.13 - England has lost its continental possessions
country_event = {
	id = flavor_hyw.13
	title = "flavor_hyw.11.t"
	desc = "flavor_hyw.13.d"
	picture = DIPLOMACY_eventPicture
	is_triggered_only = yes #on_yearly_pulse
	major = yes
	trigger = {
		has_global_flag = hundred_years_war
		tag = ENG
		is_year = 1360		# This is just to make sure demesne variables are calculated
		#TODO: those conditions are such that if other English vassal has Demesne_in_France, England will still count as looser of the HYW, I would go with all_subject_countries check
		AND = { 
			OR = { #owns no Demesne_in_France other from Calais
				AND = {
					owns = 63					# Calais
					NOT = {
						check_variable = {
							which = "Demesne_in_France"
							value = 2
						}
					}
				}
				NOT = {
					check_variable = {
						which = "Demesne_in_France"
						value = 1
					}
				}
			}
			GUY = { #GUY is not subject of England or holds no Demesne_in_France
				OR = {
					exists = no
					NOT = { is_subject_of = ROOT }
					NOT = {
						check_variable = {
							which = "Demesne_in_France"
							value = 1
						}
					}
				}
			}
			DAL = { #DAL is not subject of England or holds no Demesne_in_France
				OR = {
					exists = no
					NOT = { is_subject_of = ROOT }
					NOT = {
						check_variable = {
							which = "Demesne_in_France"
							value = 1
						}
					}
				}
			}
		}
	}
	immediate = {
		clr_global_flag = hundred_years_war
		log = "FE-TempHYW:[GetYear]::flag hundred_years_war was cleared:flavor_hyw.13"
		set_global_flag = hundred_years_war_FRA_win
		FRA = {
			country_event = {
				id = flavor_hyw.02
				days = 1
			}
		}
	}
	option = {
		name = "OPT.FIGHT"
		random_list = {
			30 = {
			}
			40 = {
				Stab_Subtract50Percent = yes
			}
			30 = {
				Stab_Subtract1 = yes
			}
		}
		capital_scope = {
			pretender_rebels = 2
		}
		custom_tooltip = hyw_ending_explanation
		log = "FE-TempHYW:[GetYear]:[Root.GetName]:lost continent posessions, but HYW continued:flavor_hyw.13"
	}
	option = {
		name = "OPT.YIELD"
		eng_loses_hyw_effect = yes
		log = "FE-TempHYW:[GetYear]:[Root.GetName]:lost continent posessions, HYW ended:flavor_hyw.13"
		custom_tooltip = hyw_ending_explanation
	}
}

# flavor_hyw.18-19 - Time for war
country_event = {
	id = flavor_hyw.18
	title = "flavor_hyw.18.t"	# Our claims in France
	desc = "flavor_hyw.18.d"	# Our fortunes on the continent have not been up to our expectations. Time has come to decide if we are still in a position to uphold our claims, or if we decide to cut our losses.
	picture = DIPLOMACY_eventPicture
	major = yes
	is_triggered_only = yes #on_yearly_pulse
	trigger = {
		has_global_flag = hundred_years_war
		tag = ENG
		NOT = { war_with = FRA }
		NOT = { truce_with = FRA }
		#why those windows are so small?
		FRA = {
			exists = yes
			check_variable = {
				which = "Demesne_in_France"
				value = 10
			}
		}
		check_variable = {
			which = "Demesne_in_France"
			value = 3
		}
		NOT = {
			check_variable = {
				which = "Demesne_in_France"
				value = 5
			}
		}
		stability = 0
		NOT = { war_exhaustion = 5 }
		is_at_war = no
		FRA = { NOT = { check_key = { lhs = Diplo_Alliance_Strength which = PREV } } } 
	}

	immediate = {
		log = "FE-TempHYW:[GetYear]:[Root.GetName]:recognized Diplo_Alliance_Strength to be [Root.GV_Diplo_Alliance_Strength], [FRA.GetName] having [FRA.GV_Diplo_Alliance_Strength]:flavor_hyw.18"
	}

	option = {
		name = "flavor_hyw.18.a"		# The Crown of France is ours.
		ai_chance = {
			factor = 75
			modifier = {
				is_year = 1450
				factor = 0
			}
		}
		add_prestige = 10
		declare_war_with_cb = {
			who = FRA
			casus_belli = cb_hyw_treaty_eng
		}

		name = "flavor_hyw.18.b"		# Yield all claims

			factor = 25

			pretender_rebels = 2
		}
		clr_global_flag = hundred_years_war
		log = "FE-TempHYW:[GetYear]::flag hundred_years_war was cleared:flavor_hyw.18"

	}
}

	picture = DEATH_OF_HEIR_eventPicture
	trigger = {
		is_year = 1358

		NOT = { has_country_flag = negociating_release_jean }
		NOT = { has_country_modifier = not_negotiating_with_english }
		OR = {
			had_country_flag = { flag = france_rejects_condition days = 365 }
			had_country_flag = { flag = negociating_release_jean days = 365 }
		}
		tag = FRA
		capetian_dynasty_trigger = yes
	}

	mean_time_to_happen = {
		months = 6
	}

	option = {
		name = "flavor_hyw.1302.a"
		Public_ChangePowerbrokerLoyaltyTooltipProv = { Powerbroker=NO Amount=-20 }
		Public_ChangePowerbrokerLoyaltyTooltipProv = { Powerbroker=BG Amount=-20 }
		Public_ChangePowerbrokerLoyaltyTooltipProv = { Powerbroker=CL Amount=-20 }
		Public_ChangePowerbrokerLoyaltyTooltipProv = { Powerbroker=TR Amount=-20 }

		if = {
			limit = {
				is_at_war = yes
			}
			add_war_exhaustion = 2
		}

		add_country_modifier = {
			name = not_negotiating_with_english
			duration = 180
		}
	}
}

# The Duke of Alencon 
country_event = {
	id = flavor_hyw.1303
	title = emp_mission_events.10.t
	desc = "emp_mission_events.10.d"
	picture = REFORM_eventPicture
	
	fire_only_once = yes
	
	trigger = {
		tag = FRA
		3374 = { 
			owned_by = FRA
			is_capital = no 
			has_province_flag = lost_to_english #set in enforce_treaty_rouen
		}
		is_at_war = no
	}
	
	mean_time_to_happen = {
		months = 3
	}
	
	immediate = {
		hidden_effect = {
			if = {
				limit = {
					3374 = {
						NOT = { is_core = NRM }
					}
				}
				3374 = { add_core = NRM	clr_province_flag = lost_to_english }
			}
		}
	}
	
	option = { # restore the duke
		name = emp_mission_events.10.a
		hidden_effect = {
			release = NRM
		}
		NRM = {
			if = {
				limit = {
					NOT = { has_country_flag = POP_Init }
				}
				country_event = { id = Init.1 }
			}
			define_ruler = {
				dynasty = FRA
			}
		}
		create_subject = {
			subject_type = appanage_subject
			subject = NRM 
		}
		if = {
			limit = {
				3374 = {
					is_core = FRA
				}
			}
			3374 = { remove_core = FRA }
		}
		Public_ChangePowerbrokerLoyaltyTooltipProv = { Powerbroker=NO Amount=20 }
	}
	
	option = {
		name = emp_mission_events.10.b
		ai_chance = { factor = 0 }
		Stab_Subtract2 = yes
		Public_ChangePowerbrokerLoyaltyTooltipProv = { Powerbroker=NO Amount=-20 }
	}
}
