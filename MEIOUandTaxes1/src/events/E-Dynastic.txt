	trigger = {
		government = monarchy

		is_female = no

			NOT = { has_dlc = "Rights of Man" }

			religion = ROOT

			is_lesser_in_union = no

			ai = yes
		}
	}
	
	mean_time_to_happen = {
		years = 33
	}
	
	immediate = {
		hidden_effect = {
			random_known_country = {
				limit = {
					religion = ROOT
					government = monarchy
					is_lesser_in_union = no
					alliance_with = ROOT
					NOT = { marriage_with = ROOT }
					ai = yes
				}
				save_event_target_as = bargaining_chip_country
			}
		}
	}
	
	option = {
		name = "EVTOPTA9454"
		define_heir = {
			dynasty = ROOT
			claim = 30
		}
	}
	option = {
		name = "EVTOPTB9454"
		trigger = {
			any_known_country = {
				religion = ROOT
				government = monarchy
				is_lesser_in_union = no
				alliance_with = ROOT
				NOT = { marriage_with = ROOT }
				ai = yes
			}
		}
		event_target:bargaining_chip_country = {
			create_marriage = ROOT
		}
		if = {
			limit = {
				has_dlc = "Rights of Man"
			}
			define_consort = {
				country_of_origin = event_target:bargaining_chip_country
				culture = event_target:bargaining_chip_country
				religion = event_target:bargaining_chip_country
			}
		}
	}
}
