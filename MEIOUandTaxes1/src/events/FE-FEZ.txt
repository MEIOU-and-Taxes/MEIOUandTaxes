namespace = flavor_fez
country_event = {
	id = flavor_fez.01
	title = "flavor_fez.01.t"
	desc = "flavor_fez.01.d"
	picture = muslim_arab_raid
	fire_only_once = yes
	trigger = {
		tag = FEZ
		MOR = { exists = yes }
		any_owned_province = {
			culture = fassi
		}
		is_free_or_tributary_trigger = yes
	}
	mean_time_to_happen = {
		months = 1
	}
	immediate = {
		every_owned_province = {
			limit = {
				culture = fassi
			}
			add_core = MOR
		}
	}
	option = {
		name = "OPT.BASTARD"
		ai_chance = {
			factor = 10
		}
		log = "FE-FEZ:[GetYear]:[Root.GetName]:refused to give Fassi provinces to [MOR.GetName], war started:flavor_fez.01"
		every_owned_province = {
			limit = {
				culture = fassi
			}
			cede_province = MOR
		}
		add_casus_belli = { #to not wait for re-calculation event
			target = MOR
			type = cb_core
			months = 24
		}
		declare_war_with_cb = {
			who = MOR
			casus_belli = cb_core
			war_goal_province = 4075 #Marrakesh
		}
	}
	option = {
		name = "OPT.VERYWELL"
		ai_chance = {
			factor = 90
		}
		log = "FE-FEZ:[GetYear]:[Root.GetName]:gave Fassi provinces to [MOR.GetName], revoked its cores:flavor_fez.01"
		every_owned_province = {
			limit = {
				culture = fassi
			}
			cede_province = MOR
			remove_core = FEZ
		}
		add_prestige = -10
		add_legitimacy = -10
	}
	after = {
		MOR = {
			hidden_effect = {
				country_event = { id = Init.1 }
				add_country_modifier = {
					name = blessing_of_god
					duration = 1825
				}
			}
		}
	}
}
