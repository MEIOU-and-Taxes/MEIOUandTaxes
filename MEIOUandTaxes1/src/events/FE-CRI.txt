### SUCCESSION IN CRIMEAN KHANATE ###

country_event = {
	id = flavor_cri.1
	title = "flavor_cri.1.t"
	desc = "flavor_cri.1.d"
	picture = RELIGION_eventPicture
	is_triggered_only = yes
	fire_only_once = yes
	major = yes
	trigger = {
		tag = CRI
		exists = TUR
		is_free_or_tributary_trigger = yes
		is_at_war = no
		TUR = {
			num_of_cities = ROOT
			is_free_or_tributary_trigger = yes
			num_of_ports = 1
		}

	}
	immediate = {
		set_global_flag = death_of_haji_giray
	}
	option = {
		name = "flavor_cri.1.a"
		trigger = {
			NOT = { war_with = TUR }
		}
		ai_chance = {
			factor = 90
			modifier = {
				factor = 1.5
				TUR = {
					num_of_cities = 40
				}
			}
			modifier = {
				factor = 0.8
				num_of_cities = 12
			}
			modifier = {
				factor = 0.8
				num_of_cities = 15
			}
			modifier = {
				factor = 0.25
				num_of_cities = 22
			}
			modifier = {
				factor = 0.1
				num_of_cities = 30
			}
			modifier = {
				factor = 0
				num_of_cities = TUR
			}
			modifier = {
				factor = 1.1
				crimea_region = {
					owned_by = TUR
				}
			}
			modifier = {
				factor = 0
				TUR = {
					NOT = {
						num_of_cities = 5
					}
				}
			}
			modifier = {
				factor = 0
				NOT = {
					north_anatolia_region = {
						owned_by = TUR
					}
					south_anatolia_region = {
						owned_by = TUR
					}
				}
			}
		}
		log = "FE-CRI:[GetYear]:[Root.GetName]:offered servitude to [TUR.GetName]:flavor_cri.1"
		if = {
			limit = {
				has_regency = yes
			}
			define_ruler = {
				name = "Mengli"
				dynasty = "Geray"
				age = 33
				claim = 75
			}
			hidden_effect = {
				if = {
					limit = {
						has_dlc = "Rights of Man"
					}
					add_ruler_personality = conqueror_personality
				}
				if = {
					limit = {
						has_heir = yes
					}
					kill_heir = {}
				}
			}
		}
		custom_tooltip = flavor_cri.1.a.tt
		TUR = {
			country_event = {
				id = flavor_cri.2
				days = 5
			}
		}
		random_owned_province = {
			limit = {
				is_capital = no
			}
			spawn_rebels = {
				size = 2
				type = pretender_rebels
				leader = "Nur Devlet Geray"
				should_take_capital = yes
			}
		}
	}
	option = {
		name = "flavor_cri.1.b"
		ai_chance = {
			factor = 10
			modifier = {
				factor = 3
				TUR = {
					NOT = { num_of_cities = 5 }
				}
			}
			modifier = {
				factor = 1.5
				TUR = {
					NOT = { num_of_cities = 10 }
				}
			}
			modifier = {
				factor = 1.5
				NOT = {
					north_anatolia_region = {
						owned_by = TUR
					}
					south_anatolia_region = {
						owned_by = TUR
					}
				}
			}
			modifier = {
				factor = 1.2
				TUR = {
					is_at_war = yes
				}
			}
		}
		log = "FE-CRI:[GetYear]:[Root.GetName]:decided not to offer servitude to [TUR.GetName]:flavor_cri.1"
		if = {
			limit = {
				has_regency = yes
			}
			define_ruler = {
				name = "Nur Devlet"
				dynasty = "Geray"
				age = 33
			}
			hidden_effect = {
				if = {
					limit = {
						has_dlc = "Rights of Man"
					}
					hidden_effect = {
						add_ruler_personality = inspiring_leader_personality
					}
				}
				if = {
					limit = {
						has_heir = yes
					}
					kill_heir = {}
				}
			}
		}
		custom_tooltip = flavor_cri.1.b.tt
		TUR = {
			country_event = {
				id = flavor_cri.3
				days = 5
			}
		}
		random_owned_province = {
			limit = {
				is_capital = no
			}
			spawn_rebels = {
				size = 2
				type = pretender_rebels
				leader = "Mengli Geray"
				friend = TUR
				should_take_capital = yes
			}
		}
	}
}

		}

		name = "flavor_cri.2.b"

			factor = 0
		}
		add_prestige = -10

			}
		}
	}
}

country_event = {
	id = flavor_cri.4
	title = "flavor_cri.4.t"
	desc = "flavor_cri.4.d"
	picture = RELIGION_eventPicture
	fire_only_once = yes
	is_triggered_only = yes
	option = {
		name = "flavor_cri.4.a"
		add_legitimacy = 30
		add_horde_unity = 30
		hidden_effect = {
			every_subject_country = {
				limit = { is_subject_other_than_tributary_trigger = yes }
				TUR = {
					create_subject = {
						subject_type = tributary_state
						subject = PREV
					}
				}
			}
		}
		TUR = {
			create_subject = {
				subject_type = close_tributary_state
				subject = CRI
			}
		}
		add_opinion = {
			who = TUR
			modifier = support_in_crimean_succession
		}
	}
}
