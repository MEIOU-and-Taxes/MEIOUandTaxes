### Concept for Estates Acquiring State Property
# If Capacity is overburdened, prov will be found where we can sell to local powerbroker
	#Weighting table is as follows:
	#########  Monarch / Tribal / Republic / Theocracy
	#Farmland   6 / 6 / 1 / 6
	#Extraction 7 / 7 / 1 / 2
	#Forestry   4 / 5 / 1 / 3
	#Fishery    5 / 4 / 1 / 5
	#Industrial 2 / 2 / 5 / 4
	#Commerce   1 / 1 / 5 / 1
	#Academic   3 / 3 / 5 / 7
	## The higher the number, the less likely it is picked for selling 
	
##	Type - Default - blocked estates
	##Farmlands - Nobles, clergy in theocracies - burghers, residents
	##Forestry - nobles - residents
	##Mines - nobles, burghers, clergy in respective governments - residents
	##Fishery - nobles in tribe/monarch, burghers in republic/theocracies, - residents
	##Industry - Burghers - clergy, tenants
	##Commerce - Burghers - residents, peasants, clergy
	##Academia - Clergy  - residents, tenants

				multiply_variable = { which = Tmp_2 value = 0.01 } # 50 to 100 -> 0.50 to 1.00
				multiply_variable = { which = Tmp_2 which = BuyOutValue } # Factor x Value

				set_variable = { which = Tmp_3 value = 1 }

				multiply_variable = { which = Tmp_3 value = 0.5 } # to have a 0 to 50 range
				change_variable = { which = Tmp_3 value = 50 } #make sure they offer at least 50%
				multiply_variable = { which = Tmp_3 value = 0.01 } # 50 to 100 -> 0.50 to 1.00
				multiply_variable = { which = Tmp_3 which = BuyOutValue } # Factor x Value
				#Commoners
				set_variable = { which = Tmp_4 value = 0.65 } #Commoners always offer 65% for now
				multiply_variable = { which = Tmp_4 which = BuyOutValue } # Factor x Value	

				ROOT = { set_variable = { which = Tmp_1 which = PREV } } # adding that to country scope for loc
				ROOT = { set_variable = { which = Tmp_2 which = PREV } } # adding that to country scope for loc
				ROOT = { set_variable = { which = Tmp_3 which = PREV } } # adding that to country scope for loc
				ROOT = { set_variable = { which = Tmp_4 which = PREV } } # adding that to country scope for loc
			}

	trigger = {
		has_country_flag = StatePropertyBurden
		check_variable = { which = BuyOutTarget value = 25 }
		check_variable = { which = StatePropertyAcademic value = 20 }
		#NOT = { has_country_flag = NoAcademicForSale }
		NOT = { has_country_flag = StatePropertySalesOngoing }
		is_year = 1360	
	}
	
	mean_time_to_happen = {
		months = 9
		modifier = {
			factor = 0.5
			check_variable = { which = StatePropertyPercCap value = 125 }
		}
		modifier = {
			factor = 0.5
			check_variable = { which = StatePropertyPercCap value = 250 }
		}
		modifier = {
			factor = 3
			government = monarchy
		}
		modifier = {
			factor = 3
			government = tribal
		}
		modifier = {
			factor = 5
			government = republic
		}
		modifier = {
			factor = 7
			government = theocracy
		}
		modifier = {
			factor = 0.5
			ai = yes
		}
	}
	
	immediate = {
		#find a province
		hidden_effect = {
			set_country_flag = StatePropertySalesOngoing
			random_owned_province = {
				limit = {
					check_variable = { which = BU_Academic value = 3 }
					#NOT = { has_province_flag = RecentlySoldProperty }
				}
				save_event_target_as = StatePropertyBuyOut
				# sell it all 
				set_variable = { which = BuyOutAmount which = BU_Academic }
				if = { 
					limit = { 
						check_variable = { which = Building_AcademicGross value = 1 }
					}
					set_variable = { which = BuyOutValue which = Building_AcademicGross } # take profits of that property
				}
				else = {
					set_variable = { which = BuyOutValue value = 1 } # to have at least some kind of value
				}
				divide_variable = { which = BuyOutValue which = Building_Academic } # divide profits with size of that property
				multiply_variable = { which = BuyOutValue value = 10 } # 10 years projected profits
				multiply_variable = { which = BuyOutValue which = BuyOutAmount } # projected profits x amount to be sold
				ROOT = { set_variable = { which = BuyOutValue which = PREV } } # Theoretical value saved to country scope
				ROOT = { set_variable = { which = BuyOutAmount which = PREV } } # Amount saved to country scope
				
				#Nobility
				set_variable = { which = Tmp_1 value = 1 }
				multiply_variable = { which = Tmp_1 which = Prov_NOLoy } # 0 to 100
				multiply_variable = { which = Tmp_1 value = 0.5 } # to have a 0 to 50 range
				change_variable = { which = Tmp_1 value = 50 } #make sure they offer at least 50%
				multiply_variable = { which = Tmp_1 value = 0.01 } # 50 to 100 -> 0.50 to 1.00
				multiply_variable = { which = Tmp_1 which = BuyOutValue } # Factor x Value
				#Clergy
				set_variable = { which = Tmp_2 value = 1 }
				multiply_variable = { which = Tmp_2 which = Prov_CLLoy } # 0 to 100
				multiply_variable = { which = Tmp_2 value = 0.5 } # to have a 0 to 50 range
				change_variable = { which = Tmp_2 value = 50 } #make sure they offer at least 50%
				multiply_variable = { which = Tmp_2 value = 0.01 } # 50 to 100 -> 0.50 to 1.00
				multiply_variable = { which = Tmp_2 which = BuyOutValue } # Factor x Value				
				#Burgher
				set_variable = { which = Tmp_3 value = 1 }
				multiply_variable = { which = Tmp_3 which = Prov_BGLoy } # 0 to 100
				multiply_variable = { which = Tmp_3 value = 0.5 } # to have a 0 to 50 range
				change_variable = { which = Tmp_3 value = 50 } #make sure they offer at least 50%
				multiply_variable = { which = Tmp_3 value = 0.01 } # 50 to 100 -> 0.50 to 1.00
				multiply_variable = { which = Tmp_3 which = BuyOutValue } # Factor x Value
				#Commoners
				set_variable = { which = Tmp_4 value = 0.65 } #Commoners always offer 65% for now
				multiply_variable = { which = Tmp_4 which = BuyOutValue } # Factor x Value	

				ROOT = { set_variable = { which = Tmp_1 which = PREV } } # adding that to country scope for loc
				ROOT = { set_variable = { which = Tmp_2 which = PREV } } # adding that to country scope for loc
				ROOT = { set_variable = { which = Tmp_3 which = PREV } } # adding that to country scope for loc
				ROOT = { set_variable = { which = Tmp_4 which = PREV } } # adding that to country scope for loc
			}
		}
	}
	#Nobility
	option = {
		name = SYS_StateProperty.008.a
		trigger = {
			event_target:StatePropertyBuyOut = {
				check_variable = { which = NO_Wealth which = Tmp_1 }
			}
		}
		hidden_effect = {
			event_target:StatePropertyBuyOut = {				
				change_variable = { which = NO_Academic which = BuyOutAmount }
				subtract_variable = { which = BU_Academic which = BuyOutAmount }
				change_variable = { which = NO_PrpAcademic which = BU_PrpAcademic }
				set_variable = { which = BU_PrpAcademic value = 0 }
				change_variable = { which = Building_AcademicShareNO which = Building_AcademicShareBU }
				set_variable = { which = Building_AcademicShareBU value = 0 }
				
				subtract_variable = { which = NO_Wealth which = Tmp_1 }
				
				set_variable = { which = Tmp_5 which = BuyOutAmount }
				multiply_variable = { which = Tmp_5 which = Modi_StatePropertyAcademic } # calculate impact of sold StateProperty
				ROOT = { set_variable = { which = Tmp_5 which = PREV } } # Updating Country Scope, to be used in next iteration
				#set_province_flag = RecentlySoldProperty
				if = {
					limit = {
						NOT = { check_variable = { which = Building_AcademicShareBU value = 0.001 } }
					}
					set_variable = { which = Building_AcademicShareBU value = 0 }
				}
			}
			subtract_variable = { which = BuyOutTarget which = Tmp_5 } # Updating Country Scope, to be used in next iteration
			set_var_from_variable = { var = ducat_gain variable = Tmp_1 }
			old_scaled_ducat_gained_country = yes
			
		}
		if = {
			limit = {
				NOT = { Rights_CLHie3Has = yes }
			}
			event_target:StatePropertyBuyOut = { Public_ChangePowerbrokerLoyaltyTooltip = { Powerbroker=CL Amount=-20 } }
		}
		ai_chance = {
			factor = 100
			modifier = {
				factor = 2.00
				government = monarchy
			}
		}
		
	}

			}

			set_var_from_variable = { var = ducat_gain variable = Tmp_2 }
			old_scaled_ducat_gained_country = yes

		}
		ai_chance = {
			factor = 100
			modifier = {
				factor = 2.00
				government = theocracy
			}
		}

		name = SYS_StateProperty.008.c
		trigger = {
			event_target:StatePropertyBuyOut = {
				check_variable = { which = BG_Wealth which = Tmp_3 }
				#Rights_BGGov4HasGreater = yes
			}
		}

			event_target:StatePropertyBuyOut = {
				change_variable = { which = BG_Academic which = BuyOutAmount }
				subtract_variable = { which = BU_Academic which = BuyOutAmount }
				change_variable = { which = BG_PrpAcademic which = BU_PrpAcademic }

				set_variable = { which = Building_AcademicShareBU value = 0 }
				
				subtract_variable = { which = BG_Wealth which = Tmp_3 }

				ROOT = { set_variable = { which = Tmp_5 which = PREV } } # Updating Country Scope, to be used in next iteration

				factor = 2.00

			}

	option = {
		name = SYS_StateProperty.008.e
		trigger = {
		}
		hidden_effect = {
			event_target:StatePropertyBuyOut = {
				change_variable = { which = Prov_BULoy value = 10 }
				
			}
			#set_country_flag = NoAcademicForSale

		}

		name = SYS_StateProperty.008.f
		trigger = {
			all_owned_province = {
				NOT = { check_variable = { which = BU_Academic value = 3 } }

		}

		}

			factor = 0
		}

	after = {
		hidden_effect = {
			event_target:StatePropertyBuyOut = {
				set_variable = { which = Tmp_1 value = 0 }
