	major = yes
	mean_time_to_happen = {
		months = 1
	}
	trigger = {
		has_country_flag = had_war_of_the_roses
		OR = {
			dynasty = York
			dynasty = Lancaster
		}
		is_lesser_in_union = no
		NOT = {
			had_country_flag = {
				flag = had_war_of_the_roses
				days = 10950
			}
		}
		ruler_age = 45 #to prevent ruler outliving the heir
	}
	option = {
		name = flavor_eng.9237.a
		ai_chance = {
			factor = 0
		}
		Stab_Subtract2 = yes
		add_prestige = 10
	}
	option = {
		name = flavor_eng.9237.b
		ai_chance = {
			factor = 100
		}
		if = {
			limit = {
				stability = 3
			}
			add_adm_power = 50
		}
		else = {
			Stab_Add1 = yes
		}
		define_heir = {
			claim = 100
			name = "Henry #0"
			dynasty = "Tudor"
			culture = english
			adm = 3
			dip = 4
			mil = 2
			age = 24
		}
		if = {
			limit = {
				WLS = {
					is_subject_of = ROOT
					OR = {
						AND = {
							has_regency = no
							dynasty = "Tudor"
						}
						AND = {
							has_regency = yes
							heir_dynasty = "Tudor"
						}
					}
				}
			}
			inherit = WLS
		}
	}
}

	}

		}

		name = hee_ENG.1.a

			type = cb_restore_personal_union
