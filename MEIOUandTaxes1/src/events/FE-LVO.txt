country_event = {
	id = flavor_lvo.001
	title = flavor_lvo.001.t
	desc = flavor_lvo.001.d
	picture = LIVONIANORDER_eventPicture
	fire_only_once = yes
	major = yes
	trigger = {
		ai = yes
		OR = {
			tag = LIV
			tag = LVO
			tag = KUR
		}
		any_owned_province = {
			area = curonia_area
		}
		any_owned_province = {
			area = livonia_area
		}
		AND = {
			OR = {
				is_subject = no
				is_subject_of_type = tributary_state
				is_subject_of_type = close_tributary_state
			}
		}
		NOT = {
			has_global_flag = livonian_war
		}
		is_religion_enabled = protestant
		is_at_war = no

		any_neighbor_country = { #bordering Russian state with strong army
			culture_group = east_slavic
			army_strength = {
				who = ROOT
				value = 2
			}
		}

		any_country = { #bordering state that can be a potential protector
			OR = {
			    AND = {
				    tag = POL
					OR = {
					    is_neighbor_of = ROOT
					    overlord_of = LIT
					}
				}
				AND = {
				    tag = PLC
					is_neighbor_of = ROOT
				}
				AND = {
				    tag = LIT
					is_neighbor_of = ROOT
				}
			}
			army_strength = {
				who = ROOT
				value = 2
			}
		}
	}
	mean_time_to_happen = {
		years = 3
	}
	option = {
		name = "flavor_lvo.001.a"
		ai_chance = {
			factor = 90
		}
		set_global_flag = livonian_war
		if = {
			limit = {
				POL = {
					exists = yes
					overlord_of = LIT
				}
			}
			POL = {
				country_event = {
					id = flavor_lvo.002
					days = 0
				}
			}
			log = "FE-LVO:[GetYear]:[Root.GetName]:asked Poland for protection against Russian neighbor:flavor_lvo.001"
		}
		if = {
			limit = {
				LIT = {
					exists = yes
					AND = {
						OR = {
							is_subject = no
							is_subject_of_type = tributary_state
							is_subject_of_type = close_tributary_state
						}
					}
				}
			}
			LIT = {
				country_event = {
					id = flavor_lvo.002
					days = 0
				}
			}
			log = "FE-LVO:[GetYear]:[Root.GetName]:asked Lithuania for protection against Russian neighbor:flavor_lvo.001"
		}
		if = {
			limit = {
				PLC = {
					exists = yes
				}
			}
			PLC = {
				country_event = {
					id = flavor_lvo.002
					days = 0
				}
			}
			log = "FE-LVO:[GetYear]:[Root.GetName]:asked Commowealth for protection against Russian neighbor:flavor_lvo.001"
		}
	}
	option = {
		name = "flavor_lvo.001.b"
		ai_chance = {
			factor = 10
		}
		set_global_flag = livonian_war
		Stab_Subtract4 = yes

country_event = {
	id = flavor_lvo.008
	title = flavor_lvo.008.t
	desc = flavor_lvo.008.d
	picture = LIVONIANORDER_eventPicture
	fire_only_once = yes
	major = yes
	trigger = {
		NOT = {
			has_global_flag = latvian_culture_flag
		}
		is_year = 1500
		curonia_area = {
			type = all
			country_or_vassal_holds = ROOT
		}
		livonia_area = {
			type = all
			country_or_vassal_holds = ROOT
		}
	}
	mean_time_to_happen = {
		years = 30
	}
	option = {
		name = "flavor_lvo.008.a"
		commonwealth_superregion = {
			limit = {
				country_or_vassal_holds = ROOT
				OR = {
					culture = curonian
					culture = lattgalian
				}
			}
			change_culture = latvian
		}
		hidden_effect = {
			every_known_country = {
				limit = {
					capital_scope = {
						superregion = commonwealth_superregion
					}
					OR = {
						accepted_culture = curonian
						accepted_culture = lattgalian
					}
				}
				remove_accepted_culture = curonian
				remove_accepted_culture = lattgalian
				add_accepted_culture = latvian
			}
		}
		if = {
			limit = {
				OR = {
					primary_culture = curonian
					primary_culture = lattgalian
				}
			}
			change_primary_culture = latvian
		}
		every_subject_country = {
			limit = {
				OR = {
					primary_culture = curonian
					primary_culture = lattgalian
				}
			}
			change_primary_culture = latvian
		}
		log = "FE-LVO:[GetYear]:[Root.GetName]:Formation of Latvian culture:flavor_lvo.008"
		
		ai_chance = {
			factor = 100
		}
	}
	option = {
		name = "flavor_lvo.008.b"
		trigger = {
			primary_culture = lattgalian
		}
		every_owned_province = {
			limit = {
				culture = curonian
			}
			add_nationalism = 10
		}
		ai_chance = {
			factor = 0
		}
	}
	option = {
		name = "flavor_lvo.008.c"
		trigger = {
			primary_culture = curonian
		}
		every_owned_province = {
			limit = {
				culture = lattgalian
			}
			add_nationalism = 10
		}
		ai_chance = {
			factor = 0
		}
	}
	after = {
		set_global_flag = latvian_culture_flag
	}
}
