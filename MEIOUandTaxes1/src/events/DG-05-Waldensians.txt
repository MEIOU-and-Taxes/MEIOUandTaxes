province_event = {
	id = DG_Waldensians.001
	title = "DG_Waldensians.001.title"
	desc = "DG_Waldensians.001.desc"
	picture = all_church_state

	major = yes
	fire_only_once = yes
	
	trigger = {
		NOT = { is_religion_enabled = waldensian }
		province_id = 1389
		NOT = { is_religion_enabled = reformed }
	}
	
	mean_time_to_happen = {
		months = 30
	}
	
	immediate = {
		enable_religion = waldensian
		hidden_effect = { set_global_flag = DG_Waldensian_Spawn }
		set_province_flag = wal_craddle
		save_global_event_target_as = wal_origins
		log = "DG:[GetYear]:[This.aab.GetValue]:Waldensian:Birth:[This.GetName]:[This.Owner.GetName]"
		#log = "DG:[GetYear]:[This.aab.GetValue]:Waldensian:Spread:[This.GetName]:[This.Owner.GetName]"
		hidden_effect = {
			MN_HeresySpread = { typefrom = which from = MN_S0 to = 5 incr = 0.02 typeIncr = value }
			MN_Attribute_Effect = { reltype = value religion = 5 typeEffect = Preach typeStrength = value strength = 3 }
			MN_Attribute_Reset = { reltype = value religion = 5 typeEffect = ENT typeStrength = value strength = 100 }
		}
	}
	
	option = {
		name = "DG_Waldensians.001.opta" #Support him
		hidden_effect = {
			owner = {
				set_country_flag = support_05
				add_opinion = { who = PAP modifier = papacy_bad }
				add_ruler_modifier = {
					name = "heretic_tolerance"
				}
				reverse_add_opinion = { who = PAP modifier = opinion_pope_religious_tolerance }
				MN_conv_resist_c = { strength = -10 }
			}
		}
	}
	
	option = {
		name = "DG_Waldensians.001.optb" #Let him speak
		hidden_effect = {
			owner = {
				set_country_flag = accept_05
				add_ruler_modifier = {
					name = "tolerant_society"
				}
				MN_conv_resist_c = { strength = -5 }
			}
		}
	}
	
	option = {
		name = "DG_Waldensians.001.optc" #Silence him
		hidden_effect = {
			owner = {
				set_country_flag = silence_05
				add_opinion = { who = PAP modifier = papacy_good }
				add_ruler_modifier = {
					name = "religious_intolerance"
				}
				MN_conv_resist_c = { strength = 10 }
			}
		}
	}
	
	after = {
		hidden_effect = {
			owner = {
				country_event = { id = DG_Waldensians.002 }
			}
		}
	}
}
