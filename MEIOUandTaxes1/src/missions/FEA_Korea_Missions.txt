					}

				OR = {
					AND = {
						province_id = 1816
						NOT = { owned_by = root }
						NOT = { is_core = root }
					}
					AND = {
						province_id = 1821
						NOT = { owned_by = root }
						NOT = { is_core = root }
					}
					AND = {
						root = {
							NOT = {
								num_of_owned_provinces_with = {
									superregion = north_china_superregion
									value = 30
								}
							}
						}
						superregion = north_china_superregion
					}
				}
			}
			else_if = {
				limit = {
					NOT = {
						root = { is_emperor_of_china = yes }
					}
				}
				owner = {
					is_emperor_of_china = yes
				}
				is_capital = yes
			}
			else = {
				always = no
			}
		}
		trigger = {
			if = {
				limit = {
					has_dlc = "Mandate of Heaven"
				}
				OR = {
					is_emperor_of_china = yes
					AND = {
						NOT = {
							any_country = {
								is_emperor_of_china = yes
							}
						}
						owns_core_province = 1816
						owns_core_province = 1821
						num_of_owned_provinces_with = {
							superregion = north_china_superregion
							value = 30
						}
					}
				}
			}
			else = {
				owns_core_province = 1816
				owns_core_province = 1821
				num_of_owned_provinces_with = {
					superregion = north_china_superregion
					value = 30
				}
			}
		}
		effect = {
		}
	}
