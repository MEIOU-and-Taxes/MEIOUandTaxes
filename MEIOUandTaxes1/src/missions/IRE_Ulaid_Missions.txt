		trigger = {
			#NOT = { government = tribal }

### Alliance with Castille/Spain
ulaid_alliance_missions = {
	slot = 3
	generic = no
	ai = yes
	potential = {
		has_country_flag = k_of_uls
		OR = {
			exists = SPA
			exists = CAS
			exists = FRA
		}
	}
	has_country_shield = yes
	
	ulaid_ally_spain = {
		icon = mission_alliances
		required_missions = { ulaid_modernise_government }
		position = 4
		trigger = {
			if = {
				limit = { exists = SPA }
				alliance_with = SPA
			}
			else_if = {
				limit = { exists = CAS }
				alliance_with = CAS
			}
			else = {
				alliance_with = FRA
			}
		}
		effect = {
			if = {
				limit = { exists = SPA }
				add_historical_friend = SPA
				SPA = { add_historical_friend = ROOT }
			}
			else_if = {
				limit = { exists = CAS }
				add_historical_friend = CAS
				CAS = { add_historical_friend = ROOT }
			}
			else = {
				add_historical_friend = FRA
				FRA = { add_historical_friend = ROOT }
			}
		}
	}
}
