# INSTRUCTIONS:
#
# condition				A diplomatic action can have any number of condition blocks, each with its own
#						tooltip, potential and allow section
#
# 	tooltip					Sets a custom text string similar to the hardcoded limits
# 							If no tooltip is scripted, the tooltip for the actual trigger will be shown
#							Note that the custom tooltip is only shown if the allow trigger is NOT met
#
# 	potential				Determines if the trigger is applicable or not
# 	allow					Determines if the action is valid or not
#
# effect				A diplomatic action can only have one effect block

# ROOT					actor
# FROM					target

tributary_state_action = { #automatically checks the same triggers but reversed FROM/ROOT for "Become their tributary"
    condition = {
        tooltip = OUTSIDE_DIPLOMATIC_RANGE
        potential = {
            always = yes
        }
        allow = {
            OR = {
                AND = {
                    ROOT = {
                        is_emperor = yes
                    }
                    FROM = { is_part_of_hre = yes }
                }
                AND = {
                    FROM = {
                        is_emperor = yes
                    }
                    ROOT = { is_part_of_hre = yes }
                }
                AND = {
                    FROM = {
                        OR = {
                            tag = PAP
                            tag = AVI
                        }     
                    }
                    ROOT = { 
                        OR = {
                            religion = catholic 
                            religion = avignonist
                        }
                    }
                }
                AND = {
                    FROM = {
                        OR = {
                            religion = catholic
                            religion = avignonist
                        }
                    }
                    ROOT = {
                        OR = {
                            tag = PAP
                            tag = AVI
                        }
                    }
                }

                    custom_tooltip = DIPLOMATIC_RANGE
                    export_to_variable = {
                        variable_name = border_distance
                        value = border_distance
                        who = FROM
                        with = ROOT
                    }
                    check_variable = {
                        which = oko
                        which = border_distance
                    }
                }
            }
        }
    }
	condition = {
		tooltip = MOSCOVY_NO_TRIBUTARIES_OUTSIDE_NOMADS
		potential = {
			OR = {
				is_subject_of_type = muscovite_tributary_state
				culture_group = east_slavic
			}
		}
		allow = {
			FROM = {
				is_nomad = yes
				is_neighbor_of = ROOT
			}
		}
	}
	condition = {
		tooltip = NOCHINATRIBUTE
		potential = {
            culture_group = chinese_group
			has_country_flag = InMiddleKingdom
			
			FROM = {
			    has_country_flag = InMiddleKingdom
			}
		}
		allow = {
			always = no
		}
	}
	condition = {
		tooltip = NOCHINATRIBUTE
		potential = {
			tag = YUA
			
			FROM = {
				culture_group = chinese_group
			}
		}
		allow = {
			always = no
		}
	}
	condition = {
		tooltip = NOCHINATRIBUTE
		potential = {
			culture_group = chinese_group
			
			FROM = {
				tag = YUA
			}
		}
		allow = {
			always = no
		}
	}
	condition = {
		tooltip = UNSTABLEHORDE 
		potential = {
			OR = {
				tag = CHG
				tag = WHI
				tag = GOL
				tag = BLU
				tag = MGH
				FROM = { tag = CHG }
				FROM = { tag = WHI }
				FROM = { tag = GOL }
				FROM = { tag = BLU }
				FROM = { tag = MGH }
			}
		}
		allow = {
			is_year = 1360
			NOT = { has_country_flag = Horde_Collapse_Ongoing }
			NOT = { has_country_flag = Horde_Ruler_Murdered }
			NOT = { has_country_modifier = horde_collapse }
			FROM = {
				NOT = { has_country_flag = Horde_Collapse_Ongoing }
				NOT = { has_country_flag = Horde_Ruler_Murdered }
				NOT = { has_country_modifier = horde_collapse }
			}
		}
	}
	condition = {
		tooltip = TIMURNOPEIRAN
		potential = {
			OR = {
				Timur_Has = yes
				FROM = {
					Timur_Has = yes
				}
			}
			OR = {
				culture_group = persian_group
				FROM = { culture_group = persian_group }
			}
			ai = yes
			FROM = { ai = yes }
		}
		allow = {
			always = no
		}
	}
	condition = {
		tooltip = TIMURNOPE 
		potential = {
			OR = {
				Timur_Has = yes
				FROM = {
					Timur_Has = yes
				}
			}
		}
		allow = {
			OR = {
				AND = {

						}
					}
				}
				AND = {
					Timur_Has = yes
					has_opinion = {
						who = FROM
						value = 125
					}
				}
			}
		}
	}
	condition = {
		tooltip = TIBETNOPE
		potential = {
			ai = yes
			FROM = {
				ai = yes
			}
			OR = {
				culture_group = tibetic

				FROM = { culture_group = tibetic }
				FROM = { culture_group = pahari_group }
			}
		}
		allow = {
			OR = {
				culture_group = FROM
				FROM = { culture_group = ROOT }
				NOT = { technology_group = indian }
				is_year = 1800
			}
		}
	}
    condition = {
        tooltip = SAHARANOPE
        potential = {
            NOT = { is_year = 1600 }
            capital_scope = {
                superregion = north_africa_superregion
            }
