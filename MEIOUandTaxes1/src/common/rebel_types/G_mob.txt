mob_rebels = {
	
	color = { 40 140 210 }
	area = nation 					# Stays within provinces of their culture
	government = anti
	defection = none				# Does not defect
	independence = none				# Does not strive for independence
	gfx_type = culture_province
	
	resilient = no
	reinforcing = no
	general = no
	smart = no
	unit_transfer = no				# Units do not remain after enforcing demands.
	will_relocate = no				# Units will relocate if they have sieged all provinces nearby and has to move over sea.
	
	artillery = 0.0
	infantry = 1
	cavalry = 0.0
	morale = 1
	
	# Possible handle actions
	handle_action_negotiate = yes
	handle_action_stability = no
	handle_action_build_core = no
	handle_action_send_missionary = yes
	
	# The rebel type with the highest modifier for this province gets picked
	spawn_chance = { #game should only use these if there is nothing else
		factor = 1

        modifier = {
            factor = 0.1

			nationalism = 1
		}
		modifier = {
			factor = 0.1

			has_owner_religion = no
		}
		modifier = { 
			factor = 100
			owner = { has_country_modifier = China_PeasantUprising }
		}
	}
	
	#This is checked for EACH province in the Area of Operations
	movement_evaluation = { #they do not know what they are doing
		factor = 1
		modifier = {
			factor = 0.1
			units_in_province = 1
		}
        modifier = {
			factor = 0.001
			controlled_by = REB
		}
        modifier = {
			factor = 0.001
			NOT = { controlled_by = REB }
			has_influencing_fort = yes
		}
		modifier = {
			factor = 10
			NOT = { controlled_by = REB }
		}
		modifier = {
			factor = 0
			is_colony = yes
		}
		modifier = {
			factor = 0
			is_empty = yes
		}
	}
	
	# Province Scope
	siege_won_trigger = {
	}
	siege_won_effect = {
		add_local_autonomy = 40
	}
	
	# Country scope
	can_negotiate_trigger = {
		always = yes
	}
	
	# Country scope
	can_enforce_trigger = {
		always = yes
	}
	
	# Localisation for their demands
	demands_description = "mob_rebels_demand"
	
	# Country scope
	demands_enforced_effect = {
		every_owned_province = {
			limit = {
				has_rebel_faction = REB
			}
			add_local_autonomy = 50
		}
		
		remove_country_modifier = Reform_CD
		
		add_country_modifier = {
			name = Reform_CD
			duration = 3625
		}

		kill_advisor = random
		add_prestige = -50
		kill_advisor = random
		#re-calculate rights priorities calculation
		if = {
			limit = {
				ai = yes
			}
			set_country_flag = AITax_Yearly # Refire Tax Calculation
		}
	}
}
