    name = invest_agri
    potential = {
        owner = { tag = FROM }
    }

        custom_trigger_tooltip = {
            FROM = {
                OR = {
                    AND = { has_country_flag = investing_10 treasury = 10 }

                    AND = { has_country_flag = investing_100 treasury = 100 }
                }
            }
            tooltip = invest_reqs_treasury_tt
        }
		custom_trigger_tooltip = {
			check_key = { lhs = Prov_BUPow value = 2.5 }

		}
		custom_trigger_tooltip = {
			check_key = { lhs = Building_Agriculture value = 0.001 }
			tooltip = Prov_Unopened_BuildingGive_tt
		}	
		custom_trigger_tooltip = {
			NOT = { is_key_equal = { lhs = Building_Agriculture which = Building_AgricultureMax } }
			tooltip = Prov_Full_BuildingGive_tt
		}
        custom_trigger_tooltip = {
            owner = { tag = FROM }
            controlled_by = owner
            tooltip = invest_reqs_control_tt

        if = {
            limit = { FROM = { has_country_flag = investing_10 } }
            change_key = { lhs = Building_AgricultureWealth value = 10 }
            FROM = { add_treasury = -10 }

            limit = { FROM = { has_country_flag = investing_50 } }

        }
        if = {
            limit = { FROM = { has_country_flag = investing_100 } }

            FROM = { add_treasury = -100 }
