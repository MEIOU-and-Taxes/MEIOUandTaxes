	AND = {
		NOT = {
			check_key = { lhs = tv_tt value = $max$ } # 56 {8 weeks} * 20 {DIPLOMAT_SPEED}
		}
		OR = {
			has_province_flag = ce_dispatch_cand
			AND = {
				has_province_flag = is_regional_capital
				NOT = { has_province_flag = regional_capital_ce_final }
			}
		}		
		NOT = { has_province_flag = tf_communication_efficiency_final }
		NOT = { is_capital_of = event_target:tt_highest }

			AND = {
				event_target:runner_dispatching_province = {
					has_province_flag = tf_communication_efficiency_final
				}
				check_key = { lhs = tv_tt value = event_target:runner_dispatching_province }
			}
			AND = {
				event_target:runner_dispatching_province = {
					NOT = { has_province_flag = tf_communication_efficiency_final }
				}				
				NOT = { check_key = { lhs = tv_tt which = event_target:runner_dispatching_province } }
			}
		}
	}	
}
