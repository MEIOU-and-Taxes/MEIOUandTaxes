#WoR version but if England wins HYW. The opposing side can use France as base
uef_plantagnet_civil_war = {
	potential = {
		has_global_flag = hundred_years_war_ENG_win
		OR = {
			AND = {
				has_global_flag = plantagenet_becomes_french_again
				tag = UEF
			}
			AND = {
				has_global_flag = plantagenet_french_regent
				tag = ENG
			}
		}
		NOT = { has_country_flag = uef_plantagnet_civil_war_happened }
		is_free_or_tributary_trigger = yes
		is_lesser_in_union = no
		num_of_owned_provinces_with = {
			value = 5
			culture = english
		}
	}
	
	can_start = {
		#has_any_disaster = no
		OR = {
			NOT = { stability = 2 }
			NOT = {  prestige = 0 }
			war_exhaustion = 5
			NOT = { legitimacy = 75 }
			AND = {
				FRA = {
					exists = yes
					liberty_desire = 50
				}
			}
			REB = { check_variable = { which = french_rebellion_desire value = 10 } }
		}
	}
	
	
	can_stop = {
		OR = {
			has_country_flag = uef_plantagnet_civil_war_ending_1
			has_country_flag = uef_plantagnet_civil_war_ending_2
		}
	}
	
	
	progress = {
		modifier = {
			factor = 0.2
			NOT = { legitimacy = 90 }
		} 
		modifier = {
			factor = 0.2
			NOT = { stability = 0 }
		}
		modifier = {
			factor = 0.5
			AND = {
				primary_culture = francien
				NOT = { has_idea = lingua_franca }
			}
		}
		modifier = {
			factor = 0.25
			AND = {
				primary_culture = francien
				has_idea = lingua_franca
			}
		}
		modifier = {
			factor = 0.2
			war_exhaustion = 5
		} 
		modifier = {
			factor = 1.5
			has_global_flag = convert_to_anglois_denied
		}
		modifier = {
			factor = 0.75
			has_global_flag = plantagenet_becomes_french_again
		}
		modifier = {
			factor = 3
			has_global_flag = plantagenet_french_regent
		}
		modifier = {
			factor = 1
			has_global_flag = king_of_isles
		}
		modifier = {
			factor = 1
			REB = { check_variable = { which = french_rebellion_desire value = 10 } }
		}
		modifier = {
			factor = 1.5
			REB = { check_variable = { which = french_rebellion_desire value = 15 } }
		}
		modifier = {
			factor = 1.5
			REB = { check_variable = { which = french_rebellion_desire value = 20 } }
		}
		modifier = {
			factor = 4
			REB = { check_variable = { which = french_rebellion_desire value = 25 } }
		}
	}
	
	modifier = {
		global_unrest = 10
	}	
	
	can_end = {
		OR = {
			has_country_flag = uef_plantagnet_civil_war_ending_1
			has_country_flag = uef_plantagnet_civil_war_ending_2
		}
		NOT = { num_of_rebel_controlled_provinces = 1 }
	}

	
	on_start = uef_civil_war.1
	on_end = uef_civil_war.100
	
	on_monthly = {
		events = {
			
		}
		
		random_events = {
			#1000 = 0
		}
	}
}
