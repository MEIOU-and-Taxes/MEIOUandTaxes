interregnum = {
	potential = {
		government = monarchy
		NOT = { is_lesser_in_union = yes }
		OR = {
			NOT = {
				has_heir = yes
				OR = {
					has_consort = yes
					heir_age = 15
				}
			}
			has_consort_regency = yes
		}					
	}
	can_start = {
		has_regency = yes
		custom_trigger_tooltip = {
			tooltip = "interregnum_happened_tootlip"
			NOT = {
				has_ruler_modifier = interregnum_happened
			}
		}
	}
	can_stop = {}
	progress = {
		modifier = {
			factor = 1
			NOT = { stability = 0 }
		}
		modifier = {
			factor = 1
			NOT = { stability = -1 }
		}
		modifier = {
			factor = 1
			NOT = { stability = -2 }
		}
		modifier = {
			factor = 1
			war_exhaustion = 10
		}
		modifier = {
			factor = 1.5
			NOT = { legitimacy = 35 }
		}
		modifier = {
			factor = 1
			num_of_loans = 10
		}
		modifier = {
			factor = 2
			is_bankrupt = yes
		}
		modifier = {
			factor = 2
			has_country_flag = ruler_prisonner_of_war
		}
		modifier = {
			factor = 0.5
			overextension_percentage = 0.25
		}
		modifier = {
			factor = 1.0
			overextension_percentage = 0.5
		}
	}
	can_end = {
		NOT = { num_of_revolts = 1 }
		NOT = {
			any_owned_province = {
				has_province_modifier = pretender_organizing
			}
		}
	}
	modifier = {
		improve_relation_modifier = -0.33
		global_unrest = 5
		discipline = -0.05
		diplomatic_reputation = -5
		diplomatic_upkeep = -2
	}
	on_start = flavor_interregnum.1
	on_end = flavor_interregnum.3
	on_monthly = {
		random_events = {
			900 = 0
			100 = flavor_interregnum.2
		}
	}
}
