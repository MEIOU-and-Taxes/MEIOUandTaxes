war_of_the_roses = {
	potential = {
		OR  = {
			tag = ENG
			AND = {
				tag = UEF
				NOT = {	exists = ENG }
			}
		}
		normal_or_historical_nations = yes
		government = monarchy
		AND = {
			OR = {
				is_subject = no
				is_subject_of_type = tributary_state
				is_subject_of_type = close_tributary_state
			}
		}
		uses_doom = no
		NOT = { has_ruler = "Edward III" }
		NOT = {	is_year = 1500 }
		NOT = { has_country_flag = had_war_of_the_roses }
	}
	can_start = {
		has_any_disaster = no
		OR = {
		    has_heir = no
			is_female = yes #having female ruler should also be cause of the war
		}
		FRA = {
			NOT = {
				is_subject_of = ENG
			}
		}
	}
	can_stop = {
		OR = {
			AND = {
				has_any_disaster = yes
				ai = yes
			}
			is_lesser_in_union = yes
			FRA = {
				is_subject_of = ENG
			}
		}
	}
	progress = {
		modifier = {
			factor = 1
			NOT = {	legitimacy = 90	}
		}
		modifier = {
			factor = 3
			OR = {
				has_ruler_modifier = low_fertility
				ruler_has_personality = infertile_personality
			}
		}
		modifier = { #commented out, as this automatically ends disasted by war_of_the_roses.106
			factor = 0.5
			NOT = { 
				dynasty = "Plantagenet"
			}
		}
		modifier = {
			factor = 0.5
			NOT = { stability = 1 }
		}
		modifier = {
			factor = 1
			NOT = { ADM = 2 }
		}
		modifier = {
			factor = 1
			NOT = { DIP = 2 }
		}
		modifier = {
			factor = 1
			NOT = { MIL = 2 }
		}
		modifier = {
			factor = 0.5
			NOT = { ADM = 3 }
		}
		modifier = {
			factor = 0.5
			NOT = { DIP = 3 }
		}
		modifier = {
			factor = 0.5
			NOT = { MIL = 3 }
		}
		modifier = {
			factor = 0.5
			AND = {
				has_global_flag = had_bourguignon_armagnac_conflict 
				has_global_flag = had_treaty_of_troyes
			}
			legitimacy = 70
		}
		modifier = {
			factor = 1
			has_country_flag = henry_iv
			legitimacy = 80
		}
	}
	can_end = {
		always = no
	}
	modifier = {
		global_unrest = 5
		global_trade_goods_size_modifier = -0.10   #I hope this still works in 3.0. I saw it being used in economic ideas
	}
	on_start = war_of_the_roses.1 #Henry becomes mad
	on_end = war_of_the_roses.100
	on_monthly = {
		random_events = {
			100 = war_of_the_roses.5
			100 = war_of_the_roses.8
			100 = war_of_the_roses.9
		}
	}
}
