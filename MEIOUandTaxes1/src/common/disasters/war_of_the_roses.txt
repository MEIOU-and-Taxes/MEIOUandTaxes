war_of_the_roses = {
	potential = {
		OR  = {
			tag = ENG
			AND = {
				tag = UEF
				NOT = {	exists = ENG }
			}
		}
		normal_or_historical_nations = yes
		government = monarchy
		AND = {
			OR = {
				is_subject = no
				is_subject_of_type = tributary_state
				is_subject_of_type = close_tributary_state
			}
		}
		uses_doom = no
		NOT = { has_ruler = "Edward III" }
		NOT = {	is_year = 1500 }
		NOT = { has_country_flag = had_war_of_the_roses }
	}
	can_start = {
		has_any_disaster = no
		OR = {
		    has_heir = no
			is_female = yes #having female ruler should also be cause of the war
		}
		has_regency = no
		is_lesser_in_union = no
	}
	can_stop = {
		OR = {
			has_any_disaster = yes
			has_regency = yes
			is_lesser_in_union = yes
			AND = {
			    is_female = no
			    has_heir = yes
			}
		}
	}
	progress = {
		modifier = {
			factor = 1
			NOT = {	legitimacy = 90	}
		}
		modifier = {
			factor = 3
			OR = {
				has_ruler_modifier = low_fertility
				ruler_has_personality = infertile_personality
			}
		}
		# modifier = { #commented out, as this automatically ends disasted by war_of_the_roses.106
		# 	factor = 0.5
		# 	NOT = { 
		# 		dynasty = "York"
		# 		dynasty = "Lancaster"
		# 	}
		# }
		modifier = {
			factor = 0.5
			NOT = { stability = 1 }
		}
		modifier = {
			factor = 1
			NOT = { ADM = 2 }
		}
		modifier = {
			factor = 1
			NOT = { DIP = 2 }
		}
		modifier = {
			factor = 1
			NOT = { MIL = 2 }
		}
		modifier = {
			factor = 0.5
			NOT = { ADM = 3 }
		}
		modifier = {
			factor = 0.5
			NOT = { DIP = 3 }
		}
		modifier = {
			factor = 0.5
			NOT = { MIL = 3 }
		}
	}
	can_end = {
		NOT = { num_of_rebel_controlled_provinces = 1 }
		NOT = { has_spawned_rebels = pretender_rebels }
		stability = 1
		OR = {
			AND = {
				has_regency = no
				has_heir = yes
			}
			NOT = { government = monarchy }
		}
	}
	modifier = {
		global_unrest = 5
		global_trade_goods_size_modifier = -0.10   #I hope this still works in 3.0. I saw it being used in economic ideas
	}
	on_start = war_of_the_roses.1
	on_end = war_of_the_roses.100
	on_monthly = {
		random_events = {
			150 = 0 #very short disaster, this is why chance for no event is reduced
			100 = war_of_the_roses.2
			100 = war_of_the_roses.3 #mutually exclusive
			100 = war_of_the_roses.4 #mutually exclusive
			100 = war_of_the_roses.5
			100 = war_of_the_roses.8
			100 = war_of_the_roses.9
		}
	}
}
