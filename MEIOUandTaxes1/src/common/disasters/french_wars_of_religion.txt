french_wars_of_religion = {
	potential = {
		tag = FRA
		government = monarchy
		has_global_flag = hundred_years_war_FRA_win
		NOT = { has_country_flag = had_religious_turmoil } 
		NOT = { has_country_flag = FRA_wars_of_religion_ended } 
		OR = {
			religion = catholic
			religion = protestant
			religion = reformed
		}
		is_force_converted = no
	}
	
	can_start = {
		is_religion_enabled = reformed
		has_any_disaster = no
		NOT = { religious_unity = 0.75 }
		any_owned_province = {
			continent = europe
			has_owner_religion = no
			OR = {
				religion = catholic
				religion = protestant
				religion = reformed
			}
		}
	}
	
	can_stop = {
		OR = {
			has_any_disaster = yes
			religious_unity = 0.75
		}
	}
	
	progress = {
		modifier = {
			factor = 1
			NOT = { religious_unity = 0.75 }
			religious_unity = 0.5
		} 		
		modifier = {
			factor = 2
			NOT = { religious_unity = 0.5 }
		} 
		modifier = {
			factor = 1
			has_idea_group = evangelical_ideas
		}
		modifier = {
			factor = 1
			has_idea_group = humanist_ideas
		}
		modifier = {
			factor = 1
			NOT = { stability = 0 }
		}
		modifier = {
			factor = 2
			has_hostile_reformation_center = yes
		}
		modifier = {
			factor = -1
			has_friendly_reformation_center = yes
		}
	}

	modifier = {
		tolerance_heretic = -3
		global_unrest = 10
	}
	
	can_end = {
		NOT = { num_of_rebel_controlled_provinces = 1 }
		stability = 1
		capital_scope = {
			controlled_by = ROOT
		}
		OR = {
			is_force_converted = yes
			religious_unity = 0.9
		}
		NOT = {
			any_owned_province = {
				region = france_region
				has_owner_religion = no
				unrest = 5
			}
		}
	}

	on_start = french_wars_of_religion.1
	on_end = french_wars_of_religion.100
	
	on_monthly = {
		random_events = {
			1000 = 0
			100 = french_wars_of_religion.2 # The St. Bartholomew's Day Massacre
			100 = french_wars_of_religion.5 # The Assassination of the Duke of Guise
			
			100 = french_wars_of_religion.50 # Tumult of Amboise
			100 = french_wars_of_religion.51 # The Surprise of Meaux
			100 = french_wars_of_religion.52 # The Heretic Succession
			
			100 = french_wars_of_religion.53 # The Catholic League
			100 = french_wars_of_religion.54 # The Protestant League

			100 = french_wars_of_religion.55 # Day of the Barricades
			100 = french_wars_of_religion.56 # Day of the Barricades

			100 = french_wars_of_religion.57 # The Assassination of the Duke of Guise
			100 = french_wars_of_religion.58 # The Assassination of the Prince of Conde
		}
	}
}
