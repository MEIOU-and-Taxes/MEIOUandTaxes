
richardii_crisis = {
	potential = {
		always = no
		tag = ENG
		NOT = { has_country_flag = had_Richard_ii_crisis }
		is_free_or_tributary_trigger = yes
		dynasty = "Plantagenet"
		has_global_flag = hundred_years_war
	}
	
	can_start = {
		is_lesser_in_union = no
		OR = {
			has_regency = yes
			AND = {
				ruler_age = 30
				has_heir = no
			}
		}
	}
	
	can_stop = {
		NOT = { government = monarchy }
	}
	
	progress = {
		modifier = {
			factor = 1
			NOT = { legitimacy = 90 }
		}
		modifier = {
			factor = 1
			has_heir = no
		}
		modifier = {
			factor = 1
			NOT = { stability = 1 }
		}
		modifier = {
			factor = 0.5
			NOT = { ADM = 4 }
		}
		modifier = {
			factor = 0.5
			NOT = { DIP = 4 }
		}
		modifier = {
			factor = 0.5
			NOT = { MIL = 4 }
		}
		modifier = {
			factor = -1
			stability = 3
		}
	}
	
	can_end = {
		OR = {
			NOT = { has_ruler_flag = richard_ii }
			NOT = { has_heir_flag = richard_ii }
		}
	}
	
	modifier = {
		global_unrest = 2
		stability_cost_modifier = 0.25
		legitimacy = -0.1
	}
	
	on_start = richardii_crisis.1 #Tension between an incrasingly autocratic king and his nervous nobility
	on_end = richardii_crisis.100 #Henry IV of house lancaster decides to usurp the throne, regardless whether he wins or not, Lancaster becomes part of crown
	
	on_monthly = {
		events = {
		}
		random_events = {
			90 = 0
			3 = richardii_crisis.10 #Earl of Arundel hears about an rumour
			8 = richardii_crisis.11 #Richard II demands judges to grant him unlimited power
			5 = richardii_crisis.12 #Extended truce with France
			5 = richardii_crisis.13 #Marriage with French princess
			5 = richardii_crisis.14 #Richard's struggle with his uncle
			5 = richardii_crisis.15 #Wonderful Parliament of 1386
			5 = richardii_crisis.15 #Lords Appellant
			5 = richardii_crisis.15 #Rigid cout culture
			5 = richardii_crisis.15 #Strong belief in the royal prerogative
		}
	}
}
