#If England pisses off French too much, it will start
uef_french_rebellion = {
	potential = {
		OR = {
			AND = {
				tag = ENG 
				OR = {
					has_global_flag = plantagenet_becomes_french_again
					has_global_flag = plantagenet_french_regent
				}
			}
			AND = {
				tag = UEF
				OR = {
					has_global_flag = plantagenet_becomes_french_again
					has_global_flag = plantagenet_french_regent
				}
			}
		}
		FRA = {
			OR = {
				is_subject_of_type_with_overlord = {
					who = ENG
					type = uef_permanent_union
				}
				is_subject_of_type_with_overlord = {
					who = ENG
					type = permanent_union_limited
				}
			}
		}
		has_global_flag = hundred_years_war_ENG_win
		NOT = { has_country_flag = uef_french_rebellion_happened }
		is_free_or_tributary_trigger = yes
		is_lesser_in_union = no
	}
	
	can_start = {
		#has_any_disaster = no
		OR = {
			NOT = { stability = 2 }
			NOT = {  prestige = 0 }
			war_exhaustion = 5
			NOT = { legitimacy = 75 }
			AND = {
				FRA = {
					exists = yes
					liberty_desire = 50
				}
			}
			REB = { check_variable = { which = french_rebellion_desire value = 10 } }
		}
	}
	
	
	can_stop = {
		OR = {
			has_country_flag = uef_french_rebellion_ending_1
			has_country_flag = uef_french_rebellion_ending_2
		}
	}
	
	
	progress = {
		modifier = {
			factor = 0.2
			NOT = { legitimacy = 75 }
		} 
		modifier = {
			factor = 0.2
			NOT = { stability = 0 }
		}
		modifier = {
			factor = 0.5
			AND = {
				primary_culture = anglois
				has_idea = lingua_franca
			}
		}
		modifier = {
			factor = 0.25
			AND = {
				NOT = { primary_culture = anglois }
				has_idea = lingua_franca
			}
		}
		modifier = {
			factor = 0.2
			war_exhaustion = 5
		} 
		modifier = {
			factor = 1.5
			has_global_flag = convert_to_anglois_denied
		} 
		modifier = {
			factor = 1
			REB = { check_variable = { which = french_rebellion_desire value = 10 } }
		}
		modifier = {
			factor = 1.5
			REB = { check_variable = { which = french_rebellion_desire value = 15 } }
		}
		modifier = {
			factor = 1.5
			REB = { check_variable = { which = french_rebellion_desire value = 20 } }
		}
		modifier = {
			factor = 4
			REB = { check_variable = { which = french_rebellion_desire value = 25 } }
		}
	}
	
	modifier = {
		stability_cost_modifier = 0.75
		global_unrest = 10
	}	
	
	can_end = {
		OR = {
			has_country_flag = uef_french_rebellion_ending_1
			has_country_flag = uef_french_rebellion_ending_2
		}
		NOT = { num_of_rebel_controlled_provinces = 1 }
	}

	
	on_start = uef_french_rebellion.1
	on_end = uef_french_rebellion.100
	
	on_monthly = {
		events = {
			
		}
		
		random_events = {
			#1000 = 0
		}
	}
}
