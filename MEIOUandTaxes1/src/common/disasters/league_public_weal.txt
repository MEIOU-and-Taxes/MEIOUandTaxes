league_public_weal = {
	potential = {
		tag = FRA
		NOT = { has_country_flag = had_league_public_weal }
		has_global_flag = hundred_years_war_FRA_win
		any_subject_country = {
			is_subject_of_type = nominal_subject
		}
		is_free_or_tributary_trigger = yes
		capetian_dynasty_trigger = yes
		government = monarchy
	}
	

	can_start = {
		is_at_war = no
		has_any_disaster = no
		calc_true_if = {
			all_subject_country = {
				OR = {
					is_subject_of_type = vassal
					is_subject_of_type = appanage_subject
					is_subject_of_type = daimyo_vassal
				}
			}
			amount = 5
		}
		is_subject = no
	}

	can_stop = {
		OR = {
			has_any_disaster = yes
			NOT = {
				calc_true_if = {
					all_subject_country = {
						OR = {
							is_subject_of_type = vassal
							is_subject_of_type = appanage_subject
							is_subject_of_type = daimyo_vassal
						}
					}
					amount = 3
				}
			}
			is_subject = yes
			NOT = { government = monarchy }
		}
	}

	progress = {
		modifier = {
			factor = 3
			NOT = { legitimacy = 90 }
		} 
		modifier = {
			factor = 1.5
			NOT = { stability = 0 }
		}
		modifier = {
			factor = 1.5
			war_exhaustion = 3
		}
		modifier = {
			factor = 1.5
			NOT = { prestige = 50 }
		}
	}	

	can_end = {
		stability = 1
		all_subject_country = {
			NOT = { tag = BUR }
			NOT = {
				liberty_desire = 50
			}
		}
		custom_trigger_tooltip = {
			tooltip = league_public_weal.tooltip
			NOT = {
				is_in_war = {
					defender_leader = FRA
					casus_belli = cb_league_public_weal
				}
			}
		}
	}

	modifier = {
		diplomatic_reputation = -2
	}	

	on_start = league_public_weal.1
	on_end = league_public_weal.100

	on_monthly = {
		random_events = { 
			100 = 0
		}
	}
}
