#Regardless whether Eng goes with Prostant/Reform, or stays loyal to Papacy,
#France will make the opposite choice and inflame the tension.
#Ultimately breaking the union. This will be the last one to do so
#It's my belief that this should be end of the union should such thing ever existed
uef_religious_civil_war = {
	potential = {
		OR = {
			AND = {
				has_global_flag = plantagenet_becomes_french_again
				tag = UEF
			}
			AND = {
				has_global_flag = plantagenet_french_regent
				tag = ENG
			}
		}
		NOT = { has_country_flag = uef_religious_civil_war_happened }
	}
	
	can_start = {
		#has_any_disaster = no
		OR = {
			NOT = { stability = 2 }
			NOT = {  prestige = 0 }
			war_exhaustion = 5
			NOT = { legitimacy = 75 }
			AND = {
				FRA = {
					exists = yes
					liberty_desire = 50
				}
			}
			REB = { check_variable = { which = french_rebellion_desire value = 10 } }
		}
	}
	
	
	can_stop = {
		OR = {
			has_country_flag = uef_religious_civil_war_ending_1
			has_country_flag = uef_religious_civil_war_ending_2
		}
	}
	
	
	progress = {
		modifier = {
			factor = 1
			NOT = { legitimacy = 90 }
		} 
		modifier = {
			factor = 1
			NOT = { stability = 0 }
		}
		modifier = {
			factor = 1.5
			AND = {
				primary_culture = anglois
				has_idea = lingua_franca
			}
		}
		modifier = {
			factor = 1
			AND = {
				NOT = { primary_culture = anglois }
				has_idea = lingua_franca
			}
		}
		modifier = {
			factor = 1
			war_exhaustion = 5
		} 
		modifier = {
			factor = 1.5
			has_global_flag = convert_to_anglois_denied
		}
		modifier = {
			factor = 1
			REB = { check_variable = { which = french_rebellion_desire value = 10 } }
		}
		modifier = {
			factor = 1.5
			REB = { check_variable = { which = french_rebellion_desire value = 15 } }
		}
		modifier = {
			factor = 1.5
			REB = { check_variable = { which = french_rebellion_desire value = 20 } }
		}
		modifier = {
			factor = 4
			REB = { check_variable = { which = french_rebellion_desire value = 25 } }
		}
		modifier = {
			factor = 1
			NOT = { religious_unity = 0.75 }
			religious_unity = 0.5
		} 		
		modifier = {
			factor = 2
			NOT = { religious_unity = 0.5 }
		}
		modifier = {
			factor = 2
			has_hostile_reformation_center = yes
		}
		modifier = {
			factor = -1
			has_friendly_reformation_center = yes
		}
	}
	
	modifier = {
		global_unrest = 20
	}	
	
	can_end = {
		OR = {
			has_country_flag = uef_religious_civil_war_ending_1
			has_country_flag = uef_religious_civil_war_ending_2
		}
		NOT = { num_of_rebel_controlled_provinces = 1 }
	}

	
	on_start = uef_religious_civil_war.1
	on_end = uef_religious_civil_war.100
	
	on_monthly = {
		events = {
			
		}
		
		random_events = {
			#1000 = 0
		}
	}
}
