sorting_setup_h = { #prov #temp #var
    set_variable = { which = $temp$ which = $quantity$ }
    set_variable = { which = $temp2$ which = $link$ }
    $prov$ = {
        set_variable = { which = $temp$ which = PREV }
        set_variable = { which = $temp2$ which = PREV }
    }
}
sorting_sort_h = { #group #temp #var
    set_variable = { which = $temp$ value = 0 }
    set_variable = { which = $temp2$ value = 0 }
    save_event_target_as = Plague
    $group$ = {
        if = {
            limit = {
                check_variable = { which = $temp$ which = event_target:Plague }
                check_variable = { which = $temp$ value = 0.1 }
            }
            save_event_target_as = Plague
        }
    }
    set_variable = { which = $temp$ which = event_target:Plague }
    set_variable = { which = $temp2$ which = event_target:Plague }
    set_variable = { which = $quantity$ which = $temp$ }
    set_variable = { which = $link$ which = $temp2$ }    
    event_target:Plague = { set_variable = { which = $temp$ value = 0 } set_variable = { which = $temp2$ value = 0 } }     
}
