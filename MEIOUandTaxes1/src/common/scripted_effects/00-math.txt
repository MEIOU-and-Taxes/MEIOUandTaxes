# Example: magnitude_effect = { which = v_my_variable }
magnitude_effect = {
	if = {
		limit = {
			NOT = {
				check_variable = { which = $which$ value = 0 }
			}
		}

		multiply_variable = { which = $which$ value = -1 }
	}
}

# Example: sqrt_effect = { which = v_my_variable }
sqrt_effect = {
	if = {
		limit = {
			check_variable = { which = $which$ value = 0.001 }
		}

		set_variable = { which = tv_sqrt_precision value = 0.001 }
		set_variable = { which = tv_sqrt_length which = $which$ }
		set_variable = { which = tv_sqrt_width value = 1 }
		set_variable = { which = tv_sqrt_delta which = $which$ }
		subtract_variable = { which = tv_sqrt_delta value = 1 }

		while = {
			limit = {
				NOT = {
					check_variable = { which = tv_sqrt_precision which = tv_sqrt_delta }
				}
			}

			# tv_sqrt_length = (tv_sqrt_length + tv_sqrt_width) / 2
			change_variable = { which = tv_sqrt_length which = tv_sqrt_width }
			divide_variable = { which = tv_sqrt_length value = 2 }

			# tv_sqrt_width = $which$ / tv_sqrt_length
			set_variable = { which = tv_sqrt_width which = $which$ }
			divide_variable = { which = tv_sqrt_width which = tv_sqrt_length }

			distance_1d_effect = {
				which	= tv_sqrt_delta
				which_x1 = tv_sqrt_length
				which_x2 = tv_sqrt_width
			}
		}

		set_variable = { which = $which$ which = tv_sqrt_length }

		# cleanup
		set_variable = { which = tv_sqrt_precision value = 0 }
		set_variable = { which = tv_sqrt_length value = 0 }
		set_variable = { which = tv_sqrt_width value = 0 }
		set_variable = { which = tv_sqrt_delta value = 0 }
	}
	else = {
		set_variable = { which = $which$ value = 0 }
	}
}

	set_variable = { which = tv_power_base which = $which$ }

	set_variable = { which = tv_power_exponent which = $exponent$ }

	while = {
		limit = {
			check_variable = { which = tv_power_exponent value = 2 }
		}

		subtract_variable = { which = tv_power_exponent value = 1 }

		multiply_variable = { which = $which$ which = tv_power_base }
	}

	subtract_variable = { which = tv_power_exponent value = 1 }

	ln_funct = {
		type = which
		inp = tv_power_base
	}

	multiply_variable = { which = tv_power_exponent which = ln_val }

		type = which

# Input/Output:
# - $out_significand$
# - $out_exponent$
# Hashcode: 5B0160E1
hp_normalize_effect = {
	if = {
		limit = {
			is_variable_equal = { which = $out_significand$ value = 0 }
		}

		set_variable = { which = $out_exponent$ value = 0 }
	}
	else = {
		set_variable = { which = 5B0160E1__quotient value = 2147483 }

		divide_variable = { which = 5B0160E1__quotient which = $out_significand$ }

		magnitude_effect = {
			which = 5B0160E1__quotient
		}

		while = {
			limit = {
				check_variable = { which = 5B0160E1__quotient value = 10 }
			}

			multiply_variable = { which = $out_significand$ value = 10 }

			subtract_variable = { which = $out_exponent$ value = 1 }

			divide_variable = { which = 5B0160E1__quotient value = 10 }
		}

		set_variable = { which = 5B0160E1__quotient value = 0 }
	}
}

# Input:
# - $which_significand_type$
# - $which_significand_value$
# - $which_exponent_type$
# - $which_exponent_value$
# - $rhs_significand_type$
# - $rhs_significand_value$
# - $rhs_exponent_type$
# - $rhs_exponent_value$
# Output:
# - $out_significand$
# - $out_exponent$
# Hashcode: AA1343C3
hp_sum_effect = {
	set_variable = { which = AA1343C3__which_significand $which_significand_type$ = $which_significand_value$ }
	set_variable = { which = AA1343C3__which_exponent $which_exponent_type$ = $which_exponent_value$ }
	set_variable = { which = AA1343C3__rhs_significand $rhs_significand_type$ = $rhs_significand_value$ }
	set_variable = { which = AA1343C3__rhs_exponent $rhs_exponent_type$ = $rhs_exponent_value$ }

	if = {
		limit = {
			NOT = {
				check_variable = { which = AA1343C3__which_exponent which = AA1343C3__rhs_exponent }
			}
		}

		while = {
			limit = {
				NOT = {
					check_variable = { which = AA1343C3__which_exponent which = AA1343C3__rhs_exponent }
				}
			}

			divide_variable = { which = AA1343C3__which_significand value = 10 }

			change_variable = { which = AA1343C3__which_exponent value = 1 }
		}
	}
	else_if = {
		limit = {
			NOT = {
				check_variable = { which = AA1343C3__rhs_exponent which = AA1343C3__which_exponent }
			}
		}

		while = {
			limit = {
				NOT = {
					check_variable = { which = AA1343C3__rhs_exponent which = AA1343C3__which_exponent }
				}
			}

			divide_variable = { which = AA1343C3__rhs_significand value = 10 }

			change_variable = { which = AA1343C3__rhs_exponent value = 1 }
		}
	}

	set_variable = { which = $out_significand$ which = AA1343C3__which_significand }

	change_variable = { which = $out_significand$ which = AA1343C3__rhs_significand }

	set_variable = { which = $out_exponent$ which = AA1343C3__which_exponent }

	# Fix over- and underflow by increasing the exponent, reducing the
	# significand and repeating the addition of the two values
	if = {
		limit = {
			OR = {
				# Overflow
				AND = {
					check_variable = { which = AA1343C3__which_significand value = 0.001 }
					check_variable = { which = AA1343C3__rhs_significand value = 0.001 }
					NOT = {
						check_variable = { which = $out_significand$ value = 0.002 }
					}
				}
				# Underflow
				AND = {
					NOT = {
						check_variable = { which = AA1343C3__which_significand value = 0 }
					}
					NOT = {
						check_variable = { which = AA1343C3__rhs_significand value = 0 }
					}
					check_variable = { which = $out_significand$ value = -0.001 }
				}
			}
		}

		divide_variable = { which = AA1343C3__which_significand value = 10 }

		divide_variable = { which = AA1343C3__rhs_significand value = 10 }

		set_variable = { which = $out_significand$ which = AA1343C3__which_significand }

		change_variable = { which = $out_significand$ which = AA1343C3__rhs_significand }

		change_variable = { which = $out_exponent$ value = 1 }
	}

	hp_normalize_effect = {
		out_significand = $out_significand$
		out_exponent	= $out_exponent$
	}

	set_variable = { which = AA1343C3__which_significand value = 0 }
	set_variable = { which = AA1343C3__which_exponent value = 0 }
	set_variable = { which = AA1343C3__rhs_significand value = 0 }
	set_variable = { which = AA1343C3__rhs_exponent value = 0 }
}

# range fn
range_fn = {
	set_variable = {	which = tmp_0	value = $start$	}
	set_variable = {	which = tmp_1	value = $end$ }
	set_variable = {	which = tmp_2	value = $step$ }

	# when positive step, needs to be = end > start
	if = {
		limit = {
			check_variable = { which = tmp_2 value = 0 }
			check_variable = { which = tmp_1 which = tmp_0 }
		}
		while = {
			limit = {
				check_variable = { which = tmp_1 which = tmp_0 }
			}
			$effect$ = yes
			change_variable = { which = tmp_0	which = tmp_2 }
		}
	}
	# when negative step, needs to be = start > end
	else_if = {
		limit = {
			check_variable = { which = tmp_0 which = tmp_1 }
		}
		while = {
			limit = {
				check_variable = { which = tmp_0 which = tmp_1 }
			}
			$effect$ = yes
			change_variable = { which = tmp_0	which = tmp_2 }
		}
	}
}
