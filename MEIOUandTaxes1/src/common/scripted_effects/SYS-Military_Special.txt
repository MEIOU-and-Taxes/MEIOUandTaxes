Military_DoSpecialReact = {
} # Code deaths and wealth distrib from Special MP
Military_DoSpecial = {
	set_variable = { which = Teuton_MPTax value = 0 }
	set_variable = { which = Knights_MPTax value = 0 }
	every_province = {
		limit = {
			isValidProv = yes
		}
		set_variable = { which = Teuton_MP value = 0 }
		set_variable = { which = Knights_MP value = 0 }
		if = {
			limit = {
				exists = TEU
			}
			set_variable = { which = Special_MPTax value = 0 }
			set_variable = { which = Teuton_MPTax value = 0 }
			set_variable = { which = Tmp_0 value = 0 }
			set_variable = { which = Tmp_1 value = 0 }
			trigger_switch = {
				on_trigger = has_province_modifier

				Teuton_Commandery_Low = {
					set_variable = { which = Tmp_0 value = 0.003 }
					set_variable = { which = Tmp_1 value = 0.04 }
				}
				Teuton_Commandery_Low_trade = {
					set_variable = { which = Tmp_0 value = 0.003 }
					set_variable = { which = Tmp_1 value = 0.04 }
				}
				Teuton_Commandery_Med = {
					set_variable = { which = Tmp_0 value = 0.007 }
					set_variable = { which = Tmp_1 value = 0.14 }
				}
				Teuton_Commandery_Med_trade = {
					set_variable = { which = Tmp_0 value = 0.007 }
					set_variable = { which = Tmp_1 value = 0.14 }
				}
				Teuton_Commandery_High = {
					set_variable = { which = Tmp_0 value = 0.014 }
					set_variable = { which = Tmp_1 value = 0.28 }
				}
				Teuton_Commandery_High_trade = {
					set_variable = { which = Tmp_0 value = 0.014 }
					set_variable = { which = Tmp_1 value = 0.28 }
				}
			}
			if = {
				limit = {
					check_variable = { which = Tmp_0 value = 0.001 }
				}
				multiply_variable = { which = Tmp_0 which = SF_MP }
				multiply_variable = { which = Tmp_1 which = NO_MP }
				change_variable = { which = Teuton_MPTax which = Tmp_0 }
				change_variable = { which = Teuton_MPTax which = Tmp_1 }
			}
			PREV = { change_variable = { which = Teuton_MPTax which = PREV } }
		}
		else = {
			set_variable = { which = Teuton_MPTax value = 0 }
		}
		if = {
			limit = {
				exists = KNI
			}
			set_variable = { which = Knights_MPTax value = 0 }
			set_variable = { which = Tmp_0 value = 0 }
			set_variable = { which = Tmp_1 value = 0 }
			trigger_switch = {
				on_trigger = has_province_modifier

				Knights_Commandery_Low = {
					set_variable = { which = Tmp_0 value = 0.001 }
					set_variable = { which = Tmp_1 value = 0.018 }
				}
				Knights_Commandery_Low_trade = {
					set_variable = { which = Tmp_0 value = 0.001 }
					set_variable = { which = Tmp_1 value = 0.018 }
				}
				Knights_Commandery_Med = {
					set_variable = { which = Tmp_0 value = 0.003 }
					set_variable = { which = Tmp_1 value = 0.07 }
				}
				Knights_Commandery_Med_trade = {
					set_variable = { which = Tmp_0 value = 0.003 }
					set_variable = { which = Tmp_1 value = 0.07 }
				}
				Knights_Commandery_High = {
					set_variable = { which = Tmp_0 value = 0.007 }
					set_variable = { which = Tmp_1 value = 0.15 }
				}
				Knights_Commandery_High_trade = {
					set_variable = { which = Tmp_0 value = 0.007 }
					set_variable = { which = Tmp_1 value = 0.15 }
				}
			}
			if = {
				limit = {
					check_variable = { which = Tmp_0 value = 0.001 }
				}
				multiply_variable = { which = Tmp_0 which = SF_MP }
				multiply_variable = { which = Tmp_1 which = NO_MP }
				change_variable = { which = Knights_MPTax which = Tmp_0 }
				change_variable = { which = Knights_MPTax which = Tmp_1 }
			}
			PREV = { change_variable = { which = Knights_MPTax which = PREV } }
		}
		else = {
			set_variable = { which = Knights_MPTax value = 0 }
		}
		set_variable = { which = Teuton_MP value = 0 }
		set_variable = { which = Knights_MP value = 0 }
	}
	set_variable = { which = Ghazi_MPTax value = 0 }
	set_variable = { which = Ghazi_MP value = 0 }
	every_province = {
		limit = {
			superregion = anatolia_superregion
			isValidProv = yes
		}
		set_variable = { which = Tmp_0 value = 0 }
		if = {
			limit = {
				exists = OTT
				OTT = { has_country_modifier = GhaziGain }
			}
			if = {
				limit = {
					owned_by = OTT
				}
				set_variable = { which = Ghazi_MP value = 0 }
				set_variable = { which = Ghazi_MPTax value = 0 }
				set_variable = { which = Tmp_0 which = Dev_Total }
			}
			else = {
				set_variable = { which = Ghazi_MP value = 0 }
				set_variable = { which = Tmp_0 value = 0.02 } # 2%
				multiply_variable = { which = Tmp_0 which = NM_MP }
				change_variable = { which = Ghazi_MPTax which = Tmp_0 }
				PREV = { change_variable = { which = Ghazi_MPTax which = PREV } }
			}
		}
		else = {
			if = {
				limit = {
					NOT = { owned_by = OTT }
				}
				set_variable = { which = Ghazi_MP value = 0 }
				multiply_variable = { which = Ghazi_MPTax value = 0.75 }
				if = {
					limit = {
						NOT = { check_variable = { which = Ghazi_MPTax value = 0.1 } }
					}
					set_variable = { which = Ghazi_MPTax value = 0 }
				}
			}
			else = {
				set_variable = { which = Ghazi_MPTax value = 0 }
				multiply_variable = { which = Ghazi_MP value = 0.75 }
				if = {
					limit = {
						NOT = { check_variable = { which = Ghazi_MP value = 0.1 } }
					}
					set_variable = { which = Ghazi_MP value = 0 }
				}
			}
		}
	}
	set_variable = { which = Tmp_1 which = Tmp_0 }
	if = {
		limit = {
			exists = OTT
			OTT = { has_country_modifier = GhaziGain }
			check_variable = { which = Ghazi_MPTax value = 0.001 }
		}
		OTT = { set_variable = { which = Ghazi_MP value = 0 } }
		every_province = {
			limit = {
				superregion = anatolia_superregion
				owned_by = OTT
				isValidProv = yes
				check_variable = { which = Tmp_0 value = 0.001 }
				PREV = { check_variable = { which = Tmp_1 value = 0.001 } }
			}
			set_variable = { which = Ghazi_MPTax which = PREV }
			set_variable = { which = Ghazi_MP which = Ghazi_MPTax }
			set_variable = { which = Ghazi_MPTax value = 0 }

			set_variable = { which = Tmp_1 which = PREV }
			divide_variable = { which = Tmp_0 which = Tmp_1 }

			multiply_variable = { which = Ghazi_MP which = Tmp_0 }
			OTT = { change_variable = { which = Ghazi_MP which = PREV } }
		}
	}
	set_variable = { which = Teuton_MP which = Teuton_MPTax }
	set_variable = { which = Knights_MP which = Knights_MPTax }
	if = {
		limit = {
			exists = TEU
			exists = LVO
		}
		set_variable = { which = Tmp_0 which = Teuton_MP }
		multiply_variable = { which = Teuton_MP value = 0.85 }
		TEU = {
			set_variable = { which = Teuton_MP which = PREV }
			capital_scope = {
				set_variable = { which = Teuton_MP which = PREV }
			}
		}
		set_variable = { which = Teuton_MP which = Tmp_0 }
		multiply_variable = { which = Teuton_MP value = 0.15 }
		LVO = {
			set_variable = { which = Teuton_MP which = PREV }
			capital_scope = {
				set_variable = { which = Teuton_MP which = PREV }
			}
		}
	}
	else_if = {
		limit = {
			exists = TEU
		}
		TEU = {
			set_variable = { which = Teuton_MP which = PREV }
			capital_scope = {
				set_variable = { which = Teuton_MP which = PREV }
			}
		}
	}
	else_if = {
		limit = {
			exists = LVO
		}
		LVO = {
			set_variable = { which = Teuton_MP which = PREV }
			capital_scope = {
				set_variable = { which = Teuton_MP which = PREV }
			}
		}
	}
	KNI = {
		set_variable = { which = Knights_MP which = PREV }
		capital_scope = {
			set_variable = { which = Knights_MP which = PREV }
		}
	}
	every_province = {
		limit = {
			isValidProv = yes
		}
		set_variable = { which = Special_MP value = 0 }
		set_variable = { which = Special_Wealth value = 0 }
		change_variable = { which = Special_MP which = Ghazi_MP }
		change_variable = { which = Special_MP which = Teuton_MP }
		change_variable = { which = Special_MP which = Knights_MP }
		if = {
			limit = {
				check_variable = { which = Ghazi_MP value = 0.001 }
			}
			set_variable = { which = Ghazi_MPShare which = Ghazi_MP }
			divide_variable = { which = Ghazi_MPShare which = Special_MP }
			multiply_variable = { which = Ghazi_MPShare value = 100 }
		}
		if = {
			limit = {
				check_variable = { which = Teuton_MP value = 0.001 }
			}
			set_variable = { which = Teuton_MPShare which = Teuton_MP }
			divide_variable = { which = Teuton_MPShare which = Special_MP }
			multiply_variable = { which = Teuton_MPShare value = 100 }
		}
		if = {
			limit = {
				check_variable = { which = Knights_MP value = 0.001 }
			}
			set_variable = { which = Knights_MPShare which = Knights_MP }
			divide_variable = { which = Knights_MPShare which = Special_MP }
			multiply_variable = { which = Knights_MPShare value = 100 }
		}
		set_variable = { which = Special_Total which = Special_MP }
	}
}
