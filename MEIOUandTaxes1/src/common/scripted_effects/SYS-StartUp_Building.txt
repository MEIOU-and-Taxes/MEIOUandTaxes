StartUp_InitBuilding = {
    regiongroup = {
        region = {
            limit = {
                isValidEmpty = yes
            }
			set_variable = { which = Building_Agriculture value = 1 }
			if = {
				limit = {
					check_variable = { which = SF_Total value = 1 }
				}
				set_variable = { which = Building_Forestry value = 1 }
				set_variable = { which = Building_Extraction value = 1 }
				set_variable = { which = Building_Fishery value = 1 }
			}
			else = {
				set_variable = { which = Building_Forestry value = 0 }

				set_variable = { which = Building_Fishery value = 0 }
			}
            set_variable = { which = Building_Commercial value = 0 }
            set_variable = { which = Building_Industrial value = 0 }
            set_variable = { which = Building_Academic value = 0 }

			# Commercial
            set_variable = { which = Building_Commercial which = BG_Total }
            multiply_variable = { which = Building_Commercial value = 66 }

			if = {
				limit = {
					check_variable = { which = CL_Total value = 0.1 }
                }
                set_variable = { which = Building_Academic which = CL_Total }
                multiply_variable = { which = Building_Academic value = 16 }

            }
			if = {
				limit = {
					check_variable = { which = RE_Total value = 0.5 }
                }
                trigger_switch = {
                    on_trigger = has_province_flag

                    small_university = {
                        change_variable = { which = Building_Academic value = 10 }
                        if = {
                            limit = {
                                check_variable = { which = LT_Total value = 0.1 }
                            }
                            set_variable = { which = Tmp_0 which = LT_Total }
                            multiply_variable = { which = Tmp_0 value = 25 }
                            change_variable = { which = Building_Academic which = Tmp_0 }

                    medium_university = {

                            limit = {
                                check_variable = { which = LT_Total value = 0.1 }
                            }
                            set_variable = { which = Tmp_0 which = LT_Total }

                            change_variable = { which = Building_Academic which = Tmp_0 }
                        }
                    }
                    large_university = {
                        change_variable = { which = Building_Academic value = 15 }
                        if = {
                            limit = {
                                check_variable = { which = LT_Total value = 0.1 }
                            }
                            set_variable = { which = Tmp_0 which = LT_Total }
                            multiply_variable = { which = Tmp_0 value = 25 }
                            change_variable = { which = Building_Academic which = Tmp_0 }

                }

					check_variable = { which = Building_Academic value = 0.1 }
                }
			    divide_variable = { which = Building_Academic value = 2 }
            }

            set_variable = { which = Tmp_0 which = RE_Total }
            multiply_variable = { which = Tmp_0 value = 1.75 }

            change_variable = { which = Building_Industrial which = Tmp_0 }

			if = {
				limit = {
					check_variable = { which = SF_Total value = 5 }
					isValidProv = yes
				}
				set_variable = { which = Tmp_0 which = SF_Total }
                if = {
                    limit = {
                        continent = indian_continent
                    }
                    divide_variable = { which = Tmp_0 value = 16 }
                }
                else = {
                    divide_variable = { which = Tmp_0 value = 17 }
                }

				change_variable = { which = Building_Industrial which = Tmp_0 }
			}

            if = {
                limit = {
                    OR = {
                        continent = indian_continent
                        superregion = central_america_superregion
                    }
                }
                set_variable = { which = Tmp_0 which = SF_Total }
                divide_variable = { which = Tmp_0 value = 2.5 }
            }
            else_if = {
                limit = {
                    OR = {
                        continent = asia
                        superregion = southeast_asia_superregion
                        superregion = japan_superregion
                        superregion = east_indies_superregion
                    }

                limit = {
                    superregion = western_south_america_superregion
                }
                set_variable = { which = Tmp_0 which = SF_Total }

            }

                        superregion = persia_superregion

                        trade_goods = gems

                        has_province_flag = gems
                        has_province_flag = marble
                        has_province_flag = silver
                        has_province_flag = gold
                        has_province_flag = obsidian
                    }

					NOT = {
						has_port = yes

						AND = {
							NOT = {
								has_province_flag = no_mines
							}

							OR = {
								has_province_flag = m
								has_province_flag = salt
							}
						}
					}

						has_province_flag = coal
						has_province_flag = no_mines

						isValidProv = no
					}

                set_variable = { which = Infra_Harbourage value = 0 }

            if = { 
                limit = { NOT = { check_variable = { which = RE_Total value = 0.001 } } }
                set_variable = { which = Infra_Amenities value = 0 }

                limit = { Infra_CanIrrigation = no }
                set_variable = { which = Infra_Irrigation value = 0 }

                limit = { NOT = { check_variable = { which = RE_Total value = 0.001 } } }
                set_variable = { which = Infra_Capitol value = 0 }
            }
            if = { 
                limit = { Infra_CanGarrison = no }
                set_variable = { which = Infra_Garrison value = 0 }
            }
            set_variable = { which = Infra_Courthouse value = 0 }
            if = {
                limit = {
                    check_variable = { which = RE_Total value = 1250 }
                }
                set_variable = { which = Infra_Amenities value = 6 }
            }
            else_if = {
                limit = {
                    check_variable = { which = RE_Total value = 350 }
                }
                set_variable = { which = Infra_Amenities value = 5 }
            }
            else_if = {
                limit = {
                    check_variable = { which = RE_Total value = 140 }
                }
                set_variable = { which = Infra_Amenities value = 4 }
            }
            else_if = {
                limit = {
                    check_variable = { which = RE_Total value = 50 }
                }
                set_variable = { which = Infra_Amenities value = 3 }
            }
            else_if = {
                limit = {
                    check_variable = { which = RE_Total value = 20 }
                }
                set_variable = { which = Infra_Amenities value = 2 }
            }
            else_if = {
                limit = {
                    check_variable = { which = RE_Total value = 5 } # 5k is often the town/city brink point
                }
                set_variable = { which = Infra_Amenities value = 1 }
            }

            if = {
                limit = {
                    has_province_modifier = lack_of_harbour

                set_variable = { which = Infra_Harbourage value = 0 }

            if = {
                limit = {
                    check_variable = { which = Infra_Pathing value = 1 }

                subtract_variable = { which = Infra_Pathing value = 1 }

                Infra_GetGoalPathing = { inp = Infra_Pathing return = Infra_PathingSize }

                change_variable = { which = Infra_Pathing value = 1 }

                Infra_GetGoalPathing = { inp = Infra_Pathing return = Infra_PathingGoal }

                change_variable = { which = Infra_PathingSize which = Infra_PathingGoal }

				multiply_variable = { which = Infra_PathingSize value = 0.001 }
                multiply_variable = { which = Infra_PathingSize value = 1000 }

                multiply_variable = { which = Infra_HarbourageSize value = 3 }
                change_variable = { which = Infra_HarbourageSize which = Infra_HarbourageGoal }
                divide_variable = { which = Infra_HarbourageSize value = 4 }

				multiply_variable = { which = Infra_HarbourageSize value = 0.001 }

                change_variable = { which = Infra_Capitol value = 1 }
