Tax_GetPopScalar = {
	if = {
		limit = {
			check_variable = { which = $TotalVar$ value = 1 }
		}
		set_variable = { which = TaxMana_Tmp4 which = $TotalVar$ }
		if = {
			limit = {
				check_variable = { which = $TotalVar$ value = $MaxPop$ }
			}
			set_variable = { which = TaxMana_Tmp4 which = $TotalVar$ }
			multiply_variable = { which = TaxMana_Tmp4 value = $ReductionPer$ }
		}
		else = {
			set_variable = { which = TaxMana_Tmp4 which = $TotalVar$ }
			divide_variable = { which = TaxMana_Tmp4 value = $MaxPop$ }
			multiply_variable = { which = TaxMana_Tmp4 which = TaxMana_Tmp4 }
			multiply_variable = { which = TaxMana_Tmp4 value = -1 }
			change_variable = { which = TaxMana_Tmp4 value = 1 }
			set_variable = { which = TaxMana_Tmp3 value = 1 }
			subtract_variable = { which = TaxMana_Tmp3 value = $ReductionPer$ }
			multiply_variable = { which = TaxMana_Tmp4 which = TaxMana_Tmp3 }
			change_variable = { which = TaxMana_Tmp4 value = $ReductionPer$ }
			multiply_variable = { which = TaxMana_Tmp4 which = $TotalVar$ }
		}
		multiply_variable = { which = TaxMana_Tmp4 which = Modi_ManaCostFromPop }
		change_variable = { which = $return$ which = TaxMana_Tmp4 }
	}
	else = {
		change_variable = { which = $return$ value = 1 }
	}
}

Tax_CalcCostTax = {
	if = {
		limit = {
			owner = { Tax_$TaxName$Can = yes }
		}
		Tax_Cost$TaxName$Calc = yes
		multiply_variable = { which = Tax_Cost$TaxName$High value = 0.45 } ### GLOBAL TAX BALANCER HERE
		[[Balance] multiply_variable = { which = Tax_Cost$TaxName$High value = $Balance$ } ] ### TAX BALANCER HERE
		if = {
			limit = {
				check_variable = { which = Tax_Cost$TaxName$High value = 250 } ### Sanity check
			}
			set_variable = { which = Tax_Cost$TaxName$High value = 250 }
		}
		else_if = {
			limit = {
				NOT = { check_variable = { which = Tax_Cost$TaxName$High value = 0.05 } }
			}
			set_variable = { which = Tax_Cost$TaxName$High value = 0.05 }
		}
	}
	else = {
		set_variable = { which = Tax_Cost$TaxName$High value = 0 }
	}
}
