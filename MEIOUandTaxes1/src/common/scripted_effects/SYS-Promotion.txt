Promotion_REPromotion = {
	if = {
		limit = {
			check_variable = { which = RE_Total value = 0.1 } # There is a city
			NOT = { check_variable = { which = RE_Total which = Ideal_ResidentSize } } # Residents < IdealSize
			NOT = { is_variable_equal = { which = RE_FillLifeC value = 0 } }
		}
		set_variable = { which = Tmp_1 value = 0.02 }

		# Get RE LifeC fulfilment modifier: ((-RE_FillLifeC + 90) / -RE_FillLifeC) * 0.025
		set_variable = { which = Tmp_5 which = RE_FillLifeC }
		multiply_variable = { which = Tmp_5 value = -1 }
		change_variable = { which = Tmp_5 value = 90 }
		set_variable = { which = Tmp_6 which = RE_FillLifeC }
		multiply_variable = { which = Tmp_6 value = -1 }
		if = {
			limit = {
				check_variable = { which = Tmp_5 value = 0.001 }
			}
			divide_variable = { which = Tmp_5 which = Tmp_6 }
		}
		else = {
			set_variable = { which = Tmp_5 value = 0 }
		}
		multiply_variable = { which = Tmp_5 value = 0.025 }

		# Upper lower case handling: [-0.01,0.01]
		if = {
			limit = {
				check_variable = { which = Tmp_5 value = 0.01 }
			}
			set_variable = { which = Tmp_5 value = 0.01 }
		}
		else_if = {
			limit = {
				NOT = { check_variable = { which = Tmp_5 value = -0.01 } }
			}
			set_variable = { which = Tmp_5 value = -0.01 }
		}

		## Get Yearly Change
		change_variable = { which = Tmp_1 which = Tmp_5 } # RE LifeC Fulfill modifier

		multiply_variable = { which = Tmp_1 which = Modi_ResidentPromoteSpeed }

		# Get Difference to Ideal Pop
		set_variable = { which = Tmp_2 which = Ideal_ResidentSize }
		subtract_variable = { which = Tmp_2 which = RE_Total }

		multiply_variable = { which = Tmp_1 which = Tmp_2 } # Get end result: Rate * DifferenceToIdealPop

		# Safety check to ensure that promotion is always in the correct direction
		if = {
			limit = {
				NOT = { check_variable = { which = Tmp_1 value = 0 } }
			}
			set_variable = { which = Tmp_1 value = 0 }
		}
		# safety check to ensure promotion cannot exceed available source
		if = {
			limit = {
				check_variable = { which = Tmp_1 which = SF_Total }
			}
			set_variable = { which = Tmp_1 which = SF_Total  }
		}

		set_variable = { which = SF_PromotedRE which = Tmp_1 }
	}
}

Promotion_DeriveNobleRestingPoint = {
	if = { # Blood Nobility
		limit = {
			owner = {
				has_reform = Pol_Nobility1
				#is_variable_equal = { which = Pol_Nobility value = 0 } 
			}
		}
		set_variable = { which = Tmp_1 value = 0.0125 } # 1 % Nobility of Total Population with Blood nobility
	}
	else_if = { # Possession Nobility
		limit = {
			owner = { 
				has_reform = Pol_Nobility2
				#is_variable_equal = { which = Pol_Nobility value = 1 } 
			}
		}
		if = {
			limit = {
				check_variable = { which = SF_FillLifeC value = 1 }
			}
			# Get SF LifeC fulfilment modifier: (SF_FillLifeC * 0.5 + 0.5) * NO_Ratio * 0.01 in % as Nobility of Total Population
			set_variable = { which = Tmp_1 which = SF_FillLifeC }
			multiply_variable = { which = Tmp_1 value = 0.5 }
			change_variable = { which = Tmp_1 value = 0.5 }
			set_variable = { which = Tmp_2 which = NO_Ratio }
			multiply_variable = { which = Tmp_1 which = Tmp_2 }
			# Upper lower case handling: [1,10]
			if = {
				limit = {
					check_variable = { which = Tmp_1 value = 10 }
				}
				set_variable = { which = Tmp_1 value = 10 }
			}
			else_if = {
				limit = {
					NOT = { check_variable = { which = Tmp_1 value = 1 } }
				}
				set_variable = { which = Tmp_1 value = 1 }
			}
			multiply_variable = { which = Tmp_1 value = 0.01 }
		}
		else = { # if below 100% life fulfillment 1 % Nobility of Total Population
			set_variable = { which = Tmp_1 value = 0.01 }
		}
	}
	else_if = { # Office Nobility
		limit = {
			owner = {
				has_reform = Pol_Nobility3
				#is_variable_equal = { which = Pol_Nobility value = 2 } 
			}
		}

		# Get ideal size based of: (0.25 * StateReach + 5) * 0.001
		set_variable = { which = Tmp_1 value = 0.25 }
		multiply_variable = { which = Tmp_1 which = Prov_BUPow }
		change_variable = { which = Tmp_1 value = 5 }
		multiply_variable = { which = Tmp_1 value = 0.001 }
	}
	else = { # None
		set_variable = { which = Tmp_1 value = 0.005 }
	}

	# Noble Power Additive % - NO Power/1000 * 0.125
	set_variable = { which = Tmp_2 which = Prov_NOPow }
	divide_variable = { which = Tmp_2 value = 1000 }
	multiply_variable = { which = Tmp_2 value = 0.2 }

	change_variable = { which = Tmp_1 which = Tmp_2}

	# Get their intended % of total population
	set_variable = { which = Ideal_NobleSize which = Class_Total }
	multiply_variable = { which = Ideal_NobleSize which = Tmp_1 }
}

Promotion_NOPromotion = {
	if = { # Check if NO_Total is at least this
		limit = {
			NOT = { check_variable = { which = NO_Total which = Ideal_NobleSize } } # Nobles < Ideal Size
		}

		set_variable = { which = Tmp_0 value = 0.015 } # Base speed of change 1% of dif <- use this for balance

		# Get difference 
		set_variable = { which = Tmp_1 which = Ideal_NobleSize }
		subtract_variable = { which = Tmp_1 which = NO_Total }

		# Get NO LifeC fulfilment modifier: ((-NOFillLifeC + 0.8) / -NOFillLifeC) * 0.01
		set_variable = { which = Tmp_2 which = NO_FillLifeC }
		multiply_variable = { which = Tmp_2 value = -1 }
		change_variable = { which = Tmp_2 value = 0.8 }
		set_variable = { which = Tmp_3 which = NO_FillLifeC }
		multiply_variable = { which = Tmp_3 value = -1 }
		divide_variable = { which = Tmp_2 which = Tmp_3 }
		multiply_variable = { which = Tmp_2 value = 0.01 }

		# Upper lower case handling: [-0.01,0.02]
		if = {
			limit = {
				check_variable = { which = Tmp_2 value = 0.02 }
			}
			set_variable = { which = Tmp_2 value = 0.02 }
		}
		else_if = {
			limit = {
				NOT = { check_variable = { which = Tmp_2 value = -0.01 } }
			}
			set_variable = { which = Tmp_2 value = -0.01 }
		}

		# Get NO Comfort fulfilment modifier: ((-NOFillComfort + 0.6) / -NOFillComfort) * 0.01
		set_variable = { which = Tmp_3 which = NO_FillComfort }
		multiply_variable = { which = Tmp_3 value = -1 }
		change_variable = { which = Tmp_3 value = 0.6 }
		set_variable = { which = Tmp_4 which = NO_FillComfort }
		multiply_variable = { which = Tmp_4 value = -1 }
		divide_variable = { which = Tmp_3 which = Tmp_4 }
		multiply_variable = { which = Tmp_3 value = 0.01 }

		# Upper lower case handling: [-0.01,0.02]
		if = {
			limit = {
				check_variable = { which = Tmp_3 value = 0.02 }
			}
			set_variable = { which = Tmp_3 value = 0.02 }
		}
		else_if = {
			limit = {
				NOT = { check_variable = { which = Tmp_3 value = -0.01 } }
			}
			set_variable = { which = Tmp_3 value = -0.01 }
		}

		change_variable = { which = Tmp_0 which = Tmp_2 }
		change_variable = { which = Tmp_0 which = Tmp_3 }
	}
	else = {
		set_variable = { which = Tmp_0 value = 0 }
	}

	if = {
		limit = {
			check_variable = { which = Tmp_0 value = 0.001 }
		}

		multiply_variable = { which = Tmp_1 which = Modi_NobilityPromoteSpeed }
		multiply_variable = { which = Tmp_0 which = Tmp_1 }


		change_variable = { which = SF_PromotedNO which = Tmp_0 }
		change_variable = { which = NM_PromotedNO which = Tmp_0 }
		change_variable = { which = RE_PromotedNO which = Tmp_0 }
		change_variable = { which = NO_Promoted which = Tmp_0 }
	}

}

Promotion_GetNomadRestingPoint = { 
	if = {
		limit = {
			check_variable = { which = NM_Total value = 0.001 } # make sure Nomads are around 
		}
		if = {
			limit = {
				OR = {
					owner = {
						OR = {
							is_colonial_nation = yes
							is_former_colonial_nation = yes
							has_country_flag = changed_from_colonial_nation
						}
					}
					AND = {
						OR = {
							continent = south_america
							continent = north_america
						}
						owner = {
							capital_scope = {
								NOT = {
									continent = south_america
									continent = north_america
								}
							}
						}
					}
				}
			}
			set_variable = { which = Ideal_NomadSize value = 0 }
		}
		else = {
			set_variable = { which = Tmp_1 value = 100 }
			set_variable = { which = Tmp_2 which = Prov_BUPow } 
			subtract_variable = { which = Tmp_1 which = Tmp_2 } # e.g. 100 - 15 = 85 
			multiply_variable = { which = Tmp_1 value = 0.01 } #  e.g 85 x 0.01 = 0.85
			multiply_variable = { which = Tmp_1 which = NM_Total }
			set_variable = { which = Ideal_NomadSize which = Tmp_1 }
		}
	}
}
Promotion_NMPromotion = {
	if = {
		limit = { 
			check_variable = { which = NM_Total which = Ideal_NomadSize } #if Nomads Size -> 100k >= 80k
		}
		set_variable = { which = Tmp_3 value = 0.02 } # Base rate of change - use for balance

		# Get Difference between Total and BurgherSize
		set_variable = { which = Tmp_4 which = Ideal_NomadSize } # -> 80k
		subtract_variable = { which = Tmp_4 which = NM_Total } # -> 80k - 100k = -20k
		multiply_variable = { which = Tmp_4 value = -1 } # -> -20k * -1 = 20k

		#Needs new variable Modi_NomadPromoteSpeed which could be used for Russia or CNs national ideas 
		#multiply_variable = { which = Tmp_3 which = Modi_NomadPromoteSpeed }
		multiply_variable = { which = Tmp_3 which = Tmp_4 } # Get end result = Rate(Tmp3) * DifferenceToIdealPop (Tmp4) -> 20k x 0,02 = 400

		# safety check to ensure no wrong direction effect
		if = {
			limit = {
				NOT = { check_variable = { which = Tmp_3 value = 0 } } # 400 not smaller than 0, therefore ok
			}
			set_variable = { which = Tmp_3 value = 0 }
		}

		# safety check to ensure promotion cannot exceed available source
		if = {
			limit = {
				check_variable = { which = Tmp_3 which = NM_Total }
			}
			set_variable = { which = Tmp_3 which = NM_Total  }
		}
		set_variable = { which = NM_PromotedSF which = Tmp_3 }
	}
}

Promotion_DeriveBurgherRestingPoint = {
	if = {
		limit = {
			check_variable = { which = RE_Total value = 0.1 }
			has_province_flag = Building_CommercialHas
		}

		# Calc and apply multiplier based on labour supply vs. demand
		if = {
			limit = {
				check_variable = { which = Tmp_3 value = 0.001 }
			}
			set_variable = { which = Tmp_4 which = Tmp_3 } 
			divide_variable = { which = Tmp_4 which = WK_BG }
		}
		else = {
			set_variable = { which = Tmp_4 value = 0 } 
		}

		# Add API effects on base rate
		change_variable = { which = Tmp_4 which = Modi_IdealBurgherSizePerc_Add }
		multiply_variable = { which = Tmp_4 which = Modi_IdealBurgherSizePerc_Multi }

		# Calc absolute idea burgher pops
		set_variable = { which = Ideal_BurgherSize which = Tmp_4 }
		if = {
			limit = {
				check_variable = { which = BG_Total value = 0.01 } ## If there are less then 10 burghers as base, use 10 as lower limit
			}
			multiply_variable = { which = Ideal_BurgherSize which = BG_Total }
		}
		else = {
			multiply_variable = { which = Ideal_BurgherSize value = 0.01 }
		}

		# Check if ideal size is below additive limit, if so use additive limit as ideal size
		if = {
			limit = { 
				check_variable = { which = Modi_IdealBurgherSize_Add which = Ideal_BurgherSize }
			}
			set_variable = { which = Ideal_BurgherSize which = Modi_IdealBurgherSize_Add }
		}

		# Ensure that Burgher Ideal cannot be larger than resident size
		if = {
			limit = {
				check_variable = { which = Ideal_BurgherSize which = RE_Total }
			}
			set_variable = { which = Ideal_BurgherSize which = RE_Total }
		}
	}
}
