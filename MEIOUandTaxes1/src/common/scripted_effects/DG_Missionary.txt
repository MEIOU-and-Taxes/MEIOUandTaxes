MN_Missionary = {
	#Export global data
	export_to_variable = {
		 which = DG_Tmp1
		 value = religion
		 who = owner
	}
	export_to_variable = {
		which = DG_Tmp2
		value = modifier:global_missionary_strength
		who = owner
	}
	export_to_variable = {
		which = DG_Tmp3
		value = nationalism
	}
	export_to_variable = {
		which = DG_Tmp4
		value = modifier:tolerance_heretic
		who = owner
	}
	export_to_variable = {
		which = DG_Tmp5
		value = modifier:tolerance_heathen
		who = owner
	}
	export_to_variable = {
		which = DG_Tmp6
		value = unrest
	}
	export_to_variable = {
		which = DG_Tmp7
		value = modifier:local_missionary_maintenance_cost
	}
	#Performs the common calculation
	multiply_variable = { which = DG_Tmp2 value = 100 }
	divide_variable = { which = DG_Tmp3 value = 5 }
	subtract_variable = { which = DG_Tmp2 which = DG_Tmp3 }
	if = {
		limit = {
			is_pagan = yes
		}
		change_variable = { which = DG_Tmp2 value = 2.5 }
	}
	if = {
		limit = { NOT = { controlled_by = owner } }
		subtract_variable = { which = DG_Tmp2 value = 5.0 }
	}
	if = {
		limit = { NOT = { any_neighbor_province = { owned_by = ROOT } } }
		subtract_variable = { which = DG_Tmp2 value = 2.0 }
	}
	if = {
		limit = { NOT = { check_variable = { which = DG_Tmp6 value = 0 } } }
		set_variable = { which = DG_Tmp6 value = 0.0 }
	}
	if = {
		limit = { is_owned_by_trade_company = yes }
		subtract_variable = { which = DG_Tmp2 value = 4.0 }
	}
	else = {
		subtract_variable = { which = DG_Tmp2 which = DG_Tmp6 }
		if = {
			limit = {
				has_owner_accepted_culture = no
				has_owner_culture = no
			}
			subtract_variable = { which = DG_Tmp2 value = 6.0 }
		}
	}
	MN_MissionaryHelper = { slot = 0 }
	MN_MissionaryHelper = { slot = 1 }
	MN_MissionaryHelper = { slot = 2 }
	MN_MissionaryHelper = { slot = 3 }
	MN_MissionaryHelper = { slot = 4 }
	MN_MissionaryHelper = { slot = 5 }

	#Cleans the temporary variables
	set_variable = { which = DG_Tmp1 value = 0 }
	set_variable = { which = DG_Tmp2 value = 0 }
	set_variable = { which = DG_Tmp3 value = 0 }
	set_variable = { which = DG_Tmp4 value = 0 }
	set_variable = { which = DG_Tmp5 value = 0 }
	set_variable = { which = DG_Tmp6 value = 0 }
}

AvailableMissionariesCount = {
	if = {
		limit = { num_of_missionaries = 10 }
		set_variable = { which = AvailableMissionaries value = 10 }
	}
	else_if = {
		limit = { num_of_missionaries = 9 }
		set_variable = { which = AvailableMissionaries value = 9 }
	}
	else_if = {
		limit = { num_of_missionaries = 8 }
		set_variable = { which = AvailableMissionaries value = 8 }
	}
	else_if = {
		limit = { num_of_missionaries = 7 }
		set_variable = { which = AvailableMissionaries value = 7 }
	}
	else_if = {
		limit = { num_of_missionaries = 6 }
		set_variable = { which = AvailableMissionaries value = 6 }
	}
	else_if = {
		limit = { num_of_missionaries = 5 }
		set_variable = { which = AvailableMissionaries value = 5 }
	}
	else_if = {
		limit = { num_of_missionaries = 4 }
		set_variable = { which = AvailableMissionaries value = 4 }
	}
	else_if = {
		limit = { num_of_missionaries = 3 }
		set_variable = { which = AvailableMissionaries value = 3 }
	}
	else_if = {
		limit = { num_of_missionaries = 2 }
		set_variable = { which = AvailableMissionaries value = 2 }
	}
	else_if = {
		limit = { num_of_missionaries = 1 }
		set_variable = { which = AvailableMissionaries value = 1 }
	}
	else = {
		set_variable = { which = AvailableMissionaries value = 0 }
	}
	every_owned_province = {
		limit = { OR = { has_missionary = yes has_province_modifier = minority_missionary } }
		owner = { change_variable = { which = AvailableMissionaries value = -1 } }
	}
}

SendingAvailableMissionaries = {
	if = {
		limit = { UnassignedMissionaries = yes }
		if = {
			limit = {
				any_owned_province = {
					NOT = { religion = owner }
					check_variable = { which = MN_S0MStrength value = 0.3 }
					OR = {
						has_province_flag = convert_50
						has_province_flag = convert_90
					}
					has_missionary = no
				}
			}
			random_owned_province = {
				limit = {
					NOT = { religion = owner }
					check_variable = { which = MN_S0MStrength value = 0.3 }
					OR = {
						has_province_flag = convert_50
						has_province_flag = convert_90
					}
					has_missionary = no
				}
				send_missionary = yes
			}
		}
		else_if = {
			limit = {
				any_owned_province = {
					religion = owner
					has_province_flag = convert_90
					OR = {
						check_variable = { which = MN_S1MStrength value = 0.3 }
						check_variable = { which = MN_S2MStrength value = 0.3 }
						check_variable = { which = MN_S3MStrength value = 0.3 }
						check_variable = { which = MN_S4MStrength value = 0.3 }
						check_variable = { which = MN_S5MStrength value = 0.3 }
					}
					NOT = { check_variable = { which = MN_S0TotP value = 90 } }
					NOT = { has_province_modifier = minority_missionary }
				}
			}
			random_owned_province = {
				limit = {
					religion = owner
					has_province_flag = convert_90
					OR = {
						check_variable = { which = MN_S1MStrength value = 0.3 }
						check_variable = { which = MN_S2MStrength value = 0.3 }
						check_variable = { which = MN_S3MStrength value = 0.3 }
						check_variable = { which = MN_S4MStrength value = 0.3 }
						check_variable = { which = MN_S5MStrength value = 0.3 }
					}
					NOT = { check_variable = { which = MN_S0TotP value = 90 } }
					NOT = { has_province_modifier = minority_missionary }
				}
				add_province_modifier = { name = minority_missionary duration = -1 hidden = yes }
			}
		}
	}
}
