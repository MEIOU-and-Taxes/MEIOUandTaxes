AI_Army_Strength_CompareAlly = {
   set_variable = { which = Diplo_Tmp0 value = 0 }
    every_ally = {
		limit = {
			isValidCountry = yes
		}
        set_variable = { which = Diplo_Tmp0 which = Diplo_Mil_Strength }
        PREV = { change_variable = { which = Diplo_Tmp0 which = PREV } }
    }
    every_coalition_member = {
		limit = {
			isValidCountry = yes
		}
        set_variable = { which = Diplo_Tmp0 which = Diplo_Mil_Strength }
        PREV = { change_variable = { which = Diplo_Tmp0 which = PREV } }
    }
    set_variable = { which = Diplo_Mil_StrengthAlly which = Diplo_Tmp0 }
}

AI_Army_Strength_CompareEnemyHelper = {
	if = {
		limit = {
			check_variable = { which = Diplo_Mil_Strength which = PREV }
		}
		set_variable = { which = Diplo_Tmp0 which = Diplo_Mil_Strength }
		PREV = { set_variable = { which = Diplo_Tmp0 which = PREV } }
		set_variable = { which = Diplo_Tmp1 which = Diplo_Mil_StrengthAlly }
		PREV = { set_variable = { which = Diplo_Tmp1 which = PREV } }
	}
	set_variable = { which = Diplo_Tmp4 which = Diplo_Mil_Strength }
	change_variable = { which = Diplo_Tmp4 which = Diplo_Mil_StrengthAlly }
	if = {
		limit = {
			check_variable = { which = Diplo_Tmp4 which = PREV }
		}
		set_variable = { which = Diplo_Tmp2 which = Diplo_Mil_Strength }
		PREV = { set_variable = { which = Diplo_Tmp2 which = PREV } }
		set_variable = { which = Diplo_Tmp3 which = Diplo_Mil_StrengthAlly }
		PREV = { set_variable = { which = Diplo_Tmp3 which = PREV } }
	}
}
AI_Army_Strength_CompareEnemy = {
	set_variable = { which = Diplo_Tmp0 value = 0 }
	set_variable = { which = Diplo_Tmp1 value = 0 }
	set_variable = { which = Diplo_Tmp2 value = 0 }
	set_variable = { which = Diplo_Tmp3 value = 0 }
	set_variable = { which = Diplo_Tmp4 which = Diplo_Mil_Strength }
	change_variable = { which = Diplo_Tmp4 which = Diplo_Mil_StrengthAlly }
    every_neighbor_country = {
		limit = {
			isValidCountry = yes
		}
		AI_Army_Strength_CompareEnemyHelper = yes
    }
    every_rival_country = {
		limit = {
			isValidCountry = yes
		}
		AI_Army_Strength_CompareEnemyHelper = yes
    }
    every_coalition_member = {
		limit = {
			isValidCountry = yes
		}
		AI_Army_Strength_CompareEnemyHelper = yes
    }
    set_variable = { which = Tax_MPDesired which = Diplo_Tmp0 }
}
