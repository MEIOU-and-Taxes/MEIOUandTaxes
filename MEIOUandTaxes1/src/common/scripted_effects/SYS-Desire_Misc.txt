Desire_GetEquilFactor = {
    set_variable = { which = $return$ which = Desire_$type$ }
    multiply_variable = { which = $return$ which = Desire_$type$ }
    divide_variable = { which = $return$ value = -100 }
}

Desire_GetPowFactor = {
    if = {
        limit = {
            check_variable = { which = Prov_$type$Pow value = 1 }
        }
        set_variable = { which = Tmp_0 which = Prov_$type$Pow }
        multiply_variable = { which = Tmp_0 value = -1 }
        set_variable = { which = Tmp_1 value = -1 }

        if = {
            limit = {
                check_variable = { which = Prov_NOPow value = 1 }
            }
            change_variable = { which = Tmp_0 which = Prov_NOPow }
            change_variable = { which = Tmp_1 value = 1 }
        }
        if = {
            limit = {
                check_variable = { which = Prov_BGPow value = 1 }
            }
            change_variable = { which = Tmp_0 which = Prov_BGPow }
            change_variable = { which = Tmp_1 value = 1 }
        }
        if = {
            limit = {
                check_variable = { which = Prov_CLPow value = 1 }
            }
            change_variable = { which = Tmp_0 which = Prov_CLPow }
            change_variable = { which = Tmp_1 value = 1 }
        }

        if = {
            limit = {
                check_variable = { which = Tmp_1 value = 1 }
            }
            divide_variable = { which = Tmp_0 which = Tmp_1 }
        }
        else = {
            set_variable = { which = Tmp_0 value = 0 }
        }

        set_variable = { which = $return$ which = Prov_$type$Pow }
        subtract_variable = { which = $return$ which = Tmp_0 }
    }
    else = {
        set_variable = { which = $return$ value = -100 }
    }
}

}
