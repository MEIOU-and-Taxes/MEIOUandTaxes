Prp_Init = {
	set_variable = { which = Tmp_0 value = 0 }
	set_variable = { which = Tmp_1 value = 0 }
	set_variable = { which = SF_$type$ value = 0 }
	set_variable = { which = NM_$type$ value = 0 }
	set_variable = { which = RE_$type$ value = 0 }
	set_variable = { which = NO_$type$ value = 0 }
	set_variable = { which = BG_$type$ value = 0 }
	set_variable = { which = CL_$type$ value = 0 }

	[[Rural]
		if = {
			limit = {
				always = $Rural$
			}
			if = {
				limit = {
					check_variable = { which = SF_Total value = 0.001 }
				}
				set_variable = { which = SF_$type$ which = SF_PrpResources }
				change_variable = { which = SF_$type$ value = 0.1 }
				change_variable = { which = Tmp_0 which = SF_$type$ }
			}
			if = {
				limit = {
					check_variable = { which = NM_Total value = 0.001 }
				}
				set_variable = { which = NM_$type$ which = NM_Ratio }
				multiply_variable = { which = NM_$type$ value = 0.01 }
				change_variable = { which = Tmp_0 which = NM_$type$ }
			}
			
			if = {
				limit = {
					check_variable = { which = NO_Total value = 0.001 }
				}
				set_variable = { which = NO_$type$ which = NO_PrpResources }
				change_variable = { which = Tmp_0 which = NO_$type$ }
			}
			
			if = {
				limit = {
					check_variable = { which = BG_Total value = 0.001 }
				}
				set_variable = { which = BG_$type$ which = BG_Ratio }
				multiply_variable = { which = BG_$type$ value = 0.1 }
				multiply_variable = { which = BG_$type$ which = BG_PrpResources }
				change_variable = { which = Tmp_0 which = BG_$type$ }
			}
			
			if = {
				limit = {
					check_variable = { which = CL_Total value = 0.001 }
				}
				set_variable = { which = CL_$type$ which = CL_PrpResources }
				change_variable = { which = Tmp_0 which = CL_$type$ }
			}
			
			if = {
				limit = {
					isValidProv = yes
				}
				set_variable = { which = BU_$type$ which = BU_PrpResources }
				change_variable = { which = Tmp_0 which = BU_$type$ }
			}
		}
	]
	[[Urban]
		if = {
			limit = {
				always = $Urban$
			}
			set_variable = { which = RE_$type$ which = RE_PrpBuilding }
			change_variable = { which = RE_$type$ value = 0.1 }
			change_variable = { which = Tmp_0 which = RE_$type$ }
			
			if = {
				limit = {
					check_variable = { which = NO_Total value = 0.001 }
				}
				set_variable = { which = NO_$type$ which = NO_PrpBuilding }
				change_variable = { which = Tmp_0 which = NO_$type$ }
			}
			
			if = {
				limit = {
					check_variable = { which = BG_Total value = 0.001 }
				}
				set_variable = { which = BG_$type$ which = BG_PrpBuilding }
				change_variable = { which = Tmp_0 which = BG_$type$ }
			}
			
			if = {
				limit = {
					check_variable = { which = CL_Total value = 0.001 }
				}
				set_variable = { which = CL_$type$ which = CL_PrpBuilding }
				change_variable = { which = Tmp_0 which = CL_$type$ }
			}
			
			if = {
				limit = {
					isValidProv = yes
				}
				set_variable = { which = BU_$type$ which = BU_PrpBuilding }
				change_variable = { which = Tmp_0 which = BU_$type$ }
			}
		}
	]
	[[Trade]
		if = {
			limit = {
				always = $Trade$
			}
			#set_variable = { which = NO_$type$ which = NO_PrpBuilding }
			#change_variable = { which = Tmp_0 which = NO_$type$ }
			
			set_variable = { which = BG_$type$ which = BG_PrpBuilding }
			change_variable = { which = Tmp_0 which = BG_$type$ }
		}
	]
	[[Knowledge]
		if = {
			limit = {
				always = $Knowledge$
			}
			if = {
				limit = {
					OR = {
						has_province_flag = small_university
						has_province_flag = medium_university
						has_province_flag = large_university
					}
				}
				set_variable = { which = RE_$type$ value = 1 }
				change_variable = { which = Tmp_0 which = RE_$type$ }
			}
		}
	]
	
	[[pm]
		if = {
			limit = {
				has_province_flag = $pm$
			}
			change_variable = { which = BU_$type$ which = SF_$type$ }
			change_variable = { which = BU_$type$ which = NM_$type$ }
			change_variable = { which = BU_$type$ which = RE_$type$ }
			change_variable = { which = BU_$type$ which = NO_$type$ }
			change_variable = { which = BU_$type$ which = CL_$type$ }
			change_variable = { which = BU_$type$ which = BG_$type$ }

			set_variable = { which = SF_$type$ value = 0 }
			set_variable = { which = NM_$type$ value = 0 }
			set_variable = { which = RE_$type$ value = 0 }
			set_variable = { which = NO_$type$ value = 0 }
			set_variable = { which = CL_$type$ value = 0 }
			set_variable = { which = BG_$type$ value = 0 }			
		}
	]
	
	if = {
		limit = {
			check_variable = { which = SF_$type$ value = 0.001 }
		}
		divide_variable = { which = SF_$type$ which = Tmp_0 }
		multiply_variable = { which = SF_$type$ which = Building_$type$ }
		
		set_variable = { which = Building_$type$ShareSF which = SF_$type$ }
		multiply_variable = { which = Building_$type$ShareSF value = 100 }
	}
	if = {
		limit = {
			check_variable = { which = NM_$type$ value = 0.001 }
		}
		divide_variable = { which = NM_$type$ which = Tmp_0 }
		multiply_variable = { which = NM_$type$ which = Building_$type$ }
		
		set_variable = { which = Building_$type$ShareNM which = NM_$type$ }
		multiply_variable = { which = Building_$type$ShareNM value = 100 }
	}
	if = {
		limit = {
			check_variable = { which = RE_$type$ value = 0.001 }
		}
		divide_variable = { which = RE_$type$ which = Tmp_0 }
		multiply_variable = { which = RE_$type$ which = Building_$type$ }
		
		set_variable = { which = Building_$type$ShareRE which = RE_$type$ }
		multiply_variable = { which = Building_$type$ShareRE value = 100 }
	}
	if = {
		limit = {
			check_variable = { which = NO_$type$ value = 0.001 }
		}
		divide_variable = { which = NO_$type$ which = Tmp_0 }
		multiply_variable = { which = NO_$type$ which = Building_$type$ }
		
		set_variable = { which = Building_$type$ShareNO which = NO_$type$ }
		multiply_variable = { which = Building_$type$ShareNO value = 100 }
	}
	if = {
		limit = {
			check_variable = { which = BG_$type$ value = 0.001 }
		}
		divide_variable = { which = BG_$type$ which = Tmp_0 }
		multiply_variable = { which = BG_$type$ which = Building_$type$ }
		
		set_variable = { which = Building_$type$ShareBG which = BG_$type$ }
		multiply_variable = { which = Building_$type$ShareBG value = 100 }
	}
	if = {
		limit = {
			check_variable = { which = CL_$type$ value = 0.001 }
		}
		divide_variable = { which = CL_$type$ which = Tmp_0 }
		multiply_variable = { which = CL_$type$ which = Building_$type$ }
		
		set_variable = { which = Building_$type$ShareCL which = CL_$type$ }
		multiply_variable = { which = Building_$type$ShareCL value = 100 }
	}
	if = {
		limit = {
			check_variable = { which = BU_$type$ value = 0.001 }
		}
		divide_variable = { which = BU_$type$ which = Tmp_0 }
		multiply_variable = { which = BU_$type$ which = Building_$type$ }
		
		set_variable = { which = Building_$type$ShareBU which = BU_$type$ }
		multiply_variable = { which = Building_$type$ShareBU value = 100 }
	}
}

SetStateMaintOnProvChange = {
    if = {
        limit = {
            NOT = { is_core = owner }
            OR = {
                check_variable = { which = BU_PrpResources value = 2 }
                check_variable = { which = BU_PrpBuilding value = 2 }
            }                
        }
        set_variable = { which = Tmp_0 which = Prov_BUPow }
        multiply_variable = { which = Tmp_0 value = 1.15 }
        owner = {
            set_variable = { which = Tmp_0 which = PREV }
            divide_variable = { which = Tmp_0 which = Prov_BUPow }
            if = {
                limit = {
                    check_variable = { which = Tmp_0 value = 1 }
                }
                PREV = { remove_province_modifier = no_maint }
                if = {
                    limit = {
                        check_variable = { which = Tmp_0 value = 2 }
                    }
                    if = {
                        limit = {
                            check_variable = { which = Tmp_0 value = 2.5 }
                        }
                        PREV = {
                            add_province_modifier = {
                                name = no_maint
                                duration = 3650
                            }
                        }
                    }
                    else = {
                        PREV = {
                            add_province_modifier = {
                                name = no_maint
                                duration = 2555
                            }
                        }
                    }
                }
                else = {
                    PREV = {
                        add_province_modifier = {
                            name = no_maint
                            duration = 1825
                        }
                    }
                }                   
            }
        }
    }
}
# This part calculates the State Properties in total and a total percentage of everything the country has.
# It also calculates the State Property Capacity

# This happens at the end of Census 017
# This calculates on province level, which is then summed up to a country variable in SYS-Nat
StatePropertyTotalCalc = {
	set_variable = { which = PropertyTotal value = 0 }
	#SumUp all properties from all provinces
	change_variable = { which = PropertyTotal which = Building_Total }
	
	### New Calc
	### Each BU Property can be modified per province, how much impact on state property it has in total
	set_variable = { which = StatePropertyAgriculture which = BU_Agriculture }
	multiply_variable = { which = StatePropertyAgriculture which = Modi_StatePropertyAgriculture }
	set_variable = { which = StatePropertyForestry which = BU_Forestry }
	multiply_variable = { which = StatePropertyForestry which = Modi_StatePropertyForestry }
	set_variable = { which = StatePropertyExtraction which = BU_Extraction }
	multiply_variable = { which = StatePropertyExtraction which = Modi_StatePropertyExtraction }
	set_variable = { which = StatePropertyFishery which = BU_Fishery }
	multiply_variable = { which = StatePropertyFishery which = Modi_StatePropertyFishery }
	set_variable = { which = StatePropertyIndustrial which = BU_Industrial }
	multiply_variable = { which = StatePropertyIndustrial which = Modi_StatePropertyIndustrial }
	set_variable = { which = StatePropertyCommercial which = BU_Commercial }
	multiply_variable = { which = StatePropertyCommercial which = Modi_StatePropertyCommercial }
	set_variable = { which = StatePropertyAcademic which = BU_Academic }
	multiply_variable = { which = StatePropertyAcademic which = Modi_StatePropertyAcademic }
	
	### Adding the upper ones together to a total value
	set_variable = { which = StatePropertyTotalModified value = 0 }
	change_variable = { which = StatePropertyTotalModified which = StatePropertyAgriculture }
	change_variable = { which = StatePropertyTotalModified which = StatePropertyForestry }
	change_variable = { which = StatePropertyTotalModified which = StatePropertyExtraction }
	change_variable = { which = StatePropertyTotalModified which = StatePropertyFishery }
	change_variable = { which = StatePropertyTotalModified which = StatePropertyIndustrial }
	change_variable = { which = StatePropertyTotalModified which = StatePropertyCommercial }
	change_variable = { which = StatePropertyTotalModified which = StatePropertyAcademic }
	
	#SumUp all unmodified BU properties
	set_variable = { which = StatePropertyTotal value = 0 }
	change_variable = { which = StatePropertyTotal which = BU_Resources }
	change_variable = { which = StatePropertyTotal which = BU_Building }
}
#Goes through all important calculations for State Property
StatePropertyEffect = {
	StatePropertyTotalPercCalc = yes
	StatePropertyCapacityCalc = yes
	StatePropertyCapacityPerc = yes
	if = { 
		limit = {
			NOT = { has_global_flag = POP_Sim }
			check_variable = { which = StatePropertyTotalModified which = StatePropertyCapacity }
			# If over State Property Cap Clean up Event
		}
		country_event = {
			id = SYS_StateProperty.001
			days = 3
		}
	}
}

## Calculates the theoretical softcap of StateProperty
## Many things can influence the maximum capacity
StatePropertyCapacityCalc = {
	set_variable = { which = StatePropertyCapacity value = 0 }
		
	change_variable = { which = StatePropertyCapacity which = Modi_StatePropertyFlat }
	multiply_variable = { which = StatePropertyCapacity which = Modi_StatePropertyPerc }

		limit = {
			has_global_flag = POP_Sim
		}

	}
}

## Calculates Percentage of state owned property of total
StatePropertyTotalPercCalc = {
	set_variable = { which = StatePropertyPercTotal value = 0 }
	
	if = {
		limit = {
			check_variable = { which = StatePropertyTotal value = 0.001 }
		}
		set_variable = { which = StatePropertyPercTotal which = StatePropertyTotal }
		divide_variable = { which = StatePropertyPercTotal which = PropertyTotal }
		multiply_variable = { which = StatePropertyPercTotal value = 100 }
	}
	else = {
		set_variable = { which = StatePropertyPercTotal value = 0 }
	}
}	

##Calculates in % how much of your Capacity you use
StatePropertyCapacityPerc = {
	set_variable = { which = StatePropertyPercCap value = 0 }

	if = {
		limit = {
			check_variable = { which = StatePropertyTotalModified value = 0.001 }
			check_variable = { which = StatePropertyCapacity value = 0.001 }
		}
		change_variable = { which = StatePropertyPercCap which = StatePropertyTotalModified }
		divide_variable = { which = StatePropertyPercCap which = StatePropertyCapacity }
		multiply_variable = { which = StatePropertyPercCap value = 100 }
	}
	else = {
		set_variable = { which = StatePropertyPercCap value = 0 }
	}
	if = {
		limit = {
			check_variable = { which = StatePropertyPercCap value = 110 }
		}
		set_country_flag = StatePropertyBurden
		set_variable = { which = BuyOutTarget which = StatePropertyTotalModified }
		subtract_variable = { which = BuyOutTarget which = StatePropertyCapacity } # How much are we over cap
	}
	else_if = {
		limit = { 
			has_country_flag = StatePropertyBurden
		}
		clr_country_flag = StatePropertyBurden
	}
}
StatePropertyAnnexation = {
	if = {
		limit = {
			check_variable = { which = BU_$type$ value = 0.001 }
		}
		### Liquidation of asset, fictional cash, we assume its from the disappearing state treasury
		set_variable = { which = Tmp_1 which = BU_$type$ }
		multiply_variable = { which = Tmp_1 value = 2.5 }
		## Transfer properties to commoners or Elite
		
		change_variable = { which = $broker$_$type$ which = BU_$type$ }
		set_variable = { which = BU_$type$ value = 0 }
		change_variable = { which = $broker$_Prp$type$ which = BU_Prp$type$ }
		set_variable = { which = BU_Prp$type$ value = 0 }
		change_variable = { which = Building_$type$Share$broker$ which = Building_$type$ShareBU }
		set_variable = { which = Building_$type$ShareBU value = 0 }
		
		owner = {
			set_variable = { which = Tmp_1 which = PREV }
			set_var_from_variable = { var = ducat_gain variable = Tmp_1 }
			old_scaled_ducat_gained_country = yes
			set_variable = { which = Tmp_1 value = 0 }
		}
		set_variable = { which = Tmp_1 value = 0 }
	}
}
