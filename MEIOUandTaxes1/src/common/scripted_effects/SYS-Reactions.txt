Rights_SeverityLandInt = {
	if = {
		limit = {
			is_key_equal = { lhs = LandInt_$Class$ value = 0 }
		}
		Rights_SeverityBU$Type$ = yes
	}
	else_if = {
		limit = {
			is_key_equal = { lhs = LandInt_$Class$ value = 1 }
		}
		Rights_SeveritySF$Type$ = yes
	}
	else_if = {
		limit = {
			is_key_equal = { lhs = LandInt_$Class$ value = 2 }
		}
		Rights_SeverityRE$Type$ = yes
	}
	else_if = {
		limit = {
			is_key_equal = { lhs = LandInt_$Class$ value = 3 }
		}
		Rights_SeverityNO$Type$ = yes
	}
	else_if = {
		limit = {
			is_key_equal = { lhs = LandInt_$Class$ value = 4 }
		}
		Rights_SeverityBG$Type$ = yes
	}
	else_if = {
		limit = {
			is_key_equal = { lhs = LandInt_$Class$ value = 5 }
		}
		Rights_SeverityCL$Type$ = yes
	}
}
