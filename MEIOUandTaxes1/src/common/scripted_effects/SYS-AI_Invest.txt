Infra_AISpendCheck_Agriculture = {
    if = {
        limit = {
            always = yes
			[[cap]
                OR = {
                    NOT = { 
                       check_variable = { which = Building_$type$ which = $cap$ } 
                    }
                   
                    NOT = { 
                       check_variable = { which = BU_Prp$type$ value = 30 } 
                    }
                }
            ]
        }
        Infra_DoSpendAI_Agriculture = {  type = $type$ return = $return$ }
    }
}

Infra_DoSpendAI_Agriculture = {
	set_variable = { which = Tmp_0 which = Building_$type$Up }
	set_variable = { which = Tmp_7 which = Building_$type$Down }
	
	subtract_variable = { which = Tmp_0 which = Tmp_7 } # how much the industry wants to expand vs shrink
    if = {
        limit = { #if wants to expand & it's profitable
            check_variable = { which = Building_$type$Gross value = 0.001 }
            check_variable = { which = Tmp_0 value = 0.01 }
        }
        set_variable = { which = $return$ which = Tmp_0 } # Local Ai Willingness to Invest
        
        multiply_variable = { which = $return$ which = pv_ce } # CE
        multiply_variable = { which = $return$ value = 0.1 }
       
        if = {
            limit = {
                is_variable_equal = { which = Building_$type$ which = Building_AgricultureMax }
            }
            multiply_variable = { which = $return$ value = 0.05 }
        }
        if = {
            limit = {
                is_capital = yes
            }
            multiply_variable = { which = $return$ value = 3.5 }
        }
        else_if = {
            limit = {
                any_province_in_state = {
                    is_capital = yes
                }
            }
            multiply_variable = { which = $return$ value = 3 }
        }
        else_if = {
            limit = {
                NOT = { is_in_capital_area = yes }
            }
            multiply_variable = { which = $return$ value = 0.5 }
        }
        
        set_variable = { which = Tax_Tmp0 which = Tax_CostPropAgricultureRate }
        multiply_variable = { which = Tax_Tmp0 value = 0.01 }
        change_variable = { which = Tax_Tmp0 value = 1 }
        multiply_variable = { which = $return$ which = Tax_Tmp0 }
        set_variable = { which = Tax_Tmp0 value = 0 }
        
        trigger_switch = {
            on_trigger = has_province_flag 
            AgricultureInvestment6 = { multiply_variable = { which = $return$ value = 0.006 } clr_province_flag = AgricultureInvestment6 }
            AgricultureInvestment8 = { multiply_variable = { which = $return$ value = 0.004 } clr_province_flag = AgricultureInvestment8 }
        }
        
        
        if = { # reduce score by size -> prevents financial collapses
            limit = {
                check_variable = { which = Building_$type$ value = 800 }
            }
            if = {
                limit = {
                    check_variable = { which = Building_$type$ value = 1200 }
                }
                if = {
                    limit = {
                        check_variable = { which = Building_$type$ value = 1600 }
                    }
                    multiply_variable = { which = $return$ value = 0.05 }
                }
                else = {
                    multiply_variable = { which = $return$ value = 0.15 }
                }
            }
            else = {
                multiply_variable = { which = $return$ value = 0.25 }
            }
        }
        else_if = {
            limit = {
                check_variable = { which = Building_$type$ value = 350 }
            }
            if = {
                limit = {
                    check_variable = { which = Building_$type$ value = 500 }
                }
                if = {
                    limit = {
                        check_variable = { which = Building_$type$ value = 650 }
                    }
                    multiply_variable = { which = $return$ value = 0.3 }
                }
                else = {
                    multiply_variable = { which = $return$ value = 0.4 }
                }
            }
            else = {
                multiply_variable = { which = $return$ value = 0.5 }
            }
        }
    }
    else = {
        set_variable = { which = $return$ value = 0 }
    }
}

        multiply_variable = { which = $return$ value = 0.1 }
        
        if = {
            limit = { 
                owner = { ai = yes }
            }
            multiply_variable = { which = $return$ value = 0.75 } #because of academic
        }
        
        if = {
            limit = {
                is_capital = yes
            }
            multiply_variable = { which = $return$ value = 2 }
        }
        else_if = {
            limit = {
                any_province_in_state = {
                    is_capital = yes
                }
            }
            multiply_variable = { which = $return$ value = 1.5 }
        }
        else_if = {
            limit = {
                NOT = { is_in_capital_area = yes }
            }
            multiply_variable = { which = $return$ value = 0.5 }
        }

        set_variable = { which = Tax_Tmp0 which = Tax_CostPropAcademicRate }
        multiply_variable = { which = Tax_Tmp0 value = 0.01 }
        change_variable = { which = Tax_Tmp0 value = 1 }
        multiply_variable = { which = $return$ which = Tax_Tmp0 }
        set_variable = { which = Tax_Tmp0 value = 0 }
        
        trigger_switch = {
            on_trigger = has_province_flag 
            AcademicInvestment6 = { multiply_variable = { which = $return$ value = 0.006 } clr_province_flag = AcademicInvestment6 }
            AcademicInvestment8 = { multiply_variable = { which = $return$ value = 0.004 } clr_province_flag = AcademicInvestment8 }
        }
        
        if = { # reduce score by size -> prevents financial collapses
            limit = {
                check_variable = { which = Building_$type$ value = 800 }
            }
            if = {
                limit = {
                    check_variable = { which = Building_$type$ value = 1200 }
                }
                if = {
                    limit = {
                        check_variable = { which = Building_$type$ value = 1600 }
                    }
                    multiply_variable = { which = $return$ value = 0.05 }
                }
                else = {
                    multiply_variable = { which = $return$ value = 0.15 }
                }
            }
            else = {
                multiply_variable = { which = $return$ value = 0.25 }
            }
        }
        else_if = {
            limit = {
                check_variable = { which = Building_$type$ value = 350 }
            }
            if = {
                limit = {
                    check_variable = { which = Building_$type$ value = 500 }
                }
                if = {
                    limit = {
                        check_variable = { which = Building_$type$ value = 650 }
                    }
                    multiply_variable = { which = $return$ value = 0.3 }
                }
                else = {
                    multiply_variable = { which = $return$ value = 0.4 }
                }
            }
            else = {
                multiply_variable = { which = $return$ value = 0.5 }
            }
        }
    }
    else = {
        set_variable = { which = $return$ value = 0 }
    }
}
Do_InvestAI = {
    if = { 
        limit = {
            check_variable = { which = $return$ value = 0.001 }
        }
           
        set_variable = { which = Tmp_9 which = PREV } # set owner's residual budget
        set_variable = { which = Tmp_8 which = Tmp_9 } 
        
            # use the proportionate residual amount corresponding to the original score
		multiply_variable = { which = $score$ value = 0.01 }
        multiply_variable = { which = Tmp_8 which = $score$ } # Tmp_8 is how much of the residual budget the property will try to spend
        
            # substract from owner's residual budget
        subtract_variable = { which = Tmp_9 which = Tmp_8 }
        
            # re set owner's residual budget
        PREV = {
            set_variable = { which = Tmp_9 which = PREV }
        }
            # add residual budget to the actual investment budget
        change_variable =  { which = $return$ which = Tmp_8 }
        
        
        set_variable = { which = Public_Tmp11 which = Building_$type$ }
        multiply_variable = { which = Public_Tmp11 value = 0.25 } # maximum possible investment depending on building size
        if = {
            limit = { # if investment is higher than the cap 
                check_variable = { which = $return$ which = Public_Tmp11 }
            }
                # save investment target
            set_variable = { which = Tmp_8 which = $return$ }
            
                # set real investment to maximum limit
            set_variable = { which = $return$ which = Public_Tmp11 }
            
                # re calculate residual budget for this case
            subtract_variable = { which = Tmp_8 which = $return$ }
            
                # add it to total budget
            change_variable = { which = Tmp_9 which = Tmp_8 }
            
                # set it to owner
            PREV = {
                set_variable = { which = Tmp_9 which = PREV }
            }
        }
         if = { # if investment is higher than 75
            limit = {
				NOT = { is_year = 1450 }
                check_variable = { which = $return$ value = 75 }
            }
                # re calculate residual budget
            set_variable = { which = Tmp_0 which = $return$ }
            subtract_variable = { which = Tmp_0 value = 75 }
            
                # add it to new residual
            change_variable = { which = Tmp_9 which = Tmp_0 }
            
                # set it to owner
            PREV = {
                set_variable = { which = Tmp_9 which = PREV }
            }
            
            set_variable = { which = $return$ value = 75 }
        }
		else_if = { # if investment is higher than 100
            limit = {
				NOT = { is_year = 1550 }
                check_variable = { which = $return$ value = 100 }
            }
                # re calculate residual budget
            set_variable = { which = Tmp_0 which = $return$ }
            subtract_variable = { which = Tmp_0 value = 100 }
            
                # add it to new residual
            change_variable = { which = Tmp_9 which = Tmp_0 }
            
                # set it to owner
            PREV = {
                set_variable = { which = Tmp_9 which = PREV }
            }
            
            set_variable = { which = $return$ value = 100 }
        }
		else_if = { # if investment is higher than 150
            limit = {
				NOT = { is_year = 1650 }
                check_variable = { which = $return$ value = 150 }
            }
                # re calculate residual budget
            set_variable = { which = Tmp_0 which = $return$ }
            subtract_variable = { which = Tmp_0 value = 150 }
            
                # add it to new residual
            change_variable = { which = Tmp_9 which = Tmp_0 }
            
                # set it to owner
            PREV = {
                set_variable = { which = Tmp_9 which = PREV }
            }
            
            set_variable = { which = $return$ value = 150 }
        }
		else_if = { # if investment is higher than 200
            limit = {
				check_variable = { which = $return$ value = 200 }
            }
                # re calculate residual budget
            set_variable = { which = Tmp_0 which = $return$ }
            subtract_variable = { which = Tmp_0 value = 200 }
            
                # add it to new residual
            change_variable = { which = Tmp_9 which = Tmp_0 }
            
                # set it to owner
            PREV = {
                set_variable = { which = Tmp_9 which = PREV }
            }
            
            set_variable = { which = $return$ value = 200 }
        }


        # Actually investing the money (for AI) starts here 
        # Set country scope variable with the full amount (this is deducted from country treasury)
        PREV = {
            set_variable = { which = $return$ which = PREV }
            change_variable = { which = Public_Tmp7 which = $return$ }
        }

        # Op hax for Midas Touched AIs, they literally make money out of nothing
        if = {
            limit = {
                owner = {
                    ruler_has_personality = midas_touched_personality
                }
            }
            multiply_variable = { which = $return$ value = 1.25 }
        }

        # Money goes into investment pool in Full
        change_variable = { which = Building_$type$Wealth which = $return$ } 
    }
}
