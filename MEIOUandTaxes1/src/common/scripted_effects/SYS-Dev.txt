Dev_ScaleDown = {
    if = {
        limit = {
			check_variable = { which = $return$ value = 1 }
		}
		set_variable = { which = Tmp_1 which = $return$ }
		divide_variable = { which = Tmp_1 value = 60 }
		if = {
			limit = {
				check_variable = { which = Tmp_1 value = 1 }
			}
			set_variable = { which = Tmp_1 value = 1 }
		}
		multiply_variable = { which = Tmp_1 which = Tmp_1 }
		set_variable = { which = Tmp_0 value = 1 }
		subtract_variable = { which = Tmp_0 which = Tmp_1 }
		multiply_variable = { which = Tmp_0 value = 0.5 }
		change_variable = { which = Tmp_0 value = 0.5 }
		multiply_variable = { which = $return$ which = Tmp_0 }
		
		set_variable = { which = Tmp_1 which = $return$ }
		divide_variable = { which = Tmp_1 value = 120 }
		if = {
			limit = {
				check_variable = { which = Tmp_1 value = 1 }
			}
			set_variable = { which = Tmp_1 value = 1 }
		}
		multiply_variable = { which = Tmp_1 which = Tmp_1 }
		set_variable = { which = Tmp_0 value = 1 }
		subtract_variable = { which = Tmp_0 which = Tmp_1 }
		multiply_variable = { which = Tmp_0 value = 0.5 }
		change_variable = { which = Tmp_0 value = 0.5 }
		multiply_variable = { which = $return$ which = Tmp_0 }
		
		set_variable = { which = Tmp_1 which = $return$ }
		divide_variable = { which = Tmp_1 value = 240 }
		if = {
			limit = {
				check_variable = { which = Tmp_1 value = 1 }
			}
			set_variable = { which = Tmp_1 value = 1 }
		}
		multiply_variable = { which = Tmp_1 which = Tmp_1 }
		set_variable = { which = Tmp_0 value = 1 }
		subtract_variable = { which = Tmp_0 which = Tmp_1 }
		multiply_variable = { which = Tmp_0 value = 0.5 }
		change_variable = { which = Tmp_0 value = 0.5 }
		multiply_variable = { which = $return$ which = Tmp_0 }
		
		set_variable = { which = Tmp_1 which = $return$ }
		divide_variable = { which = Tmp_1 value = 500 }
		if = {
			limit = {
				check_variable = { which = Tmp_1 value = 1 }
			}
			set_variable = { which = Tmp_1 value = 1 }
		}
		multiply_variable = { which = Tmp_1 which = Tmp_1 }
		set_variable = { which = Tmp_0 value = 1 }
		subtract_variable = { which = Tmp_0 which = Tmp_1 }
		multiply_variable = { which = Tmp_0 value = 0.5 }
		change_variable = { which = Tmp_0 value = 0.5 }
		multiply_variable = { which = $return$ which = Tmp_0 }
	}
}

Dev_DoUncolonised = {
    if = {
        limit = {
            NOT = {
                check_variable = { which = Dev_Rural value = 1 }
            }
        }
        set_variable = { which = Dev_Rural value = 1 }
    }
    if = {
        limit = {
            NOT = {
                check_variable = { which = Dev_Notables value = 0 }
            }
        }
        set_variable = { which = Dev_Notables value = 0 }
    }
    if = {
        limit = {
            NOT = {
                check_variable = { which = Dev_Urban value = 0 }
            }
        }
        set_variable = { which = Dev_Urban value = 0 }
    }

    set_variable = { which = Dev_Total which = Dev_Notables }
    change_variable = { which = Dev_Total which = Dev_Urban }
    change_variable = { which = Dev_Total which = Dev_Rural }

	Dev_Set = { inp = Dev_Notables type = base_tax }
	Dev_Set = { inp = Dev_Urban type = base_production }
	Dev_Set = { inp = Dev_Rural type = base_manpower }
}

Dev_DoMain = {
    Dev_GetNotables = { return = Public_Tmp1 }
    Dev_GetUrban = { return = Public_Tmp2 }
    Dev_GetRural = { return = Public_Tmp3 }

    if = {
        limit = { check_variable = { which = Public_Tmp1 value = 0.001 } }
        divide_variable = { which = Public_Tmp1 value = 14 }
    }
    if = {
        limit = { check_variable = { which = Public_Tmp2 value = 0.001 } }
        divide_variable = { which = Public_Tmp2 value = 14 }
    }
    if = {
        limit = { check_variable = { which = Public_Tmp3 value = 0.001 } }
        divide_variable = { which = Public_Tmp3 value = 14 }
    }
	
	Dev_ScaleDown = { return=Public_Tmp1 }
	Dev_ScaleDown = { return=Public_Tmp2 }
	Dev_ScaleDown = { return=Public_Tmp3 }
	
	#Merges elite into urban
    #change_variable = { which = Public_Tmp2 which = Public_Tmp1 }
    #set_variable = { which = Public_Tmp1 value = 0 }

    if = {
        limit = {
			always = $iterate$
		}
		multiply_variable = { which = Dev_Notables value = 9 }
		multiply_variable = { which = Dev_Urban value = 9 }
		multiply_variable = { which = Dev_Rural value = 9 }

		change_variable = { which = Dev_Notables which = Public_Tmp1 }
		change_variable = { which = Dev_Urban which = Public_Tmp2 }
		change_variable = { which = Dev_Rural which = Public_Tmp3 }

		divide_variable = { which = Dev_Notables value = 10 }
		divide_variable = { which = Dev_Urban value = 10 }
		divide_variable = { which = Dev_Rural value = 10 }
	}
	else = {
		set_variable = { which = Dev_Notables which = Public_Tmp1 }
		set_variable = { which = Dev_Urban which = Public_Tmp2 }
		set_variable = { which = Dev_Rural which = Public_Tmp3 }
	}

    change_variable = { which = Dev_Rural value = 0.1 }

    # Modify Dev to not have decimal digits
    divide_variable = { which = Dev_Rural value = 1000 }
    multiply_variable = { which = Dev_Rural value = 1000 }
    divide_variable = { which = Dev_Urban value = 1000 }
    multiply_variable = { which = Dev_Urban value = 1000 }
    divide_variable = { which = Dev_Notables value = 1000 }
    multiply_variable = { which = Dev_Notables value = 1000 }


    # Check if dev is <= 0 (<= 1 for rural) set to minimum of 0/1
    if = {
        limit = {
            NOT = {
                check_variable = { which = Dev_Rural value = 1 }
            }
        }
        set_variable = { which = Dev_Rural value = 1 }
    }
    if = {
        limit = {
            NOT = {
                check_variable = { which = Dev_Notables value = 0 }
            }
        }
        set_variable = { which = Dev_Notables value = 0 }
    }
    if = {
        limit = {
            NOT = {
                check_variable = { which = Dev_Urban value = 0 }
            }
        }
        set_variable = { which = Dev_Urban value = 0 }
    }

    # Sum up total development
    set_variable = { which = Dev_Total which = Dev_Notables }
    change_variable = { which = Dev_Total which = Dev_Urban }
    change_variable = { which = Dev_Total which = Dev_Rural }

	### disable for init
    if = {
        limit = {
			NOT = { has_global_flag = POP_Sim }
        }
		Dev_Set = { inp = Dev_Notables type = base_tax }
		Dev_Set = { inp = Dev_Urban type = base_production }
		Dev_Set = { inp = Dev_Rural type = base_manpower }
		DiploAnnex_Dev = yes
	}
    set_variable = { which = Public_Tmp1 value = 0 }
    set_variable = { which = Public_Tmp2 value = 0 }
    set_variable = { which = Public_Tmp3 value = 0 }
}

Dev_Set = {
    export_to_variable = { which = Tmp_0 value = $type$ }

    subtract_variable = { which = Tmp_0 which = $inp$ }
    multiply_variable = { which = Tmp_0 value = -1 }

    if = { limit = { check_variable = { which = Tmp_0 value = 64 } } subtract_variable = { which = Tmp_0 value = 64 } add_$type$ = 64 }
    if = { limit = { check_variable = { which = Tmp_0 value = 32 } } subtract_variable = { which = Tmp_0 value = 32 } add_$type$ = 32 }
    if = { limit = { check_variable = { which = Tmp_0 value = 16 } } subtract_variable = { which = Tmp_0 value = 16 } add_$type$ = 16 }
    if = { limit = { check_variable = { which = Tmp_0 value = 8 } } subtract_variable = { which = Tmp_0 value = 8 } add_$type$ = 8 }
    if = { limit = { check_variable = { which = Tmp_0 value = 4 } } subtract_variable = { which = Tmp_0 value = 4 } add_$type$ = 4 }
    if = { limit = { check_variable = { which = Tmp_0 value = 2 } } subtract_variable = { which = Tmp_0 value = 2 } add_$type$ = 2 }
    if = { limit = { check_variable = { which = Tmp_0 value = 1 } } subtract_variable = { which = Tmp_0 value = 1 } add_$type$ = 1 }

    if = {
        limit = {
            NOT = {
                check_variable = { which = Tmp_0 value = 0 }
            }
        }
        multiply_variable = { which = Tmp_0 value = -1 }

        if = { limit = { check_variable = { which = Tmp_0 value = 64 } } subtract_variable = { which = Tmp_0 value = 64 } add_$type$ = -64 }
        if = { limit = { check_variable = { which = Tmp_0 value = 32 } } subtract_variable = { which = Tmp_0 value = 32 } add_$type$ = -32 }
        if = { limit = { check_variable = { which = Tmp_0 value = 16 } } subtract_variable = { which = Tmp_0 value = 16 } add_$type$ = -16 }
        if = { limit = { check_variable = { which = Tmp_0 value = 8 } } subtract_variable = { which = Tmp_0 value = 8 } add_$type$ = -8 }
        if = { limit = { check_variable = { which = Tmp_0 value = 4 } } subtract_variable = { which = Tmp_0 value = 4 } add_$type$ = -4 }
        if = { limit = { check_variable = { which = Tmp_0 value = 2 } } subtract_variable = { which = Tmp_0 value = 2 } add_$type$ = -2 }
        if = { limit = { check_variable = { which = Tmp_0 value = 1 } } subtract_variable = { which = Tmp_0 value = 1 } add_$type$ = -1 }
    }

    set_variable = { which = Tmp_0 value = 0 }
}

Dev_GetIncomeRural = {
    if = {
        limit = {
            has_province_flag = Prod_S$slot$
			OR = {
				AND = {
					check_variable = { which = Prod_S$slot$ value = 1 } 
					NOT = { check_variable = { which = Prod_S$slot$ value = 24 } }
				}
				AND = {
					check_variable = { which = Prod_S$slot$ value = 38 } 
					NOT = { check_variable = { which = Prod_S$slot$ value = 39 } }
				}
				AND = {
					check_variable = { which = Prod_S$slot$ value = 41 } 
					NOT = { check_variable = { which = Prod_S$slot$ value = 43 } }
				}
			}

            check_variable = { which = Prod_S$slot$Income value = 0.001 }
        }
        change_variable = { which = $return$ which = Prod_S$slot$Income }
    }
}
