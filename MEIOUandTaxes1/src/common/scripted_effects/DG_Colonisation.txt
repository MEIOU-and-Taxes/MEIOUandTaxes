
#Adds some converts when colony starts
MN_AddOwnerColo = {
	#Conversion to the owner religion, exporting it as a variable
	export_to_variable = {
		 which = DG_Tmp1
		 value = religion
		 who = owner
	}
	#Adds the religion in case it isn't there already
	MN_AddReligion = { type = which religion = DG_Tmp1 }
	
	MN_CleaningUp = yes
	
	set_variable = { which = DG_Tmp2 value = 0.800 }
	set_variable = { which = DG_Tmp3 value = 0.150 }
	set_variable = { which = DG_Tmp4 value = 0.050 }
	
	MN_HeresyHelper = { to = DG_Tmp1 typeto = which from = MN_S0 typefrom = which }
	
	set_variable = { which = DG_Tmp1 value = 0 }
	set_variable = { which = DG_Tmp2 value = 0 }
	set_variable = { which = DG_Tmp3 value = 0 }
	set_variable = { which = DG_Tmp4 value = 0 }
	
	MN_GetSlotInit = yes
	#Converts the religion if necessary, and switches slots between former and new religion
	set_province_flag = DG_Update
}

#Makes sure owner has his religion set.
MN_ConvColoHelper = {
	if = {
		limit = {
			is_variable_equal = { which = MN_S$slot$ which = DG_Tmp1 }
		}

		#Changing numbers in population numbers
		change_variable = { which = MN_S$slot$RurN value = 0.038 }
		change_variable = { which = MN_S$slot$UrbN value = 0.000 }
		change_variable = { which = MN_S$slot$NotN value = 0.025 }
		
		#Re-calculating of the propotions of each religious groups
		set_variable = { which = MN_S$slot$Rural which = MN_S$slot$RurN }
		set_variable = { which = MN_S$slot$Urban which = MN_S$slot$UrbN }
		set_variable = { which = MN_S$slot$Notables which = MN_S$slot$NotN }
		if = {
			limit = { is_variable_equal = { which = Class_Rural value = 0 } }
			set_variable = { which = MN_S$slot$Rural value = 0 }
		}
		else = {
			divide_variable = { which = MN_S$slot$Rural which = Class_Rural }
		}
		if = {
			limit = { is_variable_equal = { which = Class_Urban value = 0 } }
			set_variable = { which = MN_S$slot$Urban value = 0 }
		}
		else = {
			divide_variable = { which = MN_S$slot$Urban which = Class_Urban }
		}
		if = {
			limit = { is_variable_equal = { which = Class_Notables value = 0 } }
			set_variable = { which = MN_S$slot$Notables value = 0 }
		}
		else = {
			divide_variable = { which = MN_S$slot$Notables which = Class_Notables }
		}

		set_variable = { which = MN_S$slot$RurP which = MN_S$slot$Rural }
		set_variable = { which = MN_S$slot$UrbP which = MN_S$slot$Urban }
		set_variable = { which = MN_S$slot$NotP which = MN_S$slot$Notables }
		multiply_variable = { which = MN_S$slot$RurP value = 100 }
		multiply_variable = { which = MN_S$slot$UrbP value = 100 }
		multiply_variable = { which = MN_S$slot$NotP value = 100 }
		
		#Calculating the total number for each religious groups
		set_variable = { which = MN_S$slot$TotN which = MN_S$slot$RurN }
		change_variable = { which = MN_S$slot$TotN which = MN_S$slot$UrbN }
		change_variable = { which = MN_S$slot$TotN which = MN_S$slot$NotN }
		
		if = {
			limit = {
				check_variable = { which = MN_S$slot$TotN value = 0.001 }
				check_variable = { which = Class_Total value = 0.001 }
			}
		
			set_variable = { which = MN_S$slot$TotP which = MN_S$slot$TotN }
			divide_variable = { which = MN_S$slot$TotP which = Class_Total }
			multiply_variable = { which = MN_S$slot$TotP value = 100 }
		}
		else = {
			set_variable = { which = MN_S$slot$TotP value = 0 }
		}
	}
}

MN_ConvColo = {
	#Conversion to the owner religion, exporting it as a variable
	export_to_variable = {
		 which = DG_Tmp1
		 value = religion
		 who = owner
	}
	MN_ConvColoHelper = { slot = 0 }
	MN_ConvColoHelper = { slot = 1 }
	MN_ConvColoHelper = { slot = 2 }
	MN_ConvColoHelper = { slot = 3 }
	MN_ConvColoHelper = { slot = 4 }
	MN_ConvColoHelper = { slot = 5 }
	set_variable = { which = DG_Tmp1 value = 0 }
	#Converts the religion if necessary, and switches slots between former and new religion
	set_province_flag = DG_Update
}
