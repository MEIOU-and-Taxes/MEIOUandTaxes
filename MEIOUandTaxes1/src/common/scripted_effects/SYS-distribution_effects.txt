POP_distribute_total = {
	# Custom POP distribution
	every_province = {
		limit = {
			isValidEmpty = yes
			check_key = { lhs = pop_level value = 0.5 }
			$limit_trigger$ = yes
		}
		if = {
			limit = {
				is_key_equal = { lhs = pop_level value = 0.5 }
			}
			set_key = { lhs = pop_share value = $lvl_1_points$ }
			multiply_key = { lhs = pop_share value = 0.75 }
		}
		else_if = {
			limit = {
				is_key_equal = { lhs = pop_level value = 1 }
			}
			set_key = { lhs = pop_share value = $lvl_1_points$ }
		}
		else_if = {
			limit = {
				is_key_equal = { lhs = pop_level value = 1.5 }
			}
			set_key = { lhs = pop_share value = $lvl_2_points$ }
			multiply_key = { lhs = pop_share value = 0.75 }
		}
		else_if = {
			limit = {
				is_key_equal = { lhs = pop_level value = 2 }
			}
			set_key = { lhs = pop_share value = $lvl_2_points$ }
		}
		else_if = {
			limit = {
				is_key_equal = { lhs = pop_level value = 2.5 }
			}
			set_key = { lhs = pop_share value = $lvl_3_points$ }
			multiply_key = { lhs = pop_share value = 0.75 }
		}
		else_if = {
			limit = {
				is_key_equal = { lhs = pop_level value = 3 }
			}
			set_key = { lhs = pop_share value = $lvl_3_points$ }
		}
		else_if = {
			limit = {
				is_key_equal = { lhs = pop_level value = 3.5 }
			}
			set_key = { lhs = pop_share value = $lvl_4_points$ }
			multiply_key = { lhs = pop_share value = 0.75 }
		}
		else_if = {
			limit = {
				is_key_equal = { lhs = pop_level value = 4 }
			}
			set_key = { lhs = pop_share value = $lvl_4_points$ }
		}
		else_if = {
			limit = {
				is_key_equal = { lhs = pop_level value = 4.5 }
			}
			set_key = { lhs = pop_share value = $lvl_5_points$ }
			multiply_key = { lhs = pop_share value = 0.75 }
		}
		else_if = {
			limit = {
				is_key_equal = { lhs = pop_level value = 5 }
			}
			set_key = { lhs = pop_share value = $lvl_5_points$ }
		}
		else_if = {
			limit = {
				is_key_equal = { lhs = pop_level value = 5.5 }
			}
			set_key = { lhs = pop_share value = $lvl_6_points$ }
			multiply_key = { lhs = pop_share value = 0.75 }
		}
		else_if = {
			limit = {
				is_key_equal = { lhs = pop_level value = 6 }
			}
			set_key = { lhs = pop_share value = $lvl_6_points$ }
		}
		else_if = {
			limit = {
				is_key_equal = { lhs = pop_level value = 6.5 }
			}
			set_key = { lhs = pop_share value = $lvl_7_points$ }
			multiply_key = { lhs = pop_share value = 0.75 }
		}
		else_if = {
			limit = {
				is_key_equal = { lhs = pop_level value = 7 }
			}
			set_key = { lhs = pop_share value = $lvl_7_points$ }
		}
		else_if = {
			limit = {
				is_key_equal = { lhs = pop_level value = 8 }
			}
			set_key = { lhs = pop_share value = $lvl_7_points$ }
			multiply_key = { lhs = pop_share value = 1.5 }
		}
		else_if = {
			limit = {
				is_key_equal = { lhs = pop_level value = 9 }
			}
			set_key = { lhs = pop_share value = $lvl_7_points$ }
			multiply_key = { lhs = pop_share value = 2 }
		}
		else_if = {
			limit = {
				is_key_equal = { lhs = pop_level value = 10 }
			}
			set_key = { lhs = pop_share value = $lvl_7_points$ }
			multiply_key = { lhs = pop_share value = 2.5 }
		}
		else_if = {
			limit = {
				is_key_equal = { lhs = pop_level value = 11 }
			}
			set_key = { lhs = pop_share value = $lvl_7_points$ }
			multiply_key = { lhs = pop_share value = 3 }
		}
		else_if = {
			limit = {
				is_key_equal = { lhs = pop_level value = 12 }
			}
			set_key = { lhs = pop_share value = $lvl_7_points$ }
			multiply_key = { lhs = pop_share value = 3.5 }
		}
		else_if = {
			limit = {
				is_key_equal = { lhs = pop_level value = 13 }
			}
			set_key = { lhs = pop_share value = $lvl_7_points$ }
			multiply_key = { lhs = pop_share value = 4 }
		}
		else_if = {
			limit = {
				is_key_equal = { lhs = pop_level value = 14 }
			}
			set_key = { lhs = pop_share value = $lvl_7_points$ }
			multiply_key = { lhs = pop_share value = 5 }
		}
		else_if = {
			limit = {
				is_key_equal = { lhs = pop_level value = 15 }
			}
			set_key = { lhs = pop_share value = $lvl_7_points$ }
			multiply_key = { lhs = pop_share value = 6 }
		}
		
		multiply_key = { lhs = pop_share which = Land_PSize }
		multiply_key = { lhs = pop_share value = 0.001 }
		PREV = { change_key = { lhs = pop_share which = PREV } }
	}
	set_key = { lhs = pop_share_total which = pop_share }
	set_key = { lhs = pop_share value = 0 }
	every_province = {
		limit = {
			isValidEmpty = yes
			check_key = { lhs = pop_level value = 0.5 }
			$limit_trigger$ = yes
		}
		set_key = { lhs = pop_share_total which = PREV }
		multiply_key = { lhs = pop_share value = 10 }
		divide_key = { lhs = pop_share which = pop_share_total } 
		
		multiply_key = { lhs = pop_share value = $pop_total$ }
		multiply_key = { lhs = pop_share value = 0.1 }
		set_key = { lhs = pop_share_total value = 0 }
		if = {
			limit = {
				NOT = { check_key = { lhs = pop_share value = 1.1 } }
			}
			change_key = { lhs = pop_share value = 1 }
		}
	}
	set_key = { lhs = pop_share_total value = 0 }
}
