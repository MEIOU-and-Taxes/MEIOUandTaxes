PublicHelper_LootWealth = {
    set_variable = { which = PublicHelper_Tmp1 $type1$ = $rate$ }
    multiply_variable = { which = PublicHelper_Tmp1 which = $targ$ }

    subtract_variable = { which = $targ$ which = PublicHelper_Tmp1 }

    if = {
        limit = {
            NOT = {
                check_variable = { which = $targ$ value = 0 }
            }
        }
        set_variable = { which = $targ$ value = 0 }
    }

    change_variable = { which = $return$ which = PublicHelper_Tmp1 }

    set_variable = { which = PublicHelper_Tmp1 value = 0 }

    set_province_flag = POP_ChangedWealth
}

PublicHelper_AddManpower = {
    if = {
        limit = {
            check_variable = {
                which = $inp$
                value = $val$
            }
        }
        subtract_variable = {
            which = $inp$
            value = $val$
        }

        add_manpower = $val$
    }
}
PublicHelper_SubtManpower = {
    if = {
        limit = {
            check_variable = {
                which = $inp$
                value = $val$
            }
        }
        subtract_variable = {
            which = $inp$
            value = $val$
        }

        add_manpower = -$val$
    }
}
