Tariff_SetRate = {
    save_event_target_as = Prov

    set_variable = { which = Tmp_0 value = 0 }
    set_variable = { which = Tmp_1 value = 0 }

    owner = {
        every_owned_province = {
            limit = {
                has_province_flag = TN_SectorPart
				
                check_variable = { which = TN_ProvSply51 value = 0.001 }

                same_trade_node_as = event_target:Prov
            }
            set_variable = { which = Tmp_2 which = TN_ProvSply51 }

            event_target:Prov = {
                set_variable = { which = Tmp_2 which = PREV }

                change_variable = { which = Tmp_0 which = Tmp_2 }
            }

            export_to_variable = { which = Tmp_2 value = local_autonomy }
            subtract_variable = { which = Tmp_2 value = 100 }
            divide_variable = { which = Tmp_2 value = -100 }
            multiply_variable = { which = Tmp_2 which = Prov_BUPow }
			multiply_variable = { which = Tmp_2 value = 0.01 }
    
            multiply_variable = { which = Tmp_2 which = TN_ProvSply51 }

            event_target:Prov = {
                set_variable = { which = Tmp_2 which = PREV }

                change_variable = { which = Tmp_1 which = Tmp_2 }
            }

            set_variable = { which = Tmp_2 value = 0 }
        }
    }

    if = {
        limit = {
            check_variable = { which = Tmp_0 value = 0.001 }
			NOT = { is_variable_equal = { which = Tmp_1 value = 0 } }
        }
        divide_variable = { which = Tmp_1 which = Tmp_0 }
    }

    # Edit for new produce
    Tariff_SetRateHelper = { prod = 1 }
    Tariff_SetRateHelper = { prod = 2 }
    Tariff_SetRateHelper = { prod = 4 }
    Tariff_SetRateHelper = { prod = 5 }
    Tariff_SetRateHelper = { prod = 6 }
    Tariff_SetRateHelper = { prod = 9 }
    Tariff_SetRateHelper = { prod = 10 }
    Tariff_SetRateHelper = { prod = 21 }
    Tariff_SetRateHelper = { prod = 22 }
    Tariff_SetRateHelper = { prod = 23 }
    Tariff_SetRateHelper = { prod = 24 }
    Tariff_SetRateHelper = { prod = 25 }
    Tariff_SetRateHelper = { prod = 41 }

    set_variable = { which = Tmp_0 value = 0 }
    set_variable = { which = Tmp_1 value = 0 }
    set_variable = { which = Tmp_2 value = 0 }
}

Tariff_SetRateHelper = {
    set_variable = { which = Tariff_Rate$prod$ which = owner }

    if = {
        limit = {
            NOT = {
                check_variable = { which = Tmp_1 which = Tariff_Rate$prod$ }
            }
        }
        set_variable = { which = Tmp_2 which = Tariff_Rate$prod$ }
        subtract_variable = { which = Tmp_2 which = Tmp_1 }
    }
    else = {
        set_variable = { which = Tmp_2 value = 0 }
    }

    if = {
        limit = {
            owner = {
                Privilege_HasBGSmuggling = yes
            }
        }
        change_variable = { which = Tmp_2 value = 0.1 }

        if = {
            limit = {
                check_variable = { which = Tmp_2 value = 1 }
            }
            set_variable = { which = Tmp_2 value = 1 }
        }
    }

    subtract_variable = { which = Tmp_2 value = 1 }
    multiply_variable = { which = Tmp_2 value = -1 }

    multiply_variable = { which = Tariff_Rate$prod$ which = Tmp_2 }

    # Tweak
    multiply_variable = { which = Tariff_Rate$prod$ value = 0.2 }
}
