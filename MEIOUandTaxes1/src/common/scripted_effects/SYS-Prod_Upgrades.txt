## File contain all relevant calculations and effects for upgrading industry slots

# Check all owned province, that have urban pop for its slots
# If we, in the future, can locate urban industries in a fixed set of slots, these checks would get simpler
Prod_FindProvincesForUpgrade = {
	every_owned_province = {
		limit = {
			has_province_flag = City_Present
		}
		Prod_FindProvincesForUpgrade_Helper = { slot = 0 }
		Prod_FindProvincesForUpgrade_Helper = { slot = 1 }
		Prod_FindProvincesForUpgrade_Helper = { slot = 2 }
		Prod_FindProvincesForUpgrade_Helper = { slot = 3 }
		Prod_FindProvincesForUpgrade_Helper = { slot = 4 }
		Prod_FindProvincesForUpgrade_Helper = { slot = 5 }
		Prod_FindProvincesForUpgrade_Helper = { slot = 6 }
		Prod_FindProvincesForUpgrade_Helper = { slot = 7 }
		Prod_FindProvincesForUpgrade_Helper = { slot = 8 }
		Prod_FindProvincesForUpgrade_Helper = { slot = 9 }
		Prod_FindProvincesForUpgrade_Helper = { slot = 10 }
		Prod_FindProvincesForUpgrade_Helper = { slot = 11 }
		Prod_FindProvincesForUpgrade_Helper = { slot = 12 }
		Prod_FindProvincesForUpgrade_Helper = { slot = 13 }
		Prod_FindProvincesForUpgrade_Helper = { slot = 14 }
		Prod_FindProvincesForUpgrade_Helper = { slot = 15 }
	}
}
# Check the Slot, if it has an urban industry or academic industry
# This has an edge-case for Cottage Industry as this could reach 150% skill, in particular if its in parallel with urban industries.
# We might wanna solve this cottage edge case in a different way in the future anyway. Like change the city foundation from an action to a semi-automatic conversion from Cottage Industries to urban industries?
Prod_FindProvincesForUpgrade_Helper = {
	if = {
		limit = {
			check_key = { lhs = Prod_S$slot$Soph value = 150 }
			OR = {
				has_province_flag = Prod_S$slot$DmndBIndustrial
				has_province_flag = Prod_S$slot$DmndBAcademic
			}
		}
		set_province_flag = Industry_CanUpgrade
	}
}

#Set flags for provinces with dead basic urban industries
Prod_FindDeadUrbanIndustries = {
	###Houseware Branches
	if = {
		limit = {
			NOT = {
				has_province_flag = Industry_28Present
				has_province_flag = Industry_33Present
				has_province_flag = Industry_36Present
				has_province_flag = Industry_24Present
			}
			Prod_Can_24 = yes
		}
		set_province_flag = ResurrectIndustry_24
	}
	###Textile Branches
	if = {
		limit = {
			NOT = {
				has_province_flag = Industry_34Present
				has_province_flag = Industry_39Present
				has_province_flag = Industry_25Present
			}
			Prod_Can_25 = yes
		}
		set_province_flag = ResurrectIndustry_25
	}
	###Metalworking Branches
	if = {
		limit = {
			NOT = {
				has_province_flag = Industry_35Present

			}
			Prod_Can_26 = yes
		}

		}
