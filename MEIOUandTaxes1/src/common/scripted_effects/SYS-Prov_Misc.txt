Prov_DoPowNO = {
	## Calc the resting point
	## Wealth, autonomy, rights-based power and current power are the variable factors (inertia)
	set_variable = { which = Prov_NOPowRest value = 0 }

	## Class share of wealth, 1:1
	# Essentially Class Wealth/Total Class Wealth * 100
	if = {
		limit = {
			check_variable = { which = NO_Wealth value = 1.001 }
			OR = {
				NOT = { is_variable_equal = { which = Tax_Tmp6 value = 0 } }
				NOT = { is_variable_equal = { which = Tax_Tmp7 value = 0 } }
			}
		}
		change_variable = { which = Prov_NOPowRest which = NO_Wealth }
		subtract_variable = { which = Prov_NOPowRest value = 1 }
		set_variable = { which = Tmp_1 which = Tax_Tmp6 } 
		change_variable = { which = Tmp_1 which = Tax_Tmp7 }
		divide_variable = { which = Prov_NOPowRest which = Tmp_1 }
		multiply_variable = { which = Prov_NOPowRest value = 100 } # 0-1 scale -> 0-100 scale
		multiply_variable = { which = Prov_NOPowRest which = Modi_NOPowerWealth }
		multiply_variable = { which = Prov_NOPowRest value = 0.60 } # balancing 
	}
	change_variable = { which = Prov_NOPowRest which = Modi_NOPowerRest }
	multiply_variable = { which = Prov_NOPowRest which = Modi_NOPower }
	
	## Local autonomy provides a boost to elite power
	set_variable = { which = Tmp_1 which = Tax_Tmp8 } # LA
	multiply_variable = { which = Tmp_1 value = 0.35 } # Balancing
	multiply_variable = { which = Tmp_1 which = Modi_PowerFromLANO }
	multiply_variable = { which = Tmp_1 which = Modi_PowerFromLAElites }
	change_variable = { which = Tmp_1 value = 1 }
	multiply_variable = { which = Prov_NOPowRest which = Tmp_1 }
	set_variable = { which = Tmp_1 value = 0 }
	
	change_variable = { which = Prov_NOPowRest which = Prov_NOPowEntrench }

	## National pre-calced class institutional power
	# Comes from rights and privileges, and pre-calced at the national level for performance reasons
	#owner = {
	#	set_variable = { which = Tmp_0 which = NO_PowCalc }
	#	PREV = { set_variable = { which = Tmp_0 which = PREV } }
	#	set_variable = { which = Tmp_0 value = 0 }
	#}
	#change_variable = { which = Prov_NOPowRest which = Tmp_0 }
	#set_variable = { which = Tmp_0 value = 0 }

	# Drop the last digit, 2 decimal places for display and removal of micro changes
	multiply_variable = { which = Prov_NOPowRest value = 0.1 }
	multiply_variable = { which = Prov_NOPowRest value = 10 }
	
	## Calc annual power change
	# Power has inertia, so add Pow * -1 to the monthly change. 100 power is -1.00 effect
	# Result is upward pressure and downward pressure leading to a resting point
	set_variable = { which = Prov_NOPowChange which = Prov_NOPow }
	multiply_variable = { which = Prov_NOPowChange value = -1 }
	change_variable = { which = Prov_NOPowChange which = Prov_NOPowRest }
	# Above is considered monthly change, so x12 the effect. Mimics autonomy
	#multiply_variable = { which = Prov_NOPowChange value = 12 } # For the year, 12 monthly changes
	multiply_variable = { which = Prov_NOPowChange value = 0.01 } # 1% speed of change per month, so 12% a year. Applied last to preserve precision
		
	# Rate of change
	multiply_variable = { which = Prov_NOPowChange value = 2 } # Raise this for faster change
	multiply_variable = { which = Prov_NOPowChange which = Modi_NOPowerChangeSpeed }
	
	change_variable = { which = Prov_NOPowChange which = Modi_NOPowerChange }

	# Clan power caps other elite power at 100 - TRPow
	# Change should not surpass it
	if = {
		limit = {
			check_variable = { which = Prov_TRPow value = 0.001 }
		}
		set_variable = { which = Tmp_0 which = Prov_NOPow }
		change_variable = { which = Tmp_0 which = Prov_NOPowChange }

		## If the new total will surpass the cap, then change is Cap - Pow (difference left for 1:1)
		if = {
			limit = {
				check_variable = { which = Tmp_0 which = Prov_TRPowCap }
			}
			set_variable = { which = Prov_NOPowChange which = Prov_TRPowCap }
			subtract_variable = { which = Prov_NOPowChange which = Prov_NOPow }
		}
	}

	### Apply the change
	change_variable = { which = Prov_NOPow which = Prov_NOPowChange }

	# Drop the last digit, 2 decimal places for display and removal of micro changes
	multiply_variable = { which = Prov_NOPow value = 0.1 }
	multiply_variable = { which = Prov_NOPow value = 10 }
	multiply_variable = { which = Prov_NOPowChange value = 0.1 }
	multiply_variable = { which = Prov_NOPowChange value = 10 }

	# Clamp range to 0-100
	if = {
		limit = {
			check_variable = { which = Prov_NOPow value = 100 }
		}
		set_variable = { which = Prov_NOPow value = 100 }
	}
	else_if = {
		limit = {
			NOT = { check_variable = { which = Prov_NOPow value = 0 } }
		}
		set_variable = { which = Prov_NOPow value = 0 }
	}
	
	#Inherent decay
	multiply_variable = { which = Prov_NOPowEntrench value = 0.995 }

	set_variable = { which = Tmp_0 which = Prov_NOPow }
	subtract_variable = { which = Tmp_0 which = Prov_NOPowRest }
	multiply_variable = { which = Tmp_0 value = 0.02 }
	change_variable = { which = Prov_NOPowEntrench which = Tmp_0 }
	
	if = {
		limit = {
			NOT = { check_variable = { which = Prov_NOPowEntrench value = 0 } }
		}
		set_variable = { which = Prov_NOPowEntrench value = 0 }
	}

	### Treatment effects, essentially class concern due to power changes
	# Half effect (PowChange * 0.5), representing the degree that the class can accept self-responsibiliy for their changes
	set_variable = { which = Tmp_0 which = Prov_NOPowChange }
	multiply_variable = { which = Tmp_0 value = 0.005 }
	Public_ChangeConcern = { Concern=Hierarchy type=which inp=Tmp_0 }

	set_variable = { which = Tmp_0 value = 0 }
	set_variable = { which = Tmp_1 value = 0 }
}

Prov_SetRateDisp = {
	if = {
		limit = {
			check_variable = { which = $class$_Total value = 0.01 }
			check_variable = { which = $class$_SpendTax value = 0.001 }
		}
		if = {
			limit = {
				is_variable_equal = { which = $class$_Income value = 0 }
			}
			set_variable = { which = $class$_SpendTaxR value = 0 }
		}
		else = {
			set_variable = { which = $class$_SpendTaxR which = $class$_SpendTax }
			divide_variable = { which = $class$_SpendTaxR which = $class$_Income }
			multiply_variable = { which = $class$_SpendTaxR value = 100 }
		}
		
		if = {
			limit = {
				is_variable_equal = { which = $class$_MP value = 0 }
			}
			set_variable = { which = $class$_MPTaxR value = 0 }
		}
		else = {
			set_variable = { which = $class$_MPTaxR which = $class$_MPTax }
			divide_variable = { which = $class$_MPTaxR which = $class$_MP }
			multiply_variable = { which = $class$_MPTaxR value = 100 }
		}
	}
	else = {
		set_variable = { which = $class$_SpendTaxR value = 0 }
		set_variable = { which = $class$_MPTaxR value = 0 }
	}
}

Tax_CalcClass = {
	if = {
		limit = {
			check_variable = { which = $Class$_Total value = 0.01 }
		}
		## TaxBase == class value to multiply by the base rate, such as $Class$_Wealth, $Class$_Income, $Class$_Surplus
		[[TaxBase]
		set_variable = { which = Tmp_1 which = Tax_TmpBase }
		Tax_CalcAmount = { RateBaseVar=$Class$_$TaxBase$ }
		]
		## TaxBase but less precision to avoidoverflow
		[[TaxBaseOverflow]
			if = {
				limit = {
					always = $TaxBaseOverflow$
				}
				set_variable = { which = Tmp_1 which = Tax_TmpBase }
				Tax_CalcAmountPoll = { RateBaseVar=$Class$_Total }
			}
		]
		## AnyBase == any value to multiply by the base rate, such as Investment_Income, etc
		[[AnyBase]
			set_variable = { which = Tmp_1 which = Tax_TmpBase }
			Tax_CalcAmount = { RateBaseVar=$AnyBase$ }
		]
		## if taxing an elite, use their privileges, and multiplier
		[[Notable]
			if = {
				limit = {
					always = $Notable$
				}
				multiply_variable = { which = Tmp_1 which = Privilege_$Class$Tax }
				multiply_variable = { which = Tmp_1 which = $Class$_Multi }
			}
		]
		## also scale by property
		[[Property]
			multiply_variable = { which = Tmp_1 which = $Class$_Prp$Property$ }
			multiply_variable = { which = Tmp_1 value = 0.01 }
		]
		## if burgher, reduce tax by the trade post influence if existent
		[[Burgher]
			if = {
				limit = {
					always = $Burgher$
					Post_Has = yes
				}
				multiply_variable = { which = Tmp_1 which = Post_LocalInfluence }
				multiply_variable = { which = Tmp_1 value = 0.01 }
			}
		]
		# multiply by a fixed value
		[[MultiplierVal]
			multiply_variable = { which = Tmp_1 value = $MultiplierVal$ }
		]
		# multiply by a variable
		[[Multipliervariable]
			multiply_variable = { which = Tmp_1 which = $Multipliervariable$ }
		]
		[[building]
		if = {
			limit = {
				always = $building$
			}
			change_variable = { which = TaxD_$TaxType$High which = Tmp_1 }			
			multiply_variable = { which = Tmp_1 which = Tax_TmpScalar }
		}
		]
		### actually deduct tax
		if = {
			limit = {
				check_variable = { which = Tmp_1 value = 0.001 }
			}
			Tax_CalcDucatBase = { Class=$Class$ TaxType=$TaxType$ [[Bureau] Bureau=$Bureau$ ] [[Flow] Flow=$Flow$ ] }
		}
	}
}

		subtract_variable = { which = Tmp_9 value = 5 } # base amount of wealth
		subtract_variable = { which = Tmp_9 which = $Class$_SpendTax }
		if = {
			limit = {
				check_variable = { which = Tmp_9 value = 0.001 }
			}
			# Dont take more than is present
			if = {
				limit = {
					NOT = { check_variable = { which = Tmp_9 which = Tmp_1 } }
				}
				set_variable = { which = Tmp_1 which = Tmp_9 }
			}
			change_variable = { which = Tmp_2 which = Tmp_1 }
		}
		else = {
			set_variable = { which = Tmp_1 value = 0 }
		}
		set_variable = { which = Tmp_9 value = 0 }
	}
	else = {
		set_variable = { which = Tmp_1 value = 0 }
	}
}
