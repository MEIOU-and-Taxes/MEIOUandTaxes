# who is the one 'from' taxes from.
# $from$_SpendMP = Total expenditure
# $from$_SpendMPMaint = Expenditure from training
# $from$_SpendMPBuild = Expenditure from recruitment
Military_DoAfter = {
	set_variable = { which = $from$_SpendMP value = 0 }
	set_variable = { which = $from$_SpendMPMaint value = 0 }
	set_variable = { which = $from$_SpendMPBuild value = 0 }
	
    # Calculate how much it spends from training. Mil_MaintSpend is the cost of each unit. $from$_MPMaint total amount of units
	set_variable = { which = $from$_SpendMPMaint which = $from$_MPMaint }
	multiply_variable = { which = $from$_SpendMPMaint which = Modi_Mil$from$Cost }
	multiply_variable = { which = $from$_SpendMPMaint which = Modi_Mil$from$Maint }
	if = {
		limit = {
			check_variable = { which = $from$_SpendMPMaint value = 0.001 }
			check_variable = { which = Mil_MPMaint value = 0.001 }
		}
		divide_variable = { which = $from$_SpendMPMaint which = Mil_MPMaint }
	}
	else = {
		set_variable = { which = $from$_SpendMPMaint value = 0 }
	}
	multiply_variable = { which = $from$_SpendMPMaint which = Mil_MaintSpend }
	
	change_variable = { which = $from$_SpendMP which = $from$_SpendMPMaint }
	
	[[who0] Military_DoAfterHelper1 = { from = $from$ who = $who0$ } ]
	[[who1] Military_DoAfterHelper1 = { from = $from$ who = $who1$ } ]
	[[who2] Military_DoAfterHelper1 = { from = $from$ who = $who2$ } ]
	[[who3] Military_DoAfterHelper1 = { from = $from$ who = $who3$ } ]
	[[who4] Military_DoAfterHelper1 = { from = $from$ who = $who4$ } ]
	[[who5] Military_DoAfterHelper1 = { from = $from$ who = $who5$ } ]
	[[who6] Military_DoAfterHelper1 = { from = $from$ who = $who6$ } ]
	
	if = {
		limit = {
			check_variable = { which = $from$_MPBuild value = 0.001 }
		}
		set_variable = { which = $from$_SpendMPBuild which = $from$_MPBuild }
		multiply_variable = { which = $from$_SpendMPBuild which = Modi_Mil$from$Cost }
		multiply_variable = { which = $from$_SpendMPBuild which = Modi_Mil$from$Build }
		if = {
			limit = {
				check_variable = { which = $from$_SpendMPBuild value = 0.001 }
				check_variable = { which = Mil_MPBuild value = 0.001 }
			}
			divide_variable = { which = $from$_SpendMPBuild which = Mil_MPBuild }
		}
		else = {
			set_variable = { which = $from$_SpendMPBuild value = 0 }
		}
		multiply_variable = { which = $from$_SpendMPBuild which = Mil_BuildSpend }
		
		change_variable = { which = $from$_SpendMP which = $from$_SpendMPBuild }
		
		[[who0] Military_DoAfterHelper2 = { from = $from$ who = $who0$ } ]
		[[who1] Military_DoAfterHelper2 = { from = $from$ who = $who1$ } ]
		[[who2] Military_DoAfterHelper2 = { from = $from$ who = $who2$ } ]
		[[who3] Military_DoAfterHelper2 = { from = $from$ who = $who3$ } ]
		[[who4] Military_DoAfterHelper2 = { from = $from$ who = $who4$ } ]
		[[who5] Military_DoAfterHelper2 = { from = $from$ who = $who5$ } ]
		[[who6] Military_DoAfterHelper2 = { from = $from$ who = $who6$ } ]
	}
	
	set_variable = { which = $from$_MPCur value = 0 }
	set_variable = { which = $from$_MPBuildSize value = 0 }
	set_variable = { which = $from$_MPDisband value = 0 }
	[[who0] Military_TotalsSum = { from = $from$ who = $who0$ } ]
	[[who1] Military_TotalsSum = { from = $from$ who = $who1$ } ]
	[[who2] Military_TotalsSum = { from = $from$ who = $who2$ } ]

# Get amount of goods per category needed for maintenance / training
Military_GetMaint = {
	set_variable = { which = $out$1 value = 0.35 } # Food (0.75 ducat)
	set_variable = { which = $out$2 value = 0.1 }  # Salt (0.35 ducat)
	set_variable = { which = $out$5 value = 0.1 }  # Fuel (0.2 ducat)
	set_variable = { which = $out$21 value = 0.1 } # Consumer Product (0.6 ducat)
	set_variable = { which = $out$22 value = 0.25 } # Military Product (1.8 ducat)
	set_variable = { which = $out$24 value = 0.1 } # Industrial Product (0.25 ducat)
	set_variable = { which = $out$41 value = 0.025 }  # Knowledge (0.25 ducat)
	
	# Scale up 1, 2 and 5 to equal roughly the ducat value of no 21 and 24. 24 is split between 2 and 5
	# however, fuel costs are lower compared to industrial, so nomad provs will be a bit cheaper

	if = { 
		limit = {
			check_variable = { which = Prov_TRPow value = 0.001 }
			}
		set_variable = { which = Public_Tmp9 which = Prov_TRPow }
		multiply_variable = { which = Public_Tmp9 value = 0.017 } # Nomad provinces need more general goods
		change_variable = { which = Public_Tmp9 value = 1 }
		multiply_variable = { which = $out$1 which = Public_Tmp9 }

		set_variable = { which = Public_Tmp9 which = Prov_TRPow }
		multiply_variable = { which = Public_Tmp9 value = 0.006 } # Nomad provinces need more general goods
		change_variable = { which = Public_Tmp9 value = 1 }
		multiply_variable = { which = $out$2 which = Public_Tmp9 }

		set_variable = { which = Public_Tmp9 which = Prov_TRPow }
		multiply_variable = { which = Public_Tmp9 value = 0.01 } # Nomad provinces need more general goods
		change_variable = { which = Public_Tmp9 value = 1 }
		multiply_variable = { which = $out$5 which = Public_Tmp9 }

		set_variable = { which = Public_Tmp9 which = Prov_TRPow }
		multiply_variable = { which = Public_Tmp9 value = -0.009 } # Nomad provinces need less urban for armies
		change_variable = { which = Public_Tmp9 value = 1 }
		multiply_variable = { which = $out$21 which = Public_Tmp9 }
		multiply_variable = { which = $out$24 which = Public_Tmp9 }

		set_variable = { which = Public_Tmp9 which = Prov_TRPow }
		multiply_variable = { which = Public_Tmp9 value = -0.009 } # Clan provinces need less uni's for armies
		change_variable = { which = Public_Tmp9 value = 1 }
		multiply_variable = { which = $out$41 which = Public_Tmp9 }
	}
	
	[[size]
		multiply_variable = { which = $out$1 which = $size$ }
		multiply_variable = { which = $out$2 which = $size$ }
		multiply_variable = { which = $out$5 which = $size$ }
		multiply_variable = { which = $out$21 which = $size$ }
		multiply_variable = { which = $out$22 which = $size$ }
		multiply_variable = { which = $out$24 which = $size$ }
		multiply_variable = { which = $out$41 which = $size$ }

		owner = {
			export_to_variable = { which = Tmp_0 value = trigger_value:artillery_fraction }
						
			set_variable = { which = $out$5 which = PREV }
			set_variable = { which = $out$22 which = PREV }
			set_variable = { which = $out$24 which = PREV }
			set_variable = { which = $out$41 which = PREV }

			multiply_variable = { which = $out$5  which = Tmp_0 }
			multiply_variable = { which = $out$22 which = Tmp_0 }
			multiply_variable = { which = $out$24 which = Tmp_0 }
			multiply_variable = { which = $out$41 which = Tmp_0 }			

			multiply_variable = { which = $out$5  value = 0.1 }
			multiply_variable = { which = $out$22 value = 0.15 }
			multiply_variable = { which = $out$24 value = 0.15 }
			multiply_variable = { which = $out$41 value = 0.3 }

			PREV = {
				change_variable = { which = $out$5 which = PREV }
				change_variable = { which = $out$22 which = PREV }
				change_variable = { which = $out$24 which = PREV }
				change_variable = { which = $out$41 which = PREV }
			}
		}
	]
	
	### Military tech modifier on Good costs = (MilTech * 0.005 + 0.5)
	export_to_variable = { which = Public_Tmp9 value = mil_tech who = owner }
	multiply_variable = { which = Public_Tmp9 value = 0.005 }	
	change_variable = { which = Public_Tmp9 value = 0.5 }
	
	### Modify certain good needs by mil tech modifier
	multiply_variable = { which = $out$22 which = Public_Tmp9 } # Military Product
	multiply_variable = { which = $out$24 which = Public_Tmp9 } # Industrial Product
	multiply_variable = { which = $out$41 which = Public_Tmp9 } # Knowledge
	
	### Modify all good needs by training modifier
	multiply_variable = { which = $out$1 which = Modi_MilTrain }
	multiply_variable = { which = $out$2 which = Modi_MilTrain }
	multiply_variable = { which = $out$5 which = Modi_MilTrain }
	multiply_variable = { which = $out$21 which = Modi_MilTrain }
	multiply_variable = { which = $out$22 which = Modi_MilTrain }
	multiply_variable = { which = $out$24 which = Modi_MilTrain }
	multiply_variable = { which = $out$41 which = Modi_MilTrain }
}

Military_GetBudget = {
	set_variable = { which = $from$_MilBudget value = 0 }
	if = { # If Wealth >= 1 and Class pop exist in province calculate budget
		limit = {
			check_variable = { which = $wealth$ value = 1 } 
		}
		# Calculate base budget - (25 + ClassPower) * Wealth / 125 * 0.04 * Modi_MilClassBudget
		set_variable = { which = $from$_MilBudget  value = 25 }
	
		[[clans]
		if = {
			limit = {
				always = $clans$
			}
			set_variable = { which = Public_Tmp9 which = Prov_TRPow }
			multiply_variable = { which = Public_Tmp9 value = 1.5 }
			if = {
				limit = {
					check_variable = { which = Public_Tmp9 value = 100 }
				}
				set_variable = { which = Public_Tmp9 value = 100 }
			}
			change_variable = { which = $from$_MilBudget  which = Public_Tmp9 }
			# If clans multiply budget with (TRPow*2*0.01)+Budget (Boundaries: upper TRPow = 50, lower TRPow = 2.5)
			if = {
				limit = {
					check_variable = { which = Prov_TRPow value = 50 }
				}
				set_variable = { which = Public_Tmp9 value = 100 }
			}
			else = {
				set_variable = { which = Public_Tmp9 which = Prov_TRPow }
				multiply_variable = { which = Public_Tmp9 value = 2 }
				if = {
					limit = {
						NOT = { check_variable = { which = Public_Tmp9 value = 5 } }
					}
					set_variable = { which = Public_Tmp9 value = 5 }
				}
			}
			multiply_variable = { which = $from$_MilBudget  which = Public_Tmp9 }
			multiply_variable = { which = $from$_MilBudget  value = 0.01 }
		}
		]
		[[elite]
		if = {
			limit = {
				always = $elite$
			}
			change_variable = { which = $from$_MilBudget  which = Prov_$from$Pow } # Elites are simply + power
			# If non clans elite subtract clans part of budget - ((100 - TRPow)*0.01) * Budget
			if = {
				limit = {
					check_variable = { which = Prov_TRPow value = 50 }
				}
				set_variable = { which = Public_Tmp9 value = 0 }
			}
			else = {
				set_variable = { which = Public_Tmp9 value = 50 }
				subtract_variable = { which = Public_Tmp9 which = Prov_TRPow }
				multiply_variable = { which = Public_Tmp9 value = 2 }
				multiply_variable = { which = $from$_MilBudget  which = Public_Tmp9 }
				multiply_variable = { which = $from$_MilBudget  value = 0.01 }
			}
		}
		]
		multiply_variable = { which = $from$_MilBudget  which = $wealth$ }
		if = {
			limit = {
				check_variable = { which = $from$_MilBudget value = 0.01 } 
			}
			divide_variable = { which = $from$_MilBudget  value = 125 } # /125
			multiply_variable = { which = $from$_MilBudget  value = 0.04 } # /20
			multiply_variable = { which = $from$_MilBudget  which = Modi_Mil$from$Budget }
		}
	}
}

Mil_DoBudgetBU = {
	### Get Military Expenside Slider Position
	if = {
		limit = {
			OR = {
				check_variable = { which = Mil_BUSlider value = 1.001 }
				NOT = { check_variable = { which = Mil_BUSlider value = 0.3 } } ## min from the defines, also change the calc of the slider
			}
		}
		set_variable = { which = Mil_BUSlider value = 1 }
	}
	else_if = {
		limit = {
			NOT = { check_variable = { which = Mil_BUSlider value = 0.333 } } ## rounding errors
		}
		set_variable = { which = Mil_BUSlider value = 0.333 }
	}
	set_variable = { which = Mil_BUSliderSave which = Mil_BUSlider } # Clear country variable
	# Add Slider variable to all provinces - Mil_BUSlider added monthly by event POP_Stockpile.002 in pulse system
	every_owned_province = {
		set_variable = { which = MP_Losses value = 0 }
		set_variable = { which = Mil_BUSliderSave which = PREV }
	}
	set_variable = { which = Mil_BUSlider value = 0 } # Clear country variable
	####
	# HANDLE BUDGETING EXCEPTIONS
	###
	## this is all yearly income
	## use the tax income
	set_variable = { which = BU_MilBudget which = TaxD_Total }
	subtract_variable = { which = BU_MilBudget value = 5 } ## factor out base Familial Estates income
	if = {
		limit = {
			check_variable = { which = BU_MilBudget value = 0.01 }
		}
		multiply_variable = { which = BU_MilBudget value = 0.01 } ## full budget Tmp_1
		multiply_variable = { which = BU_MilBudget which = Mil_BUBudgetMult } ## BUDGETTING % IN MENU
	}
	
	change_variable = { which = BU_MilBudget value = 5 } ## add back 5 ducats of base Familial Estates income

	set_variable = { which = Tmp_0 value = 0 }
	set_variable = { which = Mil_StateSpendMulti value = 0 }
	## Calculate Total and per Province taxed manpower
	every_owned_province = {
		set_variable = { which = BU_MilBudget value = 0 }
		# Add up all taxed manpower in Tmp_0
		set_variable = { which = Tmp_0 which = BU_MPSF }
		change_variable = { which = Tmp_0 which = BU_MPRE }
		change_variable = { which = Tmp_0 which = BU_MPNO }
		change_variable = { which = Tmp_0 which = BU_MPBG }
		change_variable = { which = Tmp_0 which = BU_MPCL }
		change_variable = { which = Tmp_0 which = BU_MPSpecial }
		if = {
			limit = {
				check_variable = { which = Tmp_0 value = 0.1 }
			}
			multiply_variable = { which = Tmp_0 value = 0.01 } # can reduce it without needing to rescale it
		}
		else_if = {
			limit = {
				check_variable = { which = Tmp_0 value = 0.001 }
			}
			set_variable = { which = Tmp_0 value = 0.001 }
		}
		else = {			#Try to prevent negative values
			set_variable = { which = Tmp_0 value = 0 }
		}
		PREV = { change_variable = { which = Tmp_0 which = PREV } } # Store total sum in owner

		### Also, we're going to figure out potential MP costs
		# Sum up the maximum manpower
		set_variable = { which = Mil_StateSpendMulti value = 0 }
		change_variable = { which = Mil_StateSpendMulti which = Mil_VolunteerHigh }
		change_variable = { which = Mil_StateSpendMulti which = Mil_LevyHigh }
		change_variable = { which = Mil_StateSpendMulti which = Mil_ConscriptHigh }
		# Price this manpower
		multiply_variable = { which = Mil_StateSpendMulti which = Mil_CostMaint }
		multiply_variable = { which = Mil_StateSpendMulti which = Modi_MilBUCost }
		multiply_variable = { which = Mil_StateSpendMulti which = Modi_MilBUMaint }
		if = {
			limit = {
				check_variable = { which = Mil_StateSpendMulti value = 0.001 }
			}
			multiply_variable = { which = Mil_StateSpendMulti value = 0.01 } # can reduce it without needing to rescale it
			if = {
				limit = {
					NOT = { check_variable = { which = Mil_StateSpendMulti value = 0.001 } }
				}
				set_variable = { which = Mil_StateSpendMulti value = 0.001 }
			}
		}
		PREV = { change_variable = { which = Mil_StateSpendMulti which = PREV } } # Store total sum in owner
	}
	set_variable = { which = Tmp_1 which = Tmp_0 }
	
	# Now we need to figure out a stat for AI Tax, so it can understand that all this potential manpower is liekly not actually achievable
	set_variable = { which = Tmp_2 which = BU_MilBudget }
	multiply_variable = { which = Tmp_2 value = 0.01 } # can reduce it without needing to rescale it
	if = {
		limit = {
			NOT = { check_variable = { which = Tmp_2 value = 0.001 } }
		}
		set_variable = { which = Tmp_2 value = 0.001 }
	}
	# If Potential Manpower is >>> Budgeted Manpower, set the AI Delegator to see that
	if = {
		limit = {
			check_variable = { which = Mil_StateSpendMulti which = Tmp_2 }

		# }
		# else = {

		# }
		if = {
			limit = {
				check_variable = { which = Tmp_0 value = 0.001 }
			}
			set_variable = { which = Tmp_1 which = PREV }
			set_variable = { which = BU_MilBudget which = PREV }
			if = {
				limit = {
					check_variable = { which = Tmp_1 value = 0.001 }
				}
				divide_variable = { which = Tmp_0 which = Tmp_1 }
			}
			multiply_variable = { which = BU_MilBudget which = Tmp_0 }
		}
		# Universally apply this visibility
	}

Military_GetCost = {
	Military_Get$type$ = { out = Mil_ } # Get amount of goods per category needed for $type$
	
	### Multiply Goods need by Sector Price to get costs
	multiply_variable = { which = Mil_1 which = TN_ProvPrc1 }   # Food
	multiply_variable = { which = Mil_2 which = TN_ProvPrc2 }   # Salt
	multiply_variable = { which = Mil_5 which = TN_ProvPrc5 }   # Fuel
	multiply_variable = { which = Mil_21 which = TN_ProvPrc21 } # Consumer Product
	multiply_variable = { which = Mil_22 which = TN_ProvPrc22 } # Military Product
	multiply_variable = { which = Mil_24 which = TN_ProvPrc24 } # Industrial Product
	multiply_variable = { which = Mil_41 which = TN_ProvPrc41 } # Knowledge
	
	### Sum up total costs in $return$
	set_variable = { which = $return$ which = Mil_1 }     # Food
	change_variable = { which = $return$ which = Mil_2 }  # Salt
	change_variable = { which = $return$ which = Mil_5 }  # Fuel
	change_variable = { which = $return$ which = Mil_21 } # Consumer Product
	change_variable = { which = $return$ which = Mil_22 } # Military Product
	change_variable = { which = $return$ which = Mil_24 } # Industrial Product
	change_variable = { which = $return$ which = Mil_41 } # Knowledge
}
