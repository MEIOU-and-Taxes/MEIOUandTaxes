### This function calculates the potential of noble recruitment and levy of peasants by the nobles
#$class$_MP: Capable manpower (fightning age men)
# Output values:
# NO_MPNO | NO_MPSF- Leviable manpower of noble / peasant class
# Mil_Noble - Total leviable manpower
Military_DoPreNO = {
	set_variable = { which = Mil_Noble value = 0 }
	set_variable = { which = Mil_NobleHigh value = 0 }
	set_variable = { which = NO_MPSpecial value = 0 }
	set_variable = { which = NO_MPNO value = 0 }
	set_variable = { which = NO_MPSF value = 0 }

	### Handling levied troops of Noble Population Class
	if = {
		limit = {
			isValidProv = yes
			
			owner = {
				Tax_LevyNobleCan = yes # Right Noble Ser < 4 and  Rural Governance < 5
				Rights_NOObl3HasGreater = no # Noble Obligations < 3
			}
		}
        #set how much of NOBLE manpower can be levied, depending on Noble obligations. Higher privliges for NO, higher the levy amount.
		if = {
			limit = {
				owner = { Rights_NOObl1Has = yes }
			}
			set_variable = { which = NO_MPNO value = 0.8 }
		}
		else = {
			set_variable = { which = NO_MPNO value = 0.4 }
		}
		
		multiply_variable = { which = NO_MPNO which = Modi_Manpower }
		if = { ### Handle Max Boundary
			limit = {
				check_variable = { which = NO_MPNO value = 1 }
			}
			set_variable = { which = NO_MPNO value = 1 }
		}
		else_if = { ### HANDLE Lower boundary
			limit = {
				NOT = { check_variable = { which = NO_MPNO value = 0.01 } }
			}
			set_variable = { which = NO_MPNO value = 0 }
		}
		multiply_variable = { which = NO_MPNO which = NO_MP } # Derive potentially leviable noble pops by fighting age men * base rate


		#Reduce levy size the higher the Tribal power is.
		if = {
			limit = {
				check_variable = { which = Prov_TRPow value = 0.001 }
			}
			if = {
				limit = {
					NOT = { check_variable = { which = Prov_TRPow value = 50.001 } }
				}
				set_variable = { which = Tmp_0 value = 50 }
				subtract_variable = { which = Tmp_0 which = Prov_TRPow }
				multiply_variable = { which = Tmp_0 value = 2 }
				multiply_variable = { which = NO_MPNO which = Tmp_0 }
				multiply_variable = { which = NO_MPNO value = 0.01 }
			}
			else = {
				set_variable = { which = NO_MPNO value = 0 }
			}
		}
	}
	else = {
		set_variable = { which = NO_MPNO value = 0 }
	}
	
	#Calculate how much Nobles can raise from peasantry based on Military Organisation privs. NO_MPSF is how much nobles can raise from peasantry, the exploiting bastards. Where is William Wallace when we need it.
	if = {
		limit = {
			isValidProv = yes
			
			owner = { Tax_LevyNobleCan = yes } # Right Noble Ser < 4 and  Rural Governance < 5
		}
		set_variable = { which = NO_MPSF which = NO_RuralPowerScalar }
		

		### Get base manpower tax rate from rights and privileges
		
		# Processing Monopoly (already in 100 scale)
		if = {
			limit = {
				owner = { Privilege_Has = { Privilege=NOProcessing Lvl=2 } }
			}
			multiply_variable = { which = NO_MPSF value = 0.4 }
		}
		else_if = {
			limit = {
				owner = { Privilege_Has = { Privilege=NOProcessing Lvl=1 } }
			}
			multiply_variable = { which = NO_MPSF value = 0.375 }
		}
		else = {
			multiply_variable = { which = NO_MPSF value = 0.35 }
		}
        # Military Organisation Reform
		if = {
			limit = {
				owner = {
					Rights_NOSer1Has = yes
				}
			}
			set_variable = { which = Tmp_4 value = 1 }
		}
		else_if = {
			limit = {
				owner = {
					Rights_NOSer2Has = yes
				}
			}
			set_variable = { which = Tmp_4 value = 0.7 }
		}
		else = {
			set_variable = { which = Tmp_4 value = 0.4 }
		}

		### If inverted state reach modifier is smaller than military organisation reform modifier use it, else use MilOrg modifier 
		set_variable = { which = Prov_BUPowInv value = 100 }
		subtract_variable = { which = Prov_BUPowInv which = Prov_BUPow }
		multiply_variable = { which = Prov_BUPowInv value = 0.006 }
		change_variable = { which = Prov_BUPowInv value = 0.4 }
		if = {
			limit = {
				check_variable = { which = Tmp_4 which = Prov_BUPowInv }
			}
			multiply_variable = { which = NO_MPSF which = Prov_BUPowInv }
		}
		else = {
			multiply_variable = { which = NO_MPSF which = Tmp_4 }
		}

		# Clan Power multiplier on Noble levies - (100 - TRPow)*0.01
		if = {
			limit = {
				check_variable = { which = Prov_TRPow value = 0.001 }
			}
			if = {
				limit = {
					NOT = { check_variable = { which = Prov_TRPow value = 50.001 } }
				}
				set_variable = { which = Tmp_0 value = 50 }
				subtract_variable = { which = Tmp_0 which = Prov_TRPow }
				multiply_variable = { which = Tmp_0 value = 2 }
				multiply_variable = { which = NO_MPSF which = Tmp_0 }
				multiply_variable = { which = NO_MPSF value = 0.01 }
			}
			else = {
				set_variable = { which = NO_MPSF value = 0 }
			}
		}
		
		# Derived base tax multiplied with 0,01 (not sure why, probably balance ?)
		multiply_variable = { which = NO_MPSF value = 0.01 }
		multiply_variable = { which = NO_MPSF which = Modi_Manpower }
		if = { ### Handle Max Boundary
			limit = {
				check_variable = { which = NO_MPSF value = 1 }
			}
			set_variable = { which = NO_MPSF value = 1 }
		}
		else_if = { ### HANDLE Lower boundary
			limit = {
				NOT = { check_variable = { which = NO_MPSF value = 0.01 } }
			}
			set_variable = { which = NO_MPSF value = 0 }
		}
		
        
		set_variable = { which = Tmp_0 which = SF_MP } # Get leviable SF raw manpower pool
		if = {
			limit = { ### Familial Estates dont get levied
				is_capital = yes
				check_variable = { which = SF_Total value = 4 }
			}
			subtract_variable = { which = Tmp_0 value = 3 }
		}
		if = { ### HANDLE Lower boundary
			limit = {
				NOT = { check_variable = { which = Tmp_0 value = 0.01 } }
			}
			set_variable = { which = Tmp_0 value = 0 }
		}
		multiply_variable = { which = NO_MPSF which = Tmp_0 } # Peasantry manpower that can be raised by the Noblity, 

		# Calculate theoretically recruited manpower lvls for AI
		change_variable = { which = Mil_Noble which = NO_MPSF } ## Add levied peasants to the total
		change_variable = { which = Mil_Noble which = NO_MPNO } ## Add noble manpower to the total
		change_variable = { which = Mil_Noble which = NO_MPSpecial }

		set_variable = { which = Mil_NobleHigh which = Mil_Noble }
		
		set_variable = { which = Prov_BUPowInv value = 0 }
	}
	else = {
		set_variable = { which = NO_MPSF value = 0 }
	}
}

### This function calculates the potential of burgher recruitment and levy of peasants by the nobles
# $class$_MP: Capable manpower (fightning age men)
# Output values:
# BG_MPBG | BG_MPRE - Leviable manpower of burgher / resident class
# Mil_Burgher - Total potential citizen militia
Military_DoPreBG = {
	set_variable = { which = Mil_Burgher value = 0 }
	set_variable = { which = BG_MPSpecial value = 0 }
	### Handle City recruitments
	if = {
		limit = {
			isValidProv = yes
			
			owner = { Tax_LevyUrbanCan = yes }
		}

		# Derive base rate from rights
		if = {
			limit = {
				owner = {
					Rights_BGGov6HasGreater = yes
				}
			}
			set_variable = { which = BG_MPBG value = 0.5 }
			set_variable = { which = BG_MPRE value = 0.5 }
		}
		else = {
			set_variable = { which = BG_MPBG value = 0.3 }
			set_variable = { which = BG_MPRE value = 0.3 }
		}
		multiply_variable = { which = BG_MPBG which = Modi_Manpower }
		multiply_variable = { which = BG_MPRE which = Modi_Manpower }
		if = { ### Handle Max Boundary
			limit = {
				check_variable = { which = BG_MPBG value = 1 }
			}
			set_variable = { which = BG_MPBG value = 1 }
		}
		else_if = { ### HANDLE Lower boundary
			limit = {
				NOT = { check_variable = { which = BG_MPBG value = 0.01 } }
			}
			set_variable = { which = BG_MPBG value = 0 }
		}
		if = { ### Handle Max Boundary
			limit = {
				check_variable = { which = BG_MPRE value = 1 }
			}
			set_variable = { which = BG_MPRE value = 1 }
		}
		else_if = { ### HANDLE Lower boundary
			limit = {
				NOT = { check_variable = { which = BG_MPRE value = 0.01 } }
			}
			set_variable = { which = BG_MPRE value = 0 }
		}
		
		# Multiply Base Rate with Raw Manpower pool
		multiply_variable = { which = BG_MPBG which = BG_MP }
		multiply_variable = { which = BG_MPRE which = RE_MP }
		
		# Multiply potential manpower with (Burgher Power * 0.01)
		multiply_variable = { which = BG_MPRE which = Prov_BGPow }
		multiply_variable = { which = BG_MPRE value = 0.01 }
		
		# Clan Power multiplier on Citizen militias: (100 - TRPow)*0.01
		if = {
			limit = {
				check_variable = { which = Prov_TRPow value = 0.001 }
			}
			if = {
				limit = {
					NOT = { check_variable = { which = Prov_TRPow value = 50.001 } }
				}
				set_variable = { which = Tmp_0 value = 50 }
				subtract_variable = { which = Tmp_0 which = Prov_TRPow }
				multiply_variable = { which = Tmp_0 value = 2 }
				multiply_variable = { which = BG_MPBG which = Tmp_0 }
				multiply_variable = { which = BG_MPRE which = Tmp_0 }
				multiply_variable = { which = BG_MPBG value = 0.01 }
				multiply_variable = { which = BG_MPRE value = 0.01 }
			}
			else = {
				set_variable = { which = BG_MPBG value = 0 }
				set_variable = { which = BG_MPRE value = 0 }
			}
		}
		
		# Calculate potentially recruited manpower for AI
		set_variable = { which = Mil_Burgher which = BG_MPBG }
		change_variable = { which = Mil_Burgher which = BG_MPRE }
		change_variable = { which = Mil_Burgher which = BG_MPSpecial }
		set_variable = { which = Mil_BurgherHigh which = Mil_Burgher }
	}
	else = {
		set_variable = { which = BG_MPBG value = 0 }
		set_variable = { which = BG_MPRE value = 0 }
		set_variable = { which = Mil_BurgherHigh value = 0 }
	}
}


### This function calculates the potential of Clan Host recruitment and levy of peasants by the nobles
# $class$_MP: Capable manpower (fightning age men)
# Output values:
# TR_MPSF | TR_MPRE | TR_MPNM | TR_MPNO | TR_MPBG | TR_MPCL - tribal manpower of various class
# Mil_Clan - Total potential Clan Host manpower
# Mil_ClanHigh | Med | Low - Leviable tribal manpower by different tax levels
Military_DoPreTR = {
	set_variable = { which = Mil_Clan value = 0 }
	set_variable = { which = TR_MPSpecial value = 0 }

	### HANDLE Clan Host recruitment
	if = {
		limit = {
			owner = { Tax_LevyHordeCan = yes }
		}
		# Base rate 0.5 * TribalPower * 0.01
		set_variable = { which = Tmp_0 value = 0.4 }
		set_variable = { which = Tmp_1 which = Prov_TRPow }
		multiply_variable = { which = Tmp_1 value = 0.02 } ## kicks in at 55%
		if = { # underflow protection
			limit = {
				check_variable = { which = Tmp_1 value = 1.001 }
			}
			set_variable = { which = Tmp_1 value = 1 }
		}
		multiply_variable = { which = Tmp_0 which = Tmp_1 }
		
		 # Get raw manpower pool values for all classes
		set_variable = { which = TR_MPSF which = SF_MP }
		if = {
			limit = { ### Familial Estates get levies for clans in Clan Hosts
				is_capital = yes
				check_variable = { which = SF_Total value = 4 }
				owner = { is_tribal = yes }
			}
			subtract_variable = { which = TR_MPSF value = 3 }
		}
		if = { # underflow protection
			limit = {
				NOT = { check_variable = { which = TR_MPSF value = 0.001 } }
			}
			set_variable = { which = TR_MPSF value = 0 }
		}
		set_variable = { which = TR_MPRE which = RE_MP }
		set_variable = { which = TR_MPNM which = NM_MP }
		set_variable = { which = TR_MPNO which = NO_MP }
		set_variable = { which = TR_MPBG which = BG_MP }
		set_variable = { which = TR_MPCL which = CL_MP }
		
		# Apply nomad government reform multpliers on raw manpower pool
		if = {
			limit = {
				owner = { has_country_flag = Modi_NomadMP }
			}
			if = {
				limit = {
					OR = {						
						owner = { has_reform = Pol_Identity3 }
						owner = { has_reform = Pol_Identity4 }
						}
						#check_variable = { which = Pol_Identity value = 2 }
					}
				multiply_variable = { which = TR_MPNM value = 1.5 }
			}
			else = {
				multiply_variable = { which = TR_MPNM value = 1.2 }
			}
		}

		multiply_variable = { which = Tmp_0 which = Modi_Manpower }
		if = { ### Handle Max Boundary
			limit = {
				check_variable = { which = Tmp_0 value = 0.5 }
			}
			set_variable = { which = Tmp_0 value = 0.5 }
		}
		else_if = { ### HANDLE Lower boundary
			limit = {
				NOT = { check_variable = { which = Tmp_0 value = 0.01 } }
			}
			set_variable = { which = Tmp_0 value = 0 }
		}
		
		# Multiply raw manpower with base tax rate
		multiply_variable = { which = TR_MPSF which = Tmp_0 }
		multiply_variable = { which = TR_MPRE which = Tmp_0 }
		multiply_variable = { which = TR_MPNM which = Tmp_0 }
		multiply_variable = { which = TR_MPNO which = Tmp_0 }
		multiply_variable = { which = TR_MPNO value = 2 } # 80% mobilisation rate
		multiply_variable = { which = TR_MPBG which = Tmp_0 }
		multiply_variable = { which = TR_MPCL which = Tmp_0 }

		if = {
			limit = { ### Familial Estates get levies for clans in Clan Hosts
				is_capital = yes
				check_variable = { which = SF_Total value = 4 }
				owner = { is_tribal = yes }
			}
			change_variable = { which = TR_MPSF value = 3 }
		}
		if = {
			limit = {
				check_variable = { which = TR_MPSF which = SF_MP }
			}
			set_variable = { which = TR_MPSF which = SF_MP }
		}
		
		# Sum up possible manpower of all classes
		change_variable = { which = Mil_Clan which = TR_MPSF }
		change_variable = { which = Mil_Clan which = TR_MPRE }
		change_variable = { which = Mil_Clan which = TR_MPNM }
		change_variable = { which = Mil_Clan which = TR_MPNO }
		change_variable = { which = Mil_Clan which = TR_MPBG }
		change_variable = { which = Mil_Clan which = TR_MPCL }
		change_variable = { which = Mil_Clan which = TR_MPSpecial }

		# Calculate potential manpower on tax lvls for AI		
		set_variable = { which = Mil_ClanHigh which = Mil_Clan }
	}
	else = {
		set_variable = { which = TR_MPSF value = 0 }
		set_variable = { which = TR_MPRE value = 0 }
		set_variable = { which = TR_MPNM value = 0 }
		set_variable = { which = TR_MPNO value = 0 }
		set_variable = { which = TR_MPBG value = 0 }
		set_variable = { which = TR_MPCL value = 0 }
		set_variable = { which = Mil_ClanHigh value = 0 }
	}
}


### Handle State Recruitment into actual manpower values
# Sources: Draft, Volunteer, Levy
Military_DoPreBU = {
	set_variable = { which = BU_MPSF value = 0 }
	set_variable = { which = BU_MPRE value = 0 }
	set_variable = { which = BU_MPNO value = 0 }
	set_variable = { which = BU_MPBG value = 0 }
	set_variable = { which = BU_MPCL value = 0 }
	set_variable = { which = BU_MPSpecial value = 0 }
	set_variable = { which = Mil_Levy value = 0 }
	set_variable = { which = Mil_Volunteer value = 0 }
	set_variable = { which = Mil_Conscript value = 0 }
	if = {
		limit = {
			owner = { check_variable = { which = BU_MercRate value = 0.01 } }
			check_variable = { which = Merc_MPShare value = 0.01 }
		}
		change_variable = { which = BU_MPSpecial which = Merc_MP }
	}
	if = {
		limit = {
			check_variable = { which = Ghazi_MP value = 0.01 }
		}
		change_variable = { which = BU_MPSpecial which = Ghazi_MP }
	}
	if = {
		limit = {
			check_variable = { which = Teuton_MP value = 0.01 }
		}
		change_variable = { which = BU_MPSpecial which = Teuton_MP }
	}
	if = {
		limit = {
			check_variable = { which = Knights_MP value = 0.01 }
		}
		change_variable = { which = BU_MPSpecial which = Knights_MP }	
	}
	
	### Handle Draft Recruitment
	# Tmp_0 = Recruitment rate
	if = {
		limit = {
			isValidProv = yes
			
			owner = { Tax_LevyDraftCan = yes } # Enabled Enlistment
		}
        set_variable = { which = Tmp_0 value = 0.5 } # Base 50% recruitment rate

		# 10% bonus to recruitment 
		if = {
			limit = {
				is_capital = yes
			}
			change_variable = { which = Tmp_0 value = 0.1 }
		}
		if = { # Service Reform 4 Mulitplier 150% on base recruitment rate
			limit = {
				owner = { Rights_BUSer4HasGreater = yes }
			}
			multiply_variable = { which = Tmp_0 value = 1.5 }
		}
		
		# Clan Power multiplier on BU recruitment - (100 - TRPow)*0.01
		if = {
			limit = {
				check_variable = { which = Prov_TRPow value = 0.001 }
			}
			if = {
				limit = {
					NOT = { check_variable = { which = Prov_TRPow value = 50.001 } }
				}
				set_variable = { which = Tmp_1 value = 50 }
				subtract_variable = { which = Tmp_1 which = Prov_TRPow }
				multiply_variable = { which = Tmp_1 value = 2 }
				multiply_variable = { which = Tmp_0 which = Tmp_1 }
				multiply_variable = { which = Tmp_0 value = 0.01 }
			}
			else = {
				set_variable = { which = Tmp_0 value = 0 }
			}
		}
		
		# Save capable manpower pool (fighting age men)
		set_variable = { which = Tmp_1 which = SF_MP }
		set_variable = { which = Tmp_2 which = RE_MP }
		set_variable = { which = Tmp_3 which = NO_MP }
		set_variable = { which = Tmp_4 which = BG_MP }
		
		# Clergy manpower pool only when having Clergy Obligation 3
		if = {
			limit = {
				owner = { Rights_CLObl3Has = yes }
			}
			set_variable = { which = Tmp_5 which = CL_MP }
		}
		else = {
			set_variable = { which = Tmp_5 value = 0 }
		}

		multiply_variable = { which = Tmp_0 which = Modi_Manpower }
		if = { ### Handle Max Boundary
			limit = {
				check_variable = { which = Tmp_0 value = 1 }
			}
			set_variable = { which = Tmp_0 value = 1 }
		}
		else_if = { ### HANDLE Lower boundary
			limit = {
				NOT = { check_variable = { which = Tmp_0 value = 0.01 } }
			}
			set_variable = { which = Tmp_0 value = 0 }
		}
		
		# Multiply raw manpower pool with recruitment rate
		multiply_variable = { which = Tmp_1 which = Tmp_0 } # SF
		multiply_variable = { which = Tmp_2 which = Tmp_0 } # RE
		multiply_variable = { which = Tmp_3 which = Tmp_0 } # NO
		multiply_variable = { which = Tmp_3 which = Privilege_NOService } # NO multiplier by Service privilege
		multiply_variable = { which = Tmp_4 which = Tmp_0 } # BG
		multiply_variable = { which = Tmp_4 which = Privilege_BGService } # BG multiplier by Service privilege
		multiply_variable = { which = Tmp_5 which = Tmp_0 } # CL
		multiply_variable = { which = Tmp_5 which = Privilege_CLService } # CL multiplier by Service privilege
		
		Mil_GetTaxRate = { TaxName=LevyDraft }
		
		# Sum up Manpower Pool * Recruitmentrate in variable Mil_Conscript
		set_variable = { which = Mil_Conscript which = Tmp_1 }
		change_variable = { which = Mil_Conscript which = Tmp_2 }
		change_variable = { which = Mil_Conscript which = Tmp_3 }
		change_variable = { which = Mil_Conscript which = Tmp_4 }
		change_variable = { which = Mil_Conscript which = Tmp_5 }
		change_variable = { which = Mil_Conscript which = BU_MPSpecial }

		# Save potential MP values for the AI by multiplying theoretic pool with theoretic rates
		set_variable = { which = Mil_ConscriptHigh which = Mil_Conscript }

		# Calculate actually recruited BU manpower
		set_variable = { which = Mil_Conscript which = Mil_ConscriptHigh }
		multiply_variable = { which = Mil_Conscript which = Mil_RateTmp }
		multiply_variable = { which = Tmp_1 which = Mil_RateTmp }
		multiply_variable = { which = Tmp_2 which = Mil_RateTmp }
		multiply_variable = { which = Tmp_3 which = Mil_RateTmp }
		multiply_variable = { which = Tmp_4 which = Mil_RateTmp }
		multiply_variable = { which = Tmp_5 which = Mil_RateTmp }
		
		# Assign calculated values to actual variables
		change_variable = { which = BU_MPSF which = Tmp_1 }
		change_variable = { which = BU_MPRE which = Tmp_2 }
		change_variable = { which = BU_MPNO which = Tmp_3 }
		change_variable = { which = BU_MPBG which = Tmp_4 }
		change_variable = { which = BU_MPCL which = Tmp_5 }
	}
	else = {
		set_variable = { which = Mil_ConscriptHigh value = 0 }
	}
	
	### Handle Volunteer Recruitment
	# Tmp_0 = Recruitment rate
	if = {
		limit = {
			isValidProv = yes
			
			owner = { Tax_LevyVoluntCan = yes }
		}
		# Base recruitment rate = 10%
        set_variable = { which = Tmp_0 value = 0.10 }

		# Capital bonus on recruitment rate = 2%
		if = {
			limit = {
				is_capital = yes
			}
			change_variable = { which = Tmp_0 value = 0.02 }
		}
		
		# Clan Power multiplier on Noble levies - (100 - TRPow)*0.01
		if = {
			limit = {
				check_variable = { which = Prov_TRPow value = 0.001 }
			}
			if = {
				limit = {
					NOT = { check_variable = { which = Prov_TRPow value = 50.001 } }
				}
				set_variable = { which = Tmp_1 value = 50 }
				subtract_variable = { which = Tmp_1 which = Prov_TRPow }
				multiply_variable = { which = Tmp_1 value = 2 }
				multiply_variable = { which = Tmp_0 which = Tmp_1 }

			limit = {
				NOT = { check_variable = { which = Tmp_1 value = -10 } }
			}
			set_variable = { which = Tmp_1 value = -10 }
		}
		multiply_variable = { which = Tmp_1 value = 0.1 }
		change_variable = { which = Tmp_1 value = 1 }

		set_variable = { which = Tmp_2 which = SF_MP }
		set_variable = { which = Tmp_3 which = RE_MP }
		set_variable = { which = Tmp_4 which = NO_MP }
		set_variable = { which = Tmp_5 which = BG_MP }
		set_variable = { which = Tmp_6 which = CL_MP }

		multiply_variable = { which = Tmp_0 which = Modi_Manpower }
		if = { ### Handle Max Boundary
			limit = {
				check_variable = { which = Tmp_0 value = 1 }
			}
			set_variable = { which = Tmp_0 value = 1 }
		}
		else_if = { ### HANDLE Lower boundary
			limit = {
				NOT = { check_variable = { which = Tmp_0 value = 0.01 } }
			}
			set_variable = { which = Tmp_0 value = 0 }
		}

		multiply_variable = { which = Tmp_3 which = Tmp_0 } # RE
		multiply_variable = { which = Tmp_4 which = Tmp_0 } # NO
		multiply_variable = { which = Tmp_5 which = Tmp_0 } # BG
		multiply_variable = { which = Tmp_6 which = Tmp_0 } # CL

		multiply_variable = { which = Tmp_5 which = Privilege_BGService } # BG * Burgher * ServicePrivilege Modi
		multiply_variable = { which = Tmp_6 which = Prov_CLLoy } # CL * Clergy Loyalty
		multiply_variable = { which = Tmp_6 which = Privilege_CLService } # CL * Clergy Service Privilege Modi
		multiply_variable = { which = Tmp_4 value = 0.05 } # Multiplier on elites 5%
		multiply_variable = { which = Tmp_5 value = 0.05 } # Multiplier on elites 5%
		multiply_variable = { which = Tmp_6 value = 0.05 } # Multiplier on elites 5%

		change_variable = { which = Mil_Volunteer which = Tmp_4 }

		change_variable = { which = Mil_Volunteer which = Tmp_6 }

		set_variable = { which = Mil_VolunteerHigh which = Mil_Volunteer }

		set_variable = { which = Mil_Volunteer which = Mil_VolunteerHigh }

		multiply_variable = { which = Tmp_2 which = Mil_RateTmp }

		multiply_variable = { which = Tmp_4 which = Mil_RateTmp }

		change_variable = { which = BU_MPNO which = Tmp_4 }
		change_variable = { which = BU_MPBG which = Tmp_5 }
		change_variable = { which = BU_MPCL which = Tmp_6 }

	else = {
		set_variable = { which = Mil_VolunteerHigh value = 0 }
	}
	
	# Handle Feudal Levy Recruitment
	if = {
		limit = {
			isValidProv = yes
			owner = { Tax_LevyFeudalCan = yes }

			}
			set_variable = { which = Tmp_0 value = 0.5 } # NOGov3 - 50%
		}

			limit = {
				owner = { Rights_NOGov4Has = yes }
			}
			set_variable = { which = Tmp_0 value = 0.35 } # NOGov4 or higher - 35%
		}

		# Capital bonus on base recruitment rate = 10%
		if = {
			limit = {
				is_capital = yes
			}
			change_variable = { which = Tmp_0 value = 0.1 }
		}

		multiply_variable = { which = Tmp_0 which = BU_PrpResources }
        multiply_variable = { which = Tmp_0 value = 0.01 }

		# Multiply base rate by noble service reforms
		if = {
			limit = {
				owner = { Rights_NOSer1Has = yes }
			}
			multiply_variable = { which = Tmp_0 value = 1 } # No Modification on NOSer1
		}
		else_if = {
			limit = {
				owner = { Rights_NOSer2Has = yes }
			}
			multiply_variable = { which = Tmp_0 value = 0.7 } # -30% Modification on NOSer2
		}
		else = {
			multiply_variable = { which = Tmp_0 value = 0.40 } # -60% Modification on NOSer3
		}
		
		# Clan Power multiplier on Noble levies - (100 - TRPow)*0.01
		if = {
			limit = {
				check_variable = { which = Prov_TRPow value = 0.001 }
			}
			if = {
				limit = {
					NOT = { check_variable = { which = Prov_TRPow value = 50.001 } }
				}
				set_variable = { which = Tmp_1 value = 50 }
				subtract_variable = { which = Tmp_1 which = Prov_TRPow }
				multiply_variable = { which = Tmp_1 value = 2 }
				multiply_variable = { which = Tmp_0 which = Tmp_1 }
				multiply_variable = { which = Tmp_0 value = 0.01 }
			}
			else = {
				set_variable = { which = Tmp_0 value = 0 }
			}
		}
		# Get Peasant base fighting age men (nobles only recruit from peasants)
		set_variable = { which = Tmp_1 which = SF_MP }

		if = {
			limit = { ### Familial Estates get levies for states in State Levies
				is_capital = yes
				check_variable = { which = SF_Total value = 4 }
				owner = { is_tribal = no }
			}
			subtract_variable = { which = Tmp_1 value = 3 }
		}

		multiply_variable = { which = Tmp_0 which = Modi_Manpower }
		if = { ### Handle Max Boundary
			limit = {
				check_variable = { which = Tmp_0 value = 1 }
			}
			set_variable = { which = Tmp_0 value = 1 }
		}
		else_if = { ### HANDLE Lower boundary
			limit = {
				NOT = { check_variable = { which = Tmp_0 value = 0.01 } }
			}
			set_variable = { which = Tmp_0 value = 0 }
		}

		multiply_variable = { which = Tmp_1 which = Tmp_0 }

		Mil_GetTaxRate = { TaxName=LevyFeudal }

		if = {
			limit = { ### Familial Estates get levies for states in State Levies
				is_capital = yes
				check_variable = { which = SF_Total value = 4 }
				owner = { is_tribal = no }
			}
			change_variable = { which = Tmp_1 value = 3 }
		}
		if = {
			limit = {
				check_variable = { which = Tmp_1 which = SF_MP }
			}
			set_variable = { which = Tmp_1 which = SF_MP }
		}

		# Calculate theoretically recruited manpower lvls for AI
		set_variable = { which = Mil_Levy which = Tmp_1 }
		set_variable = { which = Mil_LevyHigh which = Mil_Levy }

		# Calculate actuall recruited manpower
		set_variable = { which = Mil_Levy which = Mil_LevyHigh }
		multiply_variable = { which = Mil_Levy which = Mil_RateTmp }
		multiply_variable = { which = Tmp_1 which = Mil_RateTmp }
		
		# Set actual variable to calculated value
		change_variable = { which = BU_MPSF which = Tmp_1 }
	}
	else = {
		set_variable = { which = Mil_LevyHigh value = 0 }
	}
}

Military_MaintBuildCalcer = {
	# $from$_BudgetedMP === Potential Manpower
	# $from$_MPMaint === Current Manpower
	# We will fund all Current Manpower, but may not fund some of what we disband
	[[who0] Military_MaintBuildSum = { from = $from$ who = $who0$ } ]
	[[who1] Military_MaintBuildSum = { from = $from$ who = $who1$ } ]
	[[who2] Military_MaintBuildSum = { from = $from$ who = $who2$ } ]
	[[who3] Military_MaintBuildSum = { from = $from$ who = $who3$ } ]
	[[who4] Military_MaintBuildSum = { from = $from$ who = $who4$ } ]
	[[who5] Military_MaintBuildSum = { from = $from$ who = $who5$ } ]
	[[who6] Military_MaintBuildSum = { from = $from$ who = $who6$ } ]

	#Public_Tmp2 === Cost it would take to Mantain 1 unit
	set_variable = { which = Public_Tmp2 which = Mil_CostMaint } 
	multiply_variable = { which = Public_Tmp2 which = Modi_Mil$from$Cost }
	multiply_variable = { which = Public_Tmp2 which = Modi_Mil$from$Maint }
	#Public_Tmp3 === Cost it would take to Build 1 Unit
	set_variable = { which = Public_Tmp3 which = Mil_CostBuild }
	multiply_variable = { which = Public_Tmp3 which = Modi_Mil$from$Cost }
	multiply_variable = { which = Public_Tmp3 which = Modi_Mil$from$Build }

	# $from$_BudgetedMP === Convert this now into Budgeted Manpower
	if = {
		limit = {
			check_variable = { which = $from$_BudgetedMP value = 0.001 } # if there is any possible manpower to actually budget
			check_variable = { which = $from$_MilBudget value = 0.1 } # real budget, dont piss around with micro amounts
		}
		set_variable = { which = $from$_BudgetedMPCost which = $from$_BudgetedMP }
		multiply_variable = { which = $from$_BudgetedMPCost which = Public_Tmp2 }
		if = {
			limit = {
				check_variable = { which = $from$_BudgetedMPCost which = $from$_MilBudget } # cant be more than 100% of the budget
			}
			set_variable = { which = $from$_MilBudgetScalar which = $from$_MilBudget }
			divide_variable = { which = $from$_MilBudgetScalar which = $from$_BudgetedMPCost }
			multiply_variable = { which = $from$_BudgetedMP which = $from$_MilBudgetScalar } # Set Fully Budgeted to what Manpower the budget can afford
		}
		else = {
			### Leave $from$_BudgetedMP alone, its fully budgeted
			set_variable = { which = $from$_MilBudgetScalar value = 1 }
		}
	}
	else = {
		set_variable = { which = $from$_MilBudgetScalar value = 0 } # nothing
		set_variable = { which = $from$_BudgetedMP value = 0 } # no potential manpower
	}

	# Now finally, how much of the total pool are we building?
	if = {
		limit = {
			check_variable = { which = $from$_MPMaint which = $from$_BudgetedMP } # if its over budget, assume budget var has something
			NOT = { is_variable_equal = { which = $from$_MPMaint which = $from$_BudgetedMP } }
		}
		if = {
			limit = {
				NOT = { check_variable = { which = $from$_MPMaint value = 0.05 } }
			}
			set_variable = { which = Tmp_0 value = 1 } 
		}
		else = {
			# how much are we over by?
			set_variable = { which = Tmp_3 which = $from$_MPMaint } 
			subtract_variable = { which = Tmp_3 which = $from$_BudgetedMP }
			
			set_variable = { which = Tmp_4 which = $from$_BudgetedMP } # calc what 10% of budgeted looks like
			multiply_variable = { which = Tmp_4 value = 0.1 }
			set_variable = { which = Tmp_0 which = $from$_MPMaint } 
			multiply_variable = { which = Tmp_0 value = 0.2 } # calc what taking 20% of the overfunded manpower looks like

			# if 20% of overfunded is < 10% of budgeted, just take the full amount, no exponential weirdness.
			if = {
				limit = {
					check_variable = { which = Tmp_4 which = Tmp_0 }
				}
				set_variable = { which = Tmp_0 which = Tmp_3 } 
			}
			# Work out how much we are disbanding of the current military
			divide_variable = { which = Tmp_0 which = $from$_MPMaint } 
		}

		### Apply to all classes equally, note it Actually Disbands here
		[[who0] Military_DisbandCalcer = { from = $from$ who = $who0$ } ]
		[[who1] Military_DisbandCalcer = { from = $from$ who = $who1$ } ]
		[[who2] Military_DisbandCalcer = { from = $from$ who = $who2$ } ]
		[[who3] Military_DisbandCalcer = { from = $from$ who = $who3$ } ]
		[[who4] Military_DisbandCalcer = { from = $from$ who = $who4$ } ]
		[[who5] Military_DisbandCalcer = { from = $from$ who = $who5$ } ]
		[[who6] Military_DisbandCalcer = { from = $from$ who = $who6$ } ]
	}
	# Under Budget, so can Recruit
	else_if = {
		limit = {
			NOT = { check_variable = { which = $from$_MPMaint which = $from$_BudgetedMP } } # if its over budget, assume budget var has something
		}
		# Calc how much manpower is on the table
		set_variable = { which = $from$_MPBuild which = $from$_BudgetedMP } 
		subtract_variable = { which = $from$_MPBuild which = $from$_MPMaint }

		### Set minimum funding rate based on rights --> this determines how fast consript units can be build
		# Its a % of Budgeted manpower, like, 12.5% a year of that manpower, if > then cap, otherwise just finish it.
		set_variable = { which = Tmp_4 which = $from$_BudgetedMP }
		if = {
			limit = {
				owner = { has_country_modifier = ExtraFresh_State } # sped up at lower mil org
			}
			multiply_variable = { which = Tmp_4 value = 1 } # First year after spawning, will instantly train its troops
		}
		else_if = {
			limit = {
				owner = { has_country_modifier = Fresh_State } # sped up at lower mil org
			}
			multiply_variable = { which = Tmp_4 value = 0.5 } # First five years after spawning, will swiftly train its troops
		}
		else_if = {
			limit = {
				owner = { Rights_NOSer1Has = yes } # sped up at lower mil org
			}
			multiply_variable = { which = Tmp_4 value = 0.334 } # how much to fund at once, minimum
		}
		else_if = {
			limit = {
				owner = { Rights_NOSer2Has = yes } # sped up at lower mil org
			}
			multiply_variable = { which = Tmp_4 value = 0.275 } # how much to fund at once, minimum
		}
		else_if = {
			limit = {
				owner = { Rights_NOSer3Has = yes } # sped up at lower mil org
			}
			multiply_variable = { which = Tmp_4 value = 0.25 } # how much to fund at once, minimum
		}
		else_if = {
			limit = {
				owner = { Rights_NOSer4Has = yes } # sped up at lower mil org
			}
			multiply_variable = { which = Tmp_4 value = 0.225 } # how much to fund at once, minimum
		}
		else = {  # slow at higher mil org
			multiply_variable = { which = Tmp_4 value = 0.2 } # how much to fund at once, minimum
		}
		if = {
			limit = {
				check_variable = { which = $from$_MPBuild which = Tmp_4 } ## If Difference funding > potential manpower funding
			}
			set_variable = { which = $from$_MPBuild which = Tmp_4 } # Limit funding to this %
		}

		### Calc now how much the new manpower is as a portion of spare budget size
		# First how much it'll cost to build them
		set_variable = { which = Tmp_3 which = $from$_MPBuild }
		multiply_variable = { which = Tmp_3 which = Public_Tmp3 }

		# Figure out how much ongoing maintenance costs
		set_variable = { which = Tmp_4 which = $from$_MPMaint }
		multiply_variable = { which = Tmp_4 which = Public_Tmp2 }
		### Fun part, for designated recruiters, the Mil Slider will reduce their ongoing mainenance costs
		### This is so pivotal, because it allows them to now use this money to fund recruitment
		if = { limit = { always = $milSlider$ }
			multiply_variable = { which = Tmp_4 which = Mil_BUSliderSave }
		}
		
		# Create a wealth sum that doesnt count this
		set_variable = { which = Tmp_2 which = $from$_MilBudget }
		if = { limit = { NOT = { always = $milSlider$ } } # Elites will go a bit over budget to recruit
			multiply_variable = { which = Tmp_2 value = 1.1 }
		}
		subtract_variable = { which = Tmp_2 which = Tmp_4 }
		if = {
			limit = {
				check_variable = { which = Tmp_2 which = Tmp_3 }
			}
			# Scale to this sub-maint budget size
			set_variable = { which = Tmp_3 value = 1 }
		}
		else_if = {
			limit = {
				check_variable = { which = Tmp_2 value = 0.001 }
				check_variable = { which = Tmp_3 value = 0.001 }
			}
			# Scale to this sub-maint budget size
			set_variable = { which = Tmp_4 which = Tmp_2 }
			divide_variable = { which = Tmp_4 which = Tmp_3 }
			set_variable = { which = Tmp_3 which = Tmp_4 }
		}
		else = {
			set_variable = { which = Tmp_3 value = 0 }
		}

		# Only build this amount, obviously we cannot build what we cant afford
		multiply_variable = { which = $from$_MPBuild which = Tmp_3 }

		# Create a sum of on-the-table manpower from classes
		# We need to know how to disperse the BuildSize amount
		set_variable = { which = Tmp_4 value = 0 }
		[[who0] Military_BuildCalcer1 = { from = $from$ who = $who0$ } ]
		[[who1] Military_BuildCalcer1 = { from = $from$ who = $who1$ } ]
		[[who2] Military_BuildCalcer1 = { from = $from$ who = $who2$ } ]
		[[who3] Military_BuildCalcer1 = { from = $from$ who = $who3$ } ]
		[[who4] Military_BuildCalcer1 = { from = $from$ who = $who4$ } ]
		[[who5] Military_BuildCalcer1 = { from = $from$ who = $who5$ } ]
		[[who6] Military_BuildCalcer1 = { from = $from$ who = $who6$ } ]

		[[who0] Military_BuildCalcer2 = { from = $from$ who = $who0$ } ]
		[[who1] Military_BuildCalcer2 = { from = $from$ who = $who1$ } ]
		[[who2] Military_BuildCalcer2 = { from = $from$ who = $who2$ } ]
		[[who3] Military_BuildCalcer2 = { from = $from$ who = $who3$ } ]
		[[who4] Military_BuildCalcer2 = { from = $from$ who = $who4$ } ]
		[[who5] Military_BuildCalcer2 = { from = $from$ who = $who5$ } ]
		[[who6] Military_BuildCalcer2 = { from = $from$ who = $who6$ } ]
	}
	### Apply reduction in ongoing maintenance costs
	if = { limit = { always = $milSlider$ }
		multiply_variable = { which = $from$_MPMaint which = Mil_BUSliderSave }
	}
	[[who0] Military_MaintCalcer = { from = $from$ who = $who0$ } ]
	[[who1] Military_MaintCalcer = { from = $from$ who = $who1$ } ]
	[[who2] Military_MaintCalcer = { from = $from$ who = $who2$ } ]
	[[who3] Military_MaintCalcer = { from = $from$ who = $who3$ } ]
	[[who4] Military_MaintCalcer = { from = $from$ who = $who4$ } ]
	[[who5] Military_MaintCalcer = { from = $from$ who = $who5$ } ]
	[[who6] Military_MaintCalcer = { from = $from$ who = $who6$ } ]
	if = {
		limit = {
			NOT = { check_variable = { which = $from$_MPMaint value = 0 } }
		}
		set_variable = { which = $from$_MPMaint value = 0 }
	}
}

Mil_ManpowerLossCalc = {
	every_owned_province = {
		limit = {
			check_variable = { which = Stat_MP value = 0.001 }
			NOT = { has_global_flag = POP_Sim }
		}

		# Get previously assigned manpower stat
		set_variable = { which = Tmp_2 which = Stat_MP }
		change_variable = { which = Tmp_2 which = MP_Losses } # deduct manpower losses over the year

		# Calculate loss fraction based on how many losses vs. previous manpower stat
		if = {
			limit = {
				is_variable_equal = { which = Tmp_2 value = 0 }
			}
			set_variable = { which = Tmp_2 value = -1 }
		}
		else = {
			divide_variable = { which = Tmp_2 which = Stat_MP }
			subtract_variable = { which = Tmp_2 value = 1 }
		}

		### Distribute manpower losses over all provinces and classes
		if = {
			limit = {
				check_variable = { which = Stat_MPNO value = 0.001 }
			}
			set_variable = { which = Tmp_0 which = Stat_MPNO }
			multiply_variable = { which = Tmp_0 which = Tmp_2 }
			
			MP_Distrib = { from = NO who0 = SF who1 = NO who2 = Special amount = Tmp_0 }
		}
		if = {
			limit = {
				check_variable = { which = Stat_MPBG value = 0.001 }
			}
			set_variable = { which = Tmp_0 which = Stat_MPBG }
			multiply_variable = { which = Tmp_0 which = Tmp_2 }
			
			MP_Distrib = { from = BG who0 = RE who1 = BG who2 = Special amount = Tmp_0 }
		}
		if = {
			limit = {
				check_variable = { which = Stat_MPTR value = 0.001 }
			}
			set_variable = { which = Tmp_0 which = Stat_MPTR }
			multiply_variable = { which = Tmp_0 which = Tmp_2 }
			
			MP_Distrib = { from = TR who0 = SF who1 = RE who2 = NM who3 = NO who4 = BG who5 = CL who6 = Special amount = Tmp_0 }
		}
		if = {
			limit = {
				check_variable = { which = Tax_MP value = 0.001 }
			}
			set_variable = { which = Tmp_0 which = Tax_MP }
			multiply_variable = { which = Tmp_0 which = Tmp_2 }
			
			MP_Distrib = { from = BU who0 = SF who1 = RE who2 = NO who3 = BG who4 = CL who5 = Special amount = Tmp_0 }
		}
		
		### Refresh manpower variable percentage values after distributing losses
		MP_Refresh = { class = SF }
		MP_Refresh = { class = RE }
		MP_Refresh = { class = NM }
		MP_Refresh = { class = NO }
		MP_Refresh = { class = BG }
		MP_Refresh = { class = CL }
	}
	set_variable = { which = MP_Losses value = 0 }
}
