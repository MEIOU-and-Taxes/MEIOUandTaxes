Desire_DoGainAnnual = {
    if = {
        limit = {
            check_variable = { which = Prov_$type$Pow value = 1 }
        }
        Desire_GetEquilFactor = { type = $type$ return = Desire_$type$GainEquil }
        Desire_GetPowFactor = { type = $type$ return = Desire_$type$GainPow }
        Desire_GetPrivFactor = { type = $type$ return = Desire_$type$GainPriv }

        set_variable = { which = Desire_$type$Gain which = Desire_$type$GainEquil }
        change_variable = { which = Desire_$type$Gain which = Desire_$type$GainPow }
        change_variable = { which = Desire_$type$Gain which = Desire_$type$GainPriv }

        divide_variable = { which = Desire_$type$Gain value = 10 }
        
        change_variable = { which = Desire_$type$ which = Desire_$type$Gain }

        if = {
            limit = {
                NOT = {
                    check_variable = { which = Desire_$type$ value = 0 }
                }
            }
            set_variable = { which = Desire_$type$ value = 0 }
        }
        else_if = {
            limit = {
                check_variable = { which = Desire_$type$ value = 100 }
            }
            set_variable = { which = Desire_$type$ value = 100 }
        }
    }
    else = {
        set_variable = { which = Desire_$type$ value = 0 }
        set_variable = { which = Desire_$type$Gain value = 0 }
        set_variable = { which = Desire_$type$GainEquil value = 0 }
        set_variable = { which = Desire_$type$GainPow value = 0 }
        set_variable = { which = Desire_$type$GainPriv value = 0 }
    }
}
