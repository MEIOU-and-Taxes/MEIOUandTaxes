Unrest_Update = {
	if = {
		limit = {
			ai = yes
		}
		set_country_flag = Unrest_Update
	}
	else = {
		hidden_effect = { country_event = { id = POP_Census.113 } }
	}
}

Unrest_GetGainBanditry = {
    set_variable = { which = $return$ value = 0 }	
	# Welfare
    if = {
        limit = {
            NOT = {
                check_variable = { which = Concern_WelfareTreatR value = 0 }
            }
        }
        set_variable = { which = Tmp_0 which = Concern_WelfareTreatR }
        multiply_variable = { which = Tmp_0 value = -2 }

        change_variable = { which = $return$ which = Tmp_0 }

        set_variable = { which = Tmp_0 value = 0 }
    }
	# Welfare
    if = {
        limit = {
            is_colony = yes
        }
        multiply_variable = { which = $return$ value = 0.2 }
	}
}
Unrest_GetSuppressBanditry = {
	##### Armies
    set_variable = { which = $return$ value = 0 }
    change_variable = { which = $return$ which = Class_ArmyTR }
    divide_variable = { which = $return$ value = 4 }
    change_variable = { which = $return$ which = Class_ArmyNO }
    change_variable = { which = $return$ which = Class_ArmyBG }

    divide_variable = { which = $return$ which = Class_Total }
    multiply_variable = { which = $return$ value = 400 } #-4 unrest per 1% of POP
	
	# Oppression and Serfdom adds counterbalance, 50%, proportionate to Serf % of population
    if = {
        limit = {
            NOT = {
                check_variable = { which = Concern_WelfareTreatR value = 0 }
            }
            check_variable = { which = SF_Total value = 0.001 }
        }
        set_variable = { which = Tmp_0 which = SF_Total }
        divide_variable = { which = Tmp_0 which = Class_Total }
		
        set_variable = { which = Tmp_1 value = 100 }
        subtract_variable = { which = Tmp_1 which = SF_Freedom }
		
        multiply_variable = { which = Tmp_0 which = Tmp_1 }
        multiply_variable = { which = Tmp_0 value = 0.01 }
        multiply_variable = { which = Tmp_0 which = Concern_WelfareTreatR }
        multiply_variable = { which = Tmp_0 value = -1 }

        change_variable = { which = $return$ which = Tmp_0 }

        set_variable = { which = Tmp_0 value = 0 }
        set_variable = { which = Tmp_1 value = 0 }
    }
    if = {
        limit = {
            NOT = {
                check_variable = { which = $return$ value = 0 }
            }
        }
        set_variable = { which = $return$ value = 0 }
    }

    set_variable = { which = Tmp_0 value = 0 }
}

Unrest_GetGainRebellion = {
	### Elite Disloyalty
    set_variable = { which = $return$ value = 0 }

    if = {
        limit = {
            NOT = {
                check_variable = { which = Prov_NOLoy value = 50 }
            }
        }
        set_variable = { which = Tmp_0 value = 50 }
        subtract_variable = { which = Tmp_0 which = Prov_NOLoy }
        multiply_variable = { which = Tmp_0 value = 0.02 }
        multiply_variable = { which = Tmp_0 which = Tmp_0 }
        multiply_variable = { which = Tmp_0 which = Class_ArmyNO }
        multiply_variable = { which = Tmp_0 value = 2 }

        change_variable = { which = $return$ which = Tmp_0 }
    }
    if = {
        limit = {
            NOT = {
                check_variable = { which = Prov_BGLoy value = 50 }
            }
        }
        set_variable = { which = Tmp_0 value = 50 }
        subtract_variable = { which = Tmp_0 which = Prov_BGLoy }
        multiply_variable = { which = Tmp_0 value = 0.02 }
        multiply_variable = { which = Tmp_0 which = Tmp_0 }
        multiply_variable = { which = Tmp_0 which = Class_ArmyBG }
        multiply_variable = { which = Tmp_0 value = 2 }

        change_variable = { which = $return$ which = Tmp_0 }
    }
    if = {
        limit = {
            NOT = {
                check_variable = { which = Prov_TRLoy value = 50 }
            }
        }
        set_variable = { which = Tmp_0 value = 50 }
        subtract_variable = { which = Tmp_0 which = Prov_TRLoy }
        multiply_variable = { which = Tmp_0 value = 0.02 }
        multiply_variable = { which = Tmp_0 which = Tmp_0 }
        multiply_variable = { which = Tmp_0 which = Class_ArmyTR }

        change_variable = { which = $return$ which = Tmp_0 }
    }

    divide_variable = { which = $return$ which = Class_Total }
    multiply_variable = { which = $return$ value = 400 } #+4 unrest per 1% of POP

    if = {
        limit = {
            NOT = { check_variable = { which = $return$ value = 0 } }
        }
        set_variable = { which = $return$ value = 0 }
    }
	
    set_variable = { which = Tmp_0 value = 0 }
    set_variable = { which = Tmp_1 value = 0 }
    set_variable = { which = Tmp_2 value = 0 }
    set_variable = { which = Tmp_3 value = 0 }
}
Unrest_GetSuppressRebellion = {
    set_variable = { which = $return$ value = 0 }
	
	# Elite assistance
    if = {
        limit = {
            check_variable = { which = Prov_NOLoy value = 50 }
        }
        set_variable = { which = Tmp_0 which = Prov_NOLoy }
        subtract_variable = { which = Tmp_0 value = 50 }
        multiply_variable = { which = Tmp_0 value = 0.02 }
        multiply_variable = { which = Tmp_0 which = Class_ArmyNO }

        change_variable = { which = $return$ which = Tmp_0 }
    }
    if = {
        limit = {
            check_variable = { which = Prov_BGLoy value = 50 }
        }
        set_variable = { which = Tmp_0 which = Prov_BGLoy }
        subtract_variable = { which = Tmp_0 value = 50 }
        multiply_variable = { which = Tmp_0 value = 0.02 }
        multiply_variable = { which = Tmp_0 which = Class_ArmyBG }

        change_variable = { which = $return$ which = Tmp_0 }
    }
    if = {
        limit = {
            check_variable = { which = Prov_TRLoy value = 50 }
        }
        set_variable = { which = Tmp_0 which = Prov_TRLoy }
        subtract_variable = { which = Tmp_0 value = 50 }
        multiply_variable = { which = Tmp_0 value = 0.02 }
        multiply_variable = { which = Tmp_0 which = Class_ArmyTR }
        multiply_variable = { which = Tmp_0 value = 0.25 }

        change_variable = { which = $return$ which = Tmp_0 }
    }

    set_variable = { which = Tmp_0 which = Class_ArmyBU }
    multiply_variable = { which = Tmp_0 value = 0.5 }
    change_variable = { which = $return$ which = Tmp_0 }
	
    divide_variable = { which = $return$ which = Class_Total }
    multiply_variable = { which = $return$ value = 400 } # -4 unrest per 1% of population levied
	
	# Bureaucratic incomptence is equal to benign neglect, supressing rebellion
	# This only matters when theres an actual bureau neglecting them
	# AKA Unrest reduction from LA
    export_to_variable = { which = Tmp_0 value = local_autonomy }
    multiply_variable = { which = Tmp_0 value = 0.01 }
    multiply_variable = { which = Tmp_0 which = Prov_BUPow }
    multiply_variable = { which = Tmp_0 value = 0.01 }
    multiply_variable = { which = Tmp_0 value = 25 }
    change_variable = { which = $return$ which = Tmp_0 }

    if = {
        limit = {
            NOT = {
                check_variable = { which = $return$ value = 0 }
            }
        }
        set_variable = { which = $return$ value = 0 }
    }

    set_variable = { which = Tmp_0 value = 0 }
    set_variable = { which = Tmp_1 value = 0 }
}
