maint_Forestry_no = {
	cost = 0
	time =  0
	
	potential = {
		always = yes
	}
	
	build_trigger = {
		always = no
	}
	keep_trigger = {
		has_building = maint_Forestry_no
	}
	
	modifier = {
	}
	
	ai_will_do = {
		factor = 0
	}

	on_destroyed = {
		if = {
			limit = {
				NOT = {
					has_building = maint_Forestry_low
					has_building = maint_Forestry_med
					has_building = maint_Forestry_high
				}
			}
			add_building = maint_Forestry_no
		}
	}
	on_obsolete = {
	}
}
maint_Forestry_low = {
	cost = 0
	time = 0
	
	potential = {
		always = yes
	}
	
	build_trigger = {
		is_core = ROOT
		owner = { Tax_PropForestryCan = yes }
		
		NOT = {
			has_building = maint_Forestry_med
			has_building = maint_Forestry_high
		}
	}
	keep_trigger = {
		is_core = ROOT
		owner = { Tax_PropForestryCan = yes }
	}
	
	ai_will_do = {
		factor = 0
	}
	
	on_built = {
		remove_building = maint_Forestry_no
	}
	
	on_destroyed = {
		if = {
			limit = {
				NOT = {
					has_building = maint_Forestry_med
					has_building = maint_Forestry_high
				}
			}
			add_building = maint_Forestry_no
		}
	}
	on_obsolete = {
	}
}
maint_Forestry_med = {
	cost = 0
	time = 0
	
	potential = {
		always = yes
	}
	
	build_trigger = {
		is_core = ROOT
		owner = { Tax_PropForestryCan = yes }
		
		NOT = {
			has_building = maint_Forestry_high
		}
	}
	keep_trigger = {
		is_core = ROOT
		owner = { Tax_PropForestryCan = yes }
	}
	
	ai_will_do = {
		factor = 0
	}
	
	on_built = {
		remove_building = maint_Forestry_no
        remove_building = maint_Forestry_low
	}
	
	on_destroyed = {
		if = {
			limit = {
				NOT = {
					has_building = maint_Forestry_high
				}
			}
			add_building = maint_Forestry_low
		}
	}
	on_obsolete = {
	}
}
maint_Forestry_high = {
	cost = 0
	time = 0
	
	potential = {
		always = yes
	}
	
	build_trigger = {
		is_core = ROOT
		owner = { Tax_PropForestryCan = yes }
	}
	keep_trigger = {
		is_core = ROOT
		owner = { Tax_PropForestryCan = yes }
	}
	
	ai_will_do = {
		factor = 0
	}
	
	on_built = {
		remove_building = maint_Forestry_no
        remove_building = maint_Forestry_low
        remove_building = maint_Forestry_med
	}
	
	on_destroyed = {

	}

	}
