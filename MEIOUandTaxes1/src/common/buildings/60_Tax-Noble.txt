obli_noble_no = {
	cost = 0
	time =  0
	
	potential = {
		always = yes
	}
	
	build_trigger = {
		always = no
	}
	keep_trigger = {
		has_building = obli_noble_no
	}
	
	modifier = {
	}
	
	ai_will_do = {
		factor = 0
	}

	on_destroyed = {
		if = {
			limit = {
				NOT = {
					has_building = obli_noble_low
					has_building = obli_noble_med
					has_building = obli_noble_high
				}
			}
			add_building = obli_noble_no
		}
	}
	on_built = {
	}
	on_obsolete = {
	}
}
obli_noble_low = {
	cost = 0
	time = 0
	
	potential = {
		always = yes
	}
	
	build_trigger = {
		is_core = ROOT
		owner = {
			Tax_TaxNobleCan = yes
		}

		NOT = {
			has_building = obli_noble_med
			has_building = obli_noble_high
		}
	}
	keep_trigger = {
		is_core = ROOT
		owner = {
			Tax_TaxNobleCan = yes
		}
	}
	
	ai_will_do = {
		factor = 0
	}
	
	on_built = {
		remove_building = obli_noble_no
	}
	
	on_destroyed = {
		if = {
			limit = {
				NOT = {
					has_building = obli_noble_med
					has_building = obli_noble_high
				}
			}
			add_building = obli_noble_no
		}
	}
	on_obsolete = {
	}
}
