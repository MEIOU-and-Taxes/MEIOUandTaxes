give_Agriculture_10 = {
	cost = 10
	time = 0
	
	potential = {
		always = yes
	}
	
	build_trigger = {
		custom_trigger_tooltip = {
			check_variable = { which = Prov_BUPow value = 2.5 }
			tooltip = Prov_State_Reach_BuildingGive_tt
		}
		custom_trigger_tooltip = {
			check_variable = { which = Building_Agriculture value = 0.001 }
			tooltip = Prov_Unopened_BuildingGive_tt
		}	
		custom_trigger_tooltip = {
			NOT = { is_variable_equal = { which = Building_Agriculture which = Building_AgricultureMax } }
			tooltip = Prov_Full_BuildingGive_tt
		}			
		OR = {
			owner = {
				ai = no
			}
			owner = {
				overlord = {
					ai = no
				}
			}
		}
	}
	
	modifier = {
	}
	
	ai_will_do = {
		factor = 0
	}
	
	on_built = {
		hidden_effect = {
			remove_building = give_Agriculture_10

			change_variable = { which = Building_AgricultureWealth value = 10 }
		}
	}
	on_obsolete = {
	}
	on_destroyed = {
	}
}
