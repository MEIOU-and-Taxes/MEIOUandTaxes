obli_charter_no = {
	cost = 0
	time =  0
	
	potential = {
		always = yes
	}
	
	build_trigger = {
		always = no
	}
	keep_trigger = {
		has_building = obli_charter_no
	}
	
	modifier = {
	}
	
	ai_will_do = {
		factor = 0
	}

	on_destroyed = {
		if = {
			limit = {
				NOT = {
					has_building = obli_charter_low
					has_building = obli_charter_med
					has_building = obli_charter_high
				}
			}
			add_building = obli_charter_no
		}
	}
	on_built = {
	}
	on_obsolete = {
	}
}
obli_charter_low = {
	cost = 0
	time = 0
	
	potential = {
		always = yes
	}
	
	build_trigger = {
		is_core = ROOT
		owner = {
			Tax_TaxCharterCan = yes
		}

		NOT = {
			has_building = obli_charter_med
			has_building = obli_charter_high
		}
	}
	keep_trigger = {
		is_core = ROOT
		owner = {
			Tax_TaxCharterCan = yes
		}
	}
	
	ai_will_do = {
		factor = 0
	}
	
	on_built = {
		remove_building = obli_charter_no
	}
	
	on_destroyed = {
		if = {
			limit = {
				NOT = {
					has_building = obli_charter_med
					has_building = obli_charter_high
				}
			}
			add_building = obli_charter_no
		}
	}
	on_obsolete = {
	}
}
obli_charter_med = {
	cost = 0
	time = 0
	
	potential = {
		always = yes
	}
	
	build_trigger = {
		is_core = ROOT
		owner = {
			Tax_TaxCharterCan = yes
		}
		
		NOT = { 
			has_province_modifier = emergency_tax_relief 
		}

		NOT = {
			has_building = obli_charter_high
		}
	}
	keep_trigger = {
		is_core = ROOT
		owner = {
			Tax_TaxCharterCan = yes
		}
				
		NOT = { 
			has_province_modifier = emergency_tax_relief 
		}
	}
	
	ai_will_do = {
		factor = 0
	}
	
	on_built = {
		remove_building = obli_charter_no
        remove_building = obli_charter_low
	}
	
	on_destroyed = {
		if = {
			limit = {
				NOT = {
					has_building = obli_charter_high
				}
			}
			add_building = obli_charter_low
		}
	}
	on_obsolete = {
	}
}
obli_charter_high = {
	cost = 0
	time = 0
	
	potential = {
		always = yes
	}
	
	build_trigger = {
		is_core = ROOT
		owner = {
			Tax_TaxCharterCan = yes
		}		

		NOT = { 
			has_province_modifier = emergency_tax_relief 
		}

	}
	keep_trigger = {
		is_core = ROOT
		owner = {
			Tax_TaxCharterCan = yes
		}
		
		NOT = { 
			has_province_modifier = emergency_tax_relief 
		}
	}
	
	ai_will_do = {
		factor = 0
	}
	
	on_built = {
		remove_building = obli_charter_no
        remove_building = obli_charter_low
        remove_building = obli_charter_med
	}
	
	on_destroyed = {
		add_building = obli_charter_med
	}
	on_obsolete = {
	}
}
