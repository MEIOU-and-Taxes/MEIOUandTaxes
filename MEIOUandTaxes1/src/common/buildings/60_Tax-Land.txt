tax_land_no = {
	cost = 0
	time =  0
	
	potential = {
		always = yes
	}
	
	build_trigger = {
		always = no
	}
	keep_trigger = {
		has_building = tax_land_no
	}
	
	modifier = {
	}
	
	ai_will_do = {
		factor = 0
	}

	on_destroyed = {
		if = {
			limit = {
				NOT = {
					has_building = tax_land_low
					has_building = tax_land_med
					has_building = tax_land_high
				}
			}
			add_building = tax_land_no
		}
	}
	on_built = {
	}
	on_obsolete = {
	}
}
tax_land_low = {
	cost = 0
	time = 0
	
	potential = {
		always = yes
	}
	
	build_trigger = {
		owner = {
			Tax_TaxLandCan = yes
		}

		NOT = {
			has_building = tax_land_med
			has_building = tax_land_high
		}
	}
	keep_trigger = {
		is_core = ROOT
		owner = {
			Tax_TaxLandCan = yes
		}
	}
	
	ai_will_do = {
		factor = 0
	}
	
	on_built = {
		remove_building = tax_land_no
	}
	
	on_destroyed = {
		if = {
			limit = {
				NOT = {
					has_building = tax_land_med
					has_building = tax_land_high
				}
			}
			add_building = tax_land_no
		}
	}
	on_obsolete = {
	}
}
tax_land_med = {
	cost = 0
	time = 0
	
	potential = {
		always = yes
	}
	
	build_trigger = {
		owner = {
			Tax_TaxLandCan = yes
		}
		NOT = { 
			has_province_modifier = emergency_tax_relief 
		}

		NOT = {
			has_building = tax_land_high
		}
	}
	keep_trigger = {
		is_core = ROOT
		owner = {
			Tax_TaxLandCan = yes
		}
		NOT = { 
			has_province_modifier = emergency_tax_relief 
		}
	}
	
	ai_will_do = {
		factor = 0
	}
	
	on_built = {
		remove_building = tax_land_no
        remove_building = tax_land_low
	}
	
	on_destroyed = {
		if = {
			limit = {
				NOT = {
					has_building = tax_land_high
				}
			}
			add_building = tax_land_low
		}
	}
	on_obsolete = {
	}
}
tax_land_high = {
	cost = 0
	time = 0
	
	potential = {
		always = yes
	}
	
	build_trigger = {
		owner = {
			Tax_TaxLandCan = yes
		}
		NOT = { 
			has_province_modifier = emergency_tax_relief 
		}
	}
	keep_trigger = {
		is_core = ROOT
		owner = {
			Tax_TaxLandCan = yes
		}
		NOT = { 
			has_province_modifier = emergency_tax_relief 
		}
	}
	
	ai_will_do = {
		factor = 0
	}
	
	on_built = {
		remove_building = tax_land_no
        remove_building = tax_land_low
        remove_building = tax_land_med
	}
	
	on_destroyed = {
		add_building = tax_land_med
	}
	on_obsolete = {
	}
}
