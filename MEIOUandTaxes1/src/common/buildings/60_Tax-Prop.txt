tax_prop_no = {
	cost = 0
	time =  0
	
	potential = {
		always = yes
	}
	
	build_trigger = {
		always = no
	}
	keep_trigger = {
		has_building = tax_prop_no
	}
	
	modifier = {
	}
	
	ai_will_do = {
		factor = 0
	}

	on_destroyed = {
		if = {
			limit = {
				NOT = {
					has_building = tax_prop_low
					has_building = tax_prop_med
					has_building = tax_prop_high
				}
			}
			add_building = tax_prop_no
		}
	}
	on_built = {
	}
	on_obsolete = {
	}
}
tax_prop_low = {
	cost = 0
	time = 0
	
	potential = {
		always = yes
	}
	
	build_trigger = {
		is_core = ROOT
		owner = {
			Tax_TaxPropCan = yes
		}

		NOT = {
			has_building = tax_prop_med
			has_building = tax_prop_high
		}
	}
	keep_trigger = {
		is_core = ROOT
		owner = {
			Tax_TaxPropCan = yes
		}
	}
	
	ai_will_do = {
		factor = 0
	}
	
	on_built = {
		remove_building = tax_prop_no
	}
	
	on_destroyed = {
		if = {
			limit = {
				NOT = {
					has_building = tax_prop_med
					has_building = tax_prop_high
				}
			}
			add_building = tax_prop_no
		}
	}
	on_obsolete = {
	}
}
tax_prop_med = {
	cost = 0
	time = 0
	
	potential = {
		always = yes
	}
	
	build_trigger = {
		is_core = ROOT
		owner = {
			Tax_TaxPropCan = yes
		}

		NOT = {
			has_building = tax_prop_high
		}
		NOT = { 
			has_province_modifier = emergency_tax_relief 
		}
	}
	keep_trigger = {
		is_core = ROOT
		owner = {
			Tax_TaxPropCan = yes
		}
		NOT = { 
			has_province_modifier = emergency_tax_relief 
		}
	}
	
	ai_will_do = {
		factor = 0
	}
	
	on_built = {
		remove_building = tax_prop_no
        remove_building = tax_prop_low
	}
	
	on_destroyed = {
		if = {
			limit = {
				NOT = {
					has_building = tax_prop_high
				}
			}
			add_building = tax_prop_low
		}
	}
	on_obsolete = {
	}
}
tax_prop_high = {
	cost = 0
	time = 0
	
	potential = {
		always = yes
	}
	
	build_trigger = {
		is_core = ROOT
		owner = {
			Tax_TaxPropCan = yes
		}
		NOT = { 
			has_province_modifier = emergency_tax_relief 
		}
	}
	keep_trigger = {
		is_core = ROOT
		owner = {
			Tax_TaxPropCan = yes
		}
		NOT = { 
			has_province_modifier = emergency_tax_relief 
		}
	}
	
	ai_will_do = {
		factor = 0
	}
	
	on_built = {
		remove_building = tax_prop_no
        remove_building = tax_prop_low
        remove_building = tax_prop_med
	}
	
	on_destroyed = {
		add_building = tax_prop_med
	}
	on_obsolete = {
	}
}
