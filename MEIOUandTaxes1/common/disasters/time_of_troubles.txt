time_of_troubles = {
	potential = {
		NOT = {
			has_country_flag = time_of_troubles
		}
		dynasty = "Rurikovich"
		AND = {
			OR = {
				is_subject = no
				is_subject_of_type = tributary_state
				is_subject_of_type = close_tributary_state
			}
		}
		tag  = RUS
		OR = {
			AND = {
				exists = PLC
				PLC = { is_neighbor_of = ROOT }
			}
			AND = {
				exists = POL
				POL = { is_neighbor_of = ROOT }
			}
		}
	}
	can_start = {
		#current_age = age_of_reformation
		is_year = 1525
		government = monarchy
		has_any_disaster = no
		NOT = {
			stability = 0
		}
	}
	can_stop = {
		OR = {
			has_any_disaster = yes
			NOT = { government = monarchy }
			AND = {
				stability = 3
				legitimacy = 90
			}
		}
	}
	modifier = {
		global_unrest = 10
		stability_cost_modifier = 0.20
	}
	progress = {
		modifier = {
			factor = 0.5
			NOT = {
				stability = -2
			}
		}
		modifier = {
			factor = 0.5
			NOT = {
				stability = 0
			}
		}
		modifier = {
			factor = 0.5
			war_exhaustion = 8
		}
		modifier = {
			factor = 0.5
			war_exhaustion = 12
		}
		modifier = {
			factor = 0.5
			NOT = {
				legitimacy = 50
			}
		}
		modifier = {
			factor = 0.5 
			NOT = {
				legitimacy = 25
			}
		}
		modifier = {
			factor = 0.5 
			num_of_loans = 1
		}
		modifier = {
			factor = 0.5 
			num_of_loans = 4
		}
		modifier = {
			factor = 0.5 
			num_of_loans = 8
		}
		modifier = {
			factor = 1
			has_regency = yes
		} 
		modifier = {
			factor = 1
			has_heir = no
		}
	}
	can_end = {
		# NOT = { num_of_rebel_controlled_provinces = 1 } #changed to rebel armies, as it can take a very long time to get to one random province in the east of the realm
		NOT = { num_of_rebel_armies = 1 } 
		NOT = {	has_ruler_modifier = usurper_on_the_throne }
		is_at_war = no
		stability = 1
		had_country_flag = { #at least two years to give rebels and events time to unfold
			flag = time_of_troubles
			days = 730
		}
		capital_scope = {
			controlled_by = ROOT
		}
	}
	on_start = time_of_troubles.1
	on_end = time_of_troubles.10
	on_monthly = {
		events = {}
		random_events = {
			500 = 0
			100 = time_of_troubles.5
			100 = time_of_troubles.6
			100 = time_of_troubles.9
		}
	}
}
