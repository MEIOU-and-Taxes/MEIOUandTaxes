Rights_SyncTickHelper = {
	if = {
		limit = {
			NOT = { is_key_equal = { lhs = Rights_$Reform$ which = owner } }
		}
		if = {
			limit = {
				check_key = { lhs = Rights_$Reform$ which = owner }
			}
			subtract_key = { lhs = Rights_$Reform$ value = 1 }
		}
		else = {
			change_key = { lhs = Rights_$Reform$ value = 1 }
		}
	}
}
