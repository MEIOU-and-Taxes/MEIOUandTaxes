Pol_FamilyInit = {
	if = {
		limit = { 
			government = monarchy
			has_government_attribute = maintain_dynasty
			OR = {
				NOT = { religion_group = muslim }
				culture_group = tartar_group
				culture_group = old_turkic
				culture_group = altaic
				primary_culture = turkish
				primary_culture = osmanli
			}
			NOT = { religion_group = christian }
			NOT = { has_matching_religion = orthodox }
			NOT = { has_matching_religion = catholic }
			NOT = { culture_group = japanese_g }
		}
		add_government_reform = Pol_Family4
	}
	else_if = { 
		limit = {
			government = monarchy
			has_government_attribute = maintain_dynasty
			religion_group = muslim
			NOT = { culture_group = tartar_group }
			NOT = { culture_group = old_turkic }
			NOT = { culture_group = altaic }
			NOT = { primary_culture = turkish }
			NOT = { primary_culture = osmanli }
		}
		add_government_reform = Pol_Family3
	}
	else_if = {
		limit = {
			government = monarchy
			OR = {
				has_reform = Pol_Succession2 #Foreign
				#is_key_equal = { lhs = Pol_Succession value = 1 }
				religion_group = muslim
			}
			NOT = { primary_culture = turkish }
			NOT = { primary_culture = osmanli }
		}
		add_government_reform = Pol_Family2
	}
	else_if = {
		limit = {
			government = monarchy
			is_tribal = no
		}
		add_government_reform = Pol_Family1
	}
}

	#		5 = {
	#			change_key = { lhs = $FAC1$_Mood$FAC2$ value = 1 }
	#		}
	#	}
	#}

	### Relations Calculation
	if = {
		limit = {
			has_country_modifier = $FAC1$_Ally$FAC2$ # Allied case
		}
		if = {
			limit = {
				NOT = { check_key = { lhs = $FAC1$_Relation$FAC2$ value = 100 } }
			}
			change_key = { lhs = $FAC1$_Relation$FAC2$ value = 2 } # if below 100, recover 2
		}
		if = {
			limit = {
				check_key = { lhs = $FAC1$_Relation$FAC2$ value = 100.001 }
			}
			set_key = { lhs = $FAC1$_Relation$FAC2$ value = 100 } # if above 100 set upper limit
		}
	}
	else_if = {
		limit = {
			has_country_modifier = $FAC1$_Rival$FAC2$ # rival case
		}
		if = {
			limit = {
				check_key = { lhs = $FAC1$_Relation$FAC2$ value = -99.999 }
			}
			subtract_key = { lhs = $FAC1$_Relation$FAC2$ value = 2 } # if above -100 reduce by -2
		}
		if = {
			limit = {
				NOT = { check_key = { lhs = $FAC1$_Relation$FAC2$ value = -100 } }
			}
			set_key = { lhs = $FAC1$_Relation$FAC2$ value = -100 } # if below -100 set lower limit
		}
	}
	else = {
		set_key = { lhs = Pol_Tmp0 which = $FAC1$_Mood$FAC2$ } # Get Mood
		multiply_key = { lhs = Pol_Tmp0 value = 20 } # Multiply Mood by 20
		change_key = { lhs = Pol_Tmp0 which = Modi_FC$FAC1$$FAC2$ } # Add API Base Relationship
		multiply_key = { lhs = $FAC1$_Mood$FAC2$ value = 0.9 } # Make mood against state recover over time -> tweak number for balance
		
		[[DispRest]
		if = {
			limit = {
				always = $DispRest$
				ai = no
			}
			set_key = { lhs = $FAC1$_Relation$FAC2$_RestDisp which = Pol_Tmp0 }
		}
		]
		if = {
			limit = {
				check_key = { lhs = $FAC1$_Relation$FAC2$ which = Pol_Tmp0 } # if current relations higher then resting point
			}
			subtract_key = { lhs = $FAC1$_Relation$FAC2$ value = 2 } # subtract 2 per year
			if = {
				limit = {
					NOT = { check_key = { lhs = $FAC1$_Relation$FAC2$ which = Pol_Tmp0 } } 
				}
				set_key = { lhs = $FAC1$_Relation$FAC2$ which = Pol_Tmp0 } # if current relations now lower then resting point, set to resting point
			}
		}
		else_if = {
			limit = {
				NOT = { check_key = { lhs = $FAC1$_Relation$FAC2$ which = Pol_Tmp0 } } # if current relations lower then resting point
			}
			change_key = { lhs = $FAC1$_Relation$FAC2$ value = 2 } # add two relations per year
			if = {
				limit = {
					check_key = { lhs = $FAC1$_Relation$FAC2$ which = Pol_Tmp0 } # if current relations now above resting point, set to resting point
				}
				set_key = { lhs = $FAC1$_Relation$FAC2$ which = Pol_Tmp0 }
			}
		}
	}
	multiply_key = { lhs = $FAC1$_Relation$FAC2$ value = 0.001 }
	multiply_key = { lhs = $FAC1$_Relation$FAC2$ value = 1000 }
}

Pol_ChangeMood = {
	hidden_effect = {
		change_key = { lhs = $FAC1$_Mood$FAC2$ $type$ = $val$ }
		if = {
			limit = {
				NOT = { check_key = { lhs = $FAC1$_Mood$FAC2$ value = -5 } }
			}
			set_key = { lhs = $FAC1$_Mood$FAC2$ value = -5 }
		}
		else_if = {
			limit = {
				check_key = { lhs = $FAC1$_Mood$FAC2$ value = 5.001 }
			}
			set_key = { lhs = $FAC1$_Mood$FAC2$ value = 5 }
		}
		multiply_key = { lhs = $FAC1$_Mood$FAC2$ value = 0.001 }
		multiply_key = { lhs = $FAC1$_Mood$FAC2$ value = 1000 }
	}
	[[tooltipFAC1]
	if = {
		limit = {
			always = $tooltipFAC1$
		}
		custom_tooltip = $FAC1$_Mood$FAC2$_$val$
	}
	]
	[[tooltipFAC2]
	if = {
		limit = {
			always = $tooltipFAC2$
		}
		custom_tooltip = $FAC2$_Mood$FAC1$_$val$
	}
	]
}
